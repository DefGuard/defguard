42#[derive(Clone, Copy, Default)]
43pub enum ClaimsType {
44    #[default]
45    Auth,
46    Gateway,
47    YubiBridge,
48    DesktopClient,
49}
50
51/// Standard claims: https://www.iana.org/assignments/jwt/jwt.xhtml
52#[derive(Deserialize, Serialize)]
53pub struct Claims {
54    #[serde(skip_serializing, skip_deserializing)]
55    secret: String,
56    // issuer
57    pub iss: String,
58    // subject
59    pub sub: String,
60    // client identifier
61    pub client_id: String,
62    // expiration time
63    pub exp: u64,
64    // not before
65    pub nbf: u64,
66}
67
68impl Claims {
69    #[must_use]
70    pub fn new(claims_type: ClaimsType, sub: String, client_id: String, duration: u64) -> Self {
71        let now = SystemTime::now();
72        let exp = now
73            .checked_add(Duration::from_secs(duration))
74            .expect("valid time")
75            .duration_since(SystemTime::UNIX_EPOCH)
76            .expect("valid timestamp")
77            .as_secs();
78        let nbf = now
79            .duration_since(SystemTime::UNIX_EPOCH)
80            .expect("valid timestamp")
81            .as_secs();
82        Self {
83            secret: Self::get_secret(claims_type),
84            iss: JWT_ISSUER.to_string(),
85            sub,
86            client_id,
87            exp,
88            nbf,
89        }
90    }
91
92    fn get_secret(claims_type: ClaimsType) -> String {
93        let env_var = match claims_type {
94            ClaimsType::Auth | ClaimsType::DesktopClient => AUTH_SECRET_ENV,
95            ClaimsType::Gateway => GATEWAY_SECRET_ENV,
96            ClaimsType::YubiBridge => YUBIBRIDGE_SECRET_ENV,
97        };
98        env::var(env_var).unwrap_or_default()
99    }
100
101    /// Convert claims to JWT.
102    pub fn to_jwt(&self) -> Result<String, JWTError> {
103        encode(
104            &Header::default(),
105            self,
106            &EncodingKey::from_secret(self.secret.as_bytes()),
107        )
108    }
109
110    /// Verify JWT and, if successful, convert it to claims.
111    pub fn from_jwt(claims_type: ClaimsType, token: &str) -> Result<Self, JWTError> {
112        let secret = Self::get_secret(claims_type);
113        let mut validation = Validation::default();
114        validation.validate_nbf = true;
115        validation.set_issuer(&[JWT_ISSUER]);
116        validation.set_required_spec_claims(&["iss", "sub", "exp", "nbf"]);
117        decode::<Self>(
118            token,
119            &DecodingKey::from_secret(secret.as_bytes()),
120            &validation,
121        )
122        .map(|data| data.claims)
123    }
124}

126impl<S> FromRequestParts<S> for Session
127where
128    S: Send + Sync,
129    AppState: FromRef<S>,
130{
131    type Rejection = WebError;
132
133    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
134        let appstate = AppState::from_ref(state);
135
136        // first try to authenticate by API token if one is found in header
137        if is_enterprise_enabled() {
138            let maybe_auth_header: Option<TypedHeader<Authorization<Bearer>>> =
139                <TypedHeader<_> as OptionalFromRequestParts<S>>::from_request_parts(parts, state)
140                    .await
141                    .map_err(|err| {
142                        error!("Failed to extract optional auth header: {err}");
143                        WebError::Authorization("Invalid auth header".into())
144                    })?;
145            if let Some(header) = maybe_auth_header {
146                let token_string = header.token();
147                debug!("Trying to authorize request using API token: {token_string}");
148                return match ApiToken::try_find_by_auth_token(&appstate.pool, token_string).await {
149                    Ok(Some(api_token)) => {
150                        // create a dummy session and don't store it in the DB
151                        // since each request needs to be authorized anyway
152                        let ip_address = InsecureClientIp::from_request_parts(parts, state)
153                            .await
154                            .map_err(|err| {
155                            error!("Failed to get client IP: {err:?}");
156                            WebError::ClientIpError
157                        })?;
158                        Ok(Session::new(
159                            api_token.user_id,
160                            SessionState::ApiTokenVerified,
161                            ip_address.0.to_string(),
162                            None,
163                        ))
164                    }
165                    Ok(None) => Err(WebError::Authorization("Invalid API token".into())),
166                    Err(err) => Err(err.into()),
167                };
168            };
169        }
170
171        let Ok(cookies) = CookieJar::from_request_parts(parts, state).await;
172        if let Some(session_cookie) = cookies.get(SESSION_COOKIE_NAME) {
173            return {
174                match Session::find_by_id(&appstate.pool, session_cookie.value()).await {
175                    Ok(Some(session)) => {
176                        if session.expired() {
177                            let _result = session.delete(&appstate.pool).await;
178                            Err(WebError::Authorization("Session expired".into()))
179                        } else {
180                            Ok(session)
181                        }
182                    }
183                    Ok(None) => Err(WebError::Authorization("Session not found".into())),
184                    Err(err) => Err(err.into()),
185                }
186            };
187        }
188
189        Err(WebError::Authorization("Session is required".into()))
190    }
191}

193// Extension of base user session that contains user data fetched from database.
194// This represents a session for a user who completed the login process (including MFA, if enabled).
195pub struct SessionInfo {
196    pub session: Session,
197    pub user: User<Id>,
198    pub is_admin: bool,
199    groups: Vec<Group<Id>>,
200}
201
202impl SessionInfo {
203    #[must_use]
204    pub fn new(session: Session, user: User<Id>, is_admin: bool) -> Self {
205        Self {
206            session,
207            user,
208            is_admin,
209            groups: Vec::new(),
210        }
211    }
212
213    fn contains_any_group(&self, group_names: &[&str]) -> bool {
214        self.groups
215            .iter()
216            .any(|group| group_names.contains(&group.name.as_str()))
217    }
218}

220impl<S> FromRequestParts<S> for SessionInfo
221where
222    S: Send + Sync,
223    AppState: FromRef<S>,
224{
225    type Rejection = WebError;
226
227    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
228        let session = Session::from_request_parts(parts, state).await?;
229        let appstate = AppState::from_ref(state);
230        let user = User::find_by_id(&appstate.pool, session.user_id).await?;
231
232        if let Some(user) = user {
233            if user.mfa_enabled
234                && (session.state != SessionState::MultiFactorVerified
235                    && session.state != SessionState::ApiTokenVerified)
236            {
237                return Err(WebError::Authorization("MFA not verified".into()));
238            }
239            let Ok(groups) = user.member_of(&appstate.pool).await else {
240                return Err(WebError::DbError("cannot fetch groups".into()));
241            };
242            let is_admin = user.is_admin(&appstate.pool).await?;
243
244            // non-admin users are not allowed to use token auth
245            if !is_admin && session.state == SessionState::ApiTokenVerified {
246                return Err(WebError::Forbidden(
247                    "Token authentication is not allowed for normal users".into(),
248                ));
249            }
250
251            Ok(SessionInfo {
252                session,
253                user,
254                is_admin,
255                groups,
256            })
257        } else {
258            Err(WebError::Authorization("User not found".into()))
259        }
260    }
261}

263#[macro_export]
264macro_rules! role {
265    ($name:ident, $($permission:path)*) => {
266        pub struct $name;
267
268        impl<S> FromRequestParts<S> for $name
269        where
270            S: Send + Sync,
271            AppState: FromRef<S>,
272        {
273            type Rejection = WebError;
274
275            async fn from_request_parts(
276                parts: &mut Parts,
277                state: &S,
278            ) -> Result<Self, Self::Rejection> {
279                let session_info = SessionInfo::from_request_parts(parts, state).await?;
280                if !session_info.user.is_active {
281                    return Err(WebError::Forbidden("user is disabled".into()));
282                }
283                let appstate = AppState::from_ref(state);
284                $(
285                let groups_with_permission = Group::find_by_permission(
286                    &appstate.pool,
287                    $permission,
288                ).await?;
289                let group_names = groups_with_permission.iter().map(|group| group.name.as_str()).collect::<Vec<_>>();
290                if session_info.contains_any_group(&group_names) {
291                    return Ok(Self {});
292                }
293                )*
294                Err(WebError::Forbidden("access denied".into()))
295            }
296        }
297    };
298}

302// User authenticated by a valid access token
303pub struct AccessUserInfo(pub(crate) User<Id>);
304
305impl<S> FromRequestParts<S> for AccessUserInfo
306where
307    S: Send + Sync,
308    AppState: FromRef<S>,
309{
310    type Rejection = WebError;
311
312    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
313        let appstate = AppState::from_ref(state);
314        if let Some(token) = parts.headers.get(AUTHORIZATION).and_then(|value| {
315            if let Ok(value) = value.to_str() {
316                if value.to_lowercase().starts_with("bearer ") {
317                    value.get(7..)
318                } else {
319                    None
320                }
321            } else {
322                None
323            }
324        }) {
325            // TODO: #[cfg(feature = "openid")]
326            match OAuth2Token::find_access_token(&appstate.pool, token).await {
327                Ok(Some(oauth2token)) => {
328                    match OAuth2AuthorizedApp::find_by_id(
329                        &appstate.pool,
330                        oauth2token.oauth2authorizedapp_id,
331                    )
332                    .await
333                    {
334                        Ok(Some(authorized_app)) => {
335                            if let Ok(Some(user)) =
336                                User::find_by_id(&appstate.pool, authorized_app.user_id).await
337                            {
338                                return Ok(AccessUserInfo(user));
339                            }
340                        }
341                        Ok(None) => {
342                            return Err(WebError::Authorization("Authorized app not found".into()));
343                        }
344
345                        Err(err) => {
346                            return Err(err.into());
347                        }
348                    }
349                }
350                Ok(None) => {
351                    return Err(WebError::Authorization("Invalid token".into()));
352                }
353                Err(err) => {
354                    return Err(err.into());
355                }
356            }
357        }
358
359        Err(WebError::Authorization("Invalid session".into()))
360    }
