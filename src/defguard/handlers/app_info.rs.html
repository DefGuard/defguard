16#[derive(Serialize)]
17struct LicenseInfo {
18    /// Whether the enterprise features are enabled.
19    enterprise: bool,
20    /// Which limits are exceeded.
21    limits_exceeded: LimitsExceeded,
22    /// Is any of the limits exceeded.
23    any_limit_exceeded: bool,
24    /// Whether the enterprise features are used for free.
25    is_enterprise_free: bool,
26}

28#[derive(Serialize)]
29struct LdapInfo {

32    /// Whether AD is used
33    ad: bool,
34}

36/// Additional information about core state.
37#[derive(Serialize)]
38pub struct AppInfo {
39    version: String,
40    network_present: bool,
41    smtp_enabled: bool,
42    license_info: LicenseInfo,
43    ldap_info: LdapInfo,
44}
45
46pub(crate) async fn get_app_info(
47    State(appstate): State<AppState>,
48    _session: SessionInfo,
49) -> ApiResult {
50    let networks = WireguardNetwork::all(&appstate.pool).await?;
51    let settings = Settings::get_current_settings();
52    let enterprise = is_enterprise_enabled();
53    let license = get_cached_license();
54    let counts = get_counts();
55    let limits_exceeded = counts.get_exceeded_limits(license.as_ref());
56    let any_limit_exceeded = limits_exceeded.any();
57    let res = AppInfo {
58        network_present: !networks.is_empty(),
59        smtp_enabled: settings.smtp_configured(),
60        version: VERSION.into(),
61        license_info: LicenseInfo {
62            enterprise,
63            limits_exceeded,
64            any_limit_exceeded,
65            is_enterprise_free: is_enterprise_free(),
66        },
67        ldap_info: LdapInfo {
68            enabled: settings.ldap_enabled,
69            ad: settings.ldap_uses_ad,
70        },
71    };
72
73    Ok(ApiResponse::new(json!(res), StatusCode::OK))
74}
