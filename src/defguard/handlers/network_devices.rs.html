49impl NetworkDeviceInfo {
50    async fn from_device(
51        device: Device<Id>,
52        transaction: &mut PgConnection,
53    ) -> Result<Self, WebError> {
54        let network = device
55            .find_network_device_networks(&mut *transaction)
56            .await?
57            .pop()
58            .ok_or(WebError::ObjectNotFound(format!(
59                "Failed to find the network with which the network device {} is associated",
60                device.name
61            )))?;
62        let wireguard_device =
63            WireguardNetworkDevice::find(&mut *transaction, device.id, network.id)
64                .await?
65                .ok_or(WebError::ObjectNotFound(format!(
66                    "Failed to find network device {} network information in network {}",
67                    device.name, network.name
68                )))?;
69        let added_by = device.get_owner(&mut *transaction).await?;
70        let net_addr = network
71            .address
72            .first()
73            .ok_or(WebError::ObjectNotFound(format!(
74                "Failed to find the network address for network {}",
75                network.name
76            )))?;
77        let split_ip = split_ip(&wireguard_device.wireguard_ip, net_addr);
78        Ok(NetworkDeviceInfo {
79            id: device.id,
80            name: device.name,
81            assigned_ip: wireguard_device.wireguard_ip,
82            description: device.description,
83            added_by: added_by.username,
84            added_date: device.created,
85            wireguard_pubkey: device.wireguard_pubkey,
86            location: NetworkDeviceLocation {
87                id: wireguard_device.wireguard_network_id,
88                name: network.name,
89            },
90            configured: device.configured,
91            split_ip,
92        })
93    }
94}

180    transaction.commit().await?;
181
182    info!("Listed {} network devices", devices_response.len());
183    Ok(ApiResponse {
184        json: json!(devices_response),
185        status: StatusCode::OK,

204/// Checks if the IP address falls into the range of the network
205/// and if it is not already assigned to another device.
206async fn check_ip(
207    ip_addr: IpAddr,
208    network: &WireguardNetwork<Id>,
209    transaction: &mut PgConnection,
210) -> Result<(), WebError> {
211    if let Some(network_address) = network.address.first() {
212        if !network_address.contains(ip_addr) {
213            return Err(WebError::BadRequest(format!(
214                "Provided IP address {ip_addr} is not in the network ({}) range {network_address}",
215                network.name,
216            )));
217        }
218        if ip_addr == network_address.network() || ip_addr == network_address.broadcast() {
219            return Err(WebError::BadRequest(format!(
220                "Provided IP address {ip_addr} is network or broadcast address of network {}",
221                network.name
222            )));
223        }
224        if ip_addr == network_address.ip() {
225            return Err(WebError::BadRequest(format!(
226                "Provided IP address {ip_addr} may overlap with the network's gateway IP in network {}",
227                network.name
228            )));
229        }
230
231        let device = Device::find_by_ip(transaction, ip_addr, network.id).await?;
232        if let Some(device) = device {
233            return Err(WebError::BadRequest(format!(
234                "Provided IP address {ip_addr} is already assigned to device {} in network {}",
235                device.name, network.name
236            )));
237        }
238    }
239
240    Ok(())
241}

248pub(crate) async fn check_ip_availability(
249    _admin_role: AdminRole,
250    Path(network_id): Path<i64>,
251    State(appstate): State<AppState>,
252    Json(ip): Json<IpAvailabilityCheck>,
253) -> ApiResult {
254    let mut transaction = appstate.pool.begin().await?;
255    let network = WireguardNetwork::find_by_id(&appstate.pool, network_id)
256        .await?
257        .ok_or_else(|| {
258            error!(
259                "Failed to check IP availability for network with ID {network_id}, network not found",
260
261            );
262            WebError::BadRequest("Failed to check IP availability, network not found".into())
263        })?;
264
265    let Ok(ip) = IpAddr::from_str(&ip.ip) else {
266        warn!(
267            "Failed to check IP availability for network with ID {network_id}, invalid IP address",
268        );
269        return Ok(ApiResponse {
270            json: json!({
271                "available": false,
272                "valid": false,
273            }),
274            status: StatusCode::OK,
275        });
276    };
277
278    if let Some(network_address) = network.address.first() {
279        if !network_address.contains(ip) {
280            warn!(
281                "Provided device IP address is not in the network ({}) range {network_address}",
282                network.name
283            );
284            return Ok(ApiResponse {
285                json: json!({
286                    "available": false,
287                    "valid": false,
288                }),
289                status: StatusCode::OK,
290            });
291        }
292        if ip == network_address.network() || ip == network_address.broadcast() {
293            warn!(
294                "Provided device IP address is network or broadcast address of network {}",
295                network.name
296            );
297            return Ok(ApiResponse {
298                json: json!({
299                    "available": false,
300                    "valid": true,
301                }),
302                status: StatusCode::OK,
303            });
304        }
305        if ip == network_address.ip() {
306            warn!(
307                "Provided device IP address may overlap with the gateway's IP address on network {}",
308                network.name
309            );
310            return Ok(ApiResponse {
311                json: json!({
312                    "available": false,
313                    "valid": true,
314                }),
315                status: StatusCode::OK,
316            });
317        }
318    }
319
320    if let Some(device) = Device::find_by_ip(&mut *transaction, ip, network.id).await? {
321        warn!(
322            "Provided device IP is already assigned to device {} in network {}",
323            device.name, network.name
324        );
325        Ok(ApiResponse {
326            json: json!({
327                "available": false,
328                "valid": true,
329            }),
330            status: StatusCode::OK,
331        })
332    } else {
333        Ok(ApiResponse {
334            json: json!({
335                "available": true,
336                "valid": true,
337            }),
338            status: StatusCode::OK,
339        })
340    }
341}

343pub(crate) async fn find_available_ip(
344    _admin_role: AdminRole,
345    Path(network_id): Path<i64>,
346    State(appstate): State<AppState>,
347) -> ApiResult {
348    let network = WireguardNetwork::find_by_id(&appstate.pool, network_id)
349        .await?
350        .ok_or_else(|| {
351            error!(
352                "Failed to find available IP for network with ID {}",
353                network_id
354            );
355            WebError::BadRequest("Failed to find available IP, network not found".to_string())
356        })?;
357
358    let mut transaction = appstate.pool.begin().await?;
359    if let Some(network_address) = network.address.first() {
360        let net_ip = network_address.ip();
361        let net_network = network_address.network();
362        let net_broadcast = network_address.broadcast();
363        for ip in network_address {
364            if ip == net_ip || ip == net_network || ip == net_broadcast {
365                continue;
366            }
367
368            // Break loop if IP is unassigned and return network device
369            if Device::find_by_ip(&mut *transaction, ip, network.id)
370                .await?
371                .is_none()
372            {
373                let split_ip = split_ip(&ip, network_address);
374                transaction.commit().await?;
375                return Ok(ApiResponse {
376                    json: json!(split_ip),
377                    status: StatusCode::OK,
378                });
379            }
380        }
381    }
382
383    Ok(ApiResponse {
384        json: json!({}),
385        status: StatusCode::NOT_FOUND,
386    })
387}

557pub(crate) async fn add_network_device(
558    _admin_role: AdminRole,
559    session: SessionInfo,
560    State(appstate): State<AppState>,
561    Json(add_network_device): Json<AddNetworkDevice>,
562) -> ApiResult {
563    let device_name = add_network_device.name.clone();
564    debug!(
565        "User {} adding network device {device_name} in location {}.",
566        session.user.username, add_network_device.location_id
567    );
568
569    let user = session.user;
570    let network = WireguardNetwork::find_by_id(&appstate.pool, add_network_device.location_id)
571        .await?
572        .ok_or_else(|| {
573            error!(
574                "Failed to add device {device_name}, network with ID {} not found",
575                add_network_device.location_id
576            );
577            WebError::BadRequest("Failed to add device, network not found".to_string())
578        })?;
579
580    Device::validate_pubkey(&add_network_device.wireguard_pubkey)
581        .map_err(WebError::PubkeyValidation)?;
582
583    // Make sure there is no device with the same pubkey, such state may lead to unexpected issues
584    if Device::find_by_pubkey(&appstate.pool, &add_network_device.wireguard_pubkey)
585        .await?
586        .is_some()
587    {
588        return Err(WebError::PubkeyExists(format!(
589            "Failed to add device {device_name}, identical pubkey ({}) already exists",
590            add_network_device.wireguard_pubkey
591        )));
592    }
593
594    let mut transaction = appstate.pool.begin().await?;
595    let device = Device::new(
596        add_network_device.name,
597        add_network_device.wireguard_pubkey,
598        user.id,
599        DeviceType::Network,
600        add_network_device.description,
601        true,
602    )
603    .save(&mut *transaction)
604    .await?;
605
606    let ip: IpAddr = add_network_device.assigned_ip.parse().map_err(|e| {
607        error!("Failed to add network device {device_name}, invalid IP address: {e}");
608        WebError::BadRequest("Invalid IP address".to_string())
609    })?;
610    check_ip(ip, &network, &mut transaction).await?;
611
612    let (network_info, config) = device
613        .add_to_network(&network, ip, &mut transaction)
614        .await?;
615
616    appstate.send_wireguard_event(GatewayEvent::DeviceCreated(DeviceInfo {
617        device: device.clone(),
618        network_info: vec![network_info.clone()],
619    }));
620
621    update_counts(&mut *transaction).await?;
622
623    // send firewall update event if ACLs & enterprise features are enabled
624    if let Some(firewall_config) = network.try_get_firewall_config(&mut transaction).await? {
625        appstate.send_wireguard_event(GatewayEvent::FirewallConfigChanged(
626            network.id,
627            firewall_config,
628        ));
629    }
630
631    let template_locations = vec![TemplateLocation {
632        name: config.network_name.clone(),
633        assigned_ip: config.address.to_string(),
634    }];
635
636    send_new_device_added_email(
637        &device.name,
638        &device.wireguard_pubkey,
639        &template_locations,
640        &user.email,
641        &appstate.mail_tx,
642        Some(session.session.ip_address.as_str()),
643        session.session.device_info.clone().as_deref(),
644    )?;
645
646    info!(
647        "User {} added a new network device {device_name}.",
648        user.username
649    );
650
651    let result = AddNetworkDeviceResult {
652        config,
653        device: NetworkDeviceInfo::from_device(device, &mut transaction).await?,
654    };
655
656    transaction.commit().await?;
657
658    Ok(ApiResponse {
659        json: json!(result),
660        status: StatusCode::CREATED,
661    })
662}

671pub async fn modify_network_device(
672    _admin_role: AdminRole,
673    session: SessionInfo,
674    Path(device_id): Path<i64>,
675    State(appstate): State<AppState>,
676    Json(data): Json<ModifyNetworkDevice>,
677) -> ApiResult {
678    debug!("User {} updating device {device_id}", session.user.username);
679    let mut transaction = appstate.pool.begin().await?;
680    let mut device = Device::find_by_id(&mut *transaction, device_id)
681        .await?
682        .ok_or_else(|| {
683            error!("Failed to update device {device_id}, device not found");
684            WebError::ObjectNotFound(format!("Device {device_id} not found"))
685        })?;
686    let device_network = device
687        .find_network_device_networks(&mut *transaction)
688        .await?
689        .pop()
690        .ok_or_else(|| {
691            error!("Failed to update device {device_id}, device not found in any network");
692            WebError::ObjectNotFound(format!("Device {device_id} not found in any network"))
693        })?;
694    let mut wireguard_network_device =
695        WireguardNetworkDevice::find(&mut *transaction, device.id, device_network.id)
696            .await?
697            .ok_or_else(|| {
698                error!("Failed to update device {device_id}, device not found in any network");
699                WebError::ObjectNotFound(format!("Device {device_id} not found in any network"))
700            })?;
701    let new_ip = IpAddr::from_str(&data.assigned_ip).map_err(|e| {
702        WebError::BadRequest(format!(
703            "Failed to update device {device_id}, invalid IP address: {e}"
704        ))
705    })?;
706
707    device.name = data.name;
708    device.description = data.description;
709    device.save(&mut *transaction).await?;
710
711    // IP address has changed, so remove device from network and add it again with new IP address.
712    if new_ip != wireguard_network_device.wireguard_ip {
713        check_ip(new_ip, &device_network, &mut transaction).await?;
714        wireguard_network_device.wireguard_ip = new_ip;
715        wireguard_network_device.update(&mut *transaction).await?;
716        let device_info = DeviceInfo::from_device(&mut *transaction, device.clone()).await?;
717        appstate.send_wireguard_event(GatewayEvent::DeviceModified(device_info));
718
719        // send firewall update event if ACLs are enabled
720        if device_network.acl_enabled {
721            if let Some(firewall_config) = device_network
722                .try_get_firewall_config(&mut transaction)
723                .await?
724            {
725                appstate.send_wireguard_event(GatewayEvent::FirewallConfigChanged(
726                    device_network.id,
727                    firewall_config,
728                ));
729            }
730        }
731
732        info!(
733            "User {} changed IP address of network device {} from {} to {new_ip} in network {}",
734            session.user.username,
735            device.name,
736            wireguard_network_device.wireguard_ip,
737            device_network.name
738        );
739    }
740
741    let network_device_info = NetworkDeviceInfo::from_device(device, &mut transaction).await?;
742    transaction.commit().await?;
743
744    Ok(ApiResponse {
745        json: json!(network_device_info),
746        status: StatusCode::OK,
747    })
748}

750#[derive(Debug, Serialize)]
751struct SplitIP {
752    network_part: String,
753    modifiable_part: String,
754    network_prefix: String,
755    ip: String,
756}
757
758/// Splits the IP address (IPv4 or IPv6) into three parts: network part, modifiable part and prefix
759/// The network part is the part that can't be changed by the user.
760/// This is to display an IP address in the UI like this: 192.168.(1.1)/16, where the part in the parenthesis can be changed by the user.
761// The algorithm works as follows:
762// 1. Get the network address, last address and IP address segments, e.g. 192.1.1.1 would be [192, 1, 1, 1]
763// 2. Iterate over the segments and compare the last address and network segments, as long as the current segments are equal, append the segment to the network part.
764//    If they are not equal, we found the first modifiable segment (one of the segments of an address that may change between hosts in the same network),
765//    append the rest of the segments to the modifiable part.
766// 3. Join the segments with the delimiter and return the network part, modifiable part and the network prefix
767fn split_ip(ip: &IpAddr, network: &IpNetwork) -> SplitIP {
768    let network_addr = network.network();
769    let network_prefix = network.prefix();
770
771    let ip_segments = match ip {
772        IpAddr::V4(ip) => ip.octets().iter().map(|x| u16::from(*x)).collect(),
773        IpAddr::V6(ip) => ip.segments().to_vec(),
774    };
775
776    let last_addr_segments = match network {
777        IpNetwork::V4(net) => {
778            let last_ip = u32::from(net.ip()) | (!u32::from(net.mask()));
779            let last_ip: Ipv4Addr = last_ip.into();
780            last_ip.octets().iter().map(|x| u16::from(*x)).collect()
781        }
782        IpNetwork::V6(net) => {
783            let last_ip = u128::from(net.ip()) | (!u128::from(net.mask()));
784            let last_ip: Ipv6Addr = last_ip.into();
785            last_ip.segments().to_vec()
786        }
787    };
788
789    let network_segments = match network_addr {
790        IpAddr::V4(ip) => ip.octets().iter().map(|x| u16::from(*x)).collect(),
791        IpAddr::V6(ip) => ip.segments().to_vec(),
792    };
793
794    let mut network_part = String::new();
795    let mut modifiable_part = String::new();
796    let delimiter = if ip.is_ipv4() { "." } else { ":" };
797    let formatter = |x: &u16| {
798        if ip.is_ipv4() {
799            x.to_string()
800        } else {
801            format!("{x:04x}")
802        }
803    };
804
805    for (i, ((last_addr_segment, network_segment), ip_segment)) in last_addr_segments
806        .iter()
807        .zip(network_segments.iter())
808        .zip(ip_segments.iter())
809        .enumerate()
810    {
811        if last_addr_segment != network_segment {
812            let parts = ip_segments.split_at(i).1;
813            let joined = parts
814                .iter()
815                .map(formatter)
816                .collect::<Vec<String>>()
817                .join(delimiter);
818            modifiable_part.push_str(&joined);
819            break;
820        }
821        let formatted = formatter(ip_segment);
822        network_part.push_str(&format!("{formatted}{delimiter}"));
823    }
824
825    SplitIP {
826        ip: ip.to_string(),
827        network_part,
828        modifiable_part,
829        network_prefix: network_prefix.to_string(),
830    }
831}

834mod test {
835    use super::*;
836
837    #[test]
838    fn test_ip_splitter() {
839        let net = split_ip(
840            &IpAddr::from_str("192.168.3.1").unwrap(),
841            &IpNetwork::from_str("192.168.3.1/30").unwrap(),
842        );
843
844        assert_eq!(net.network_part, "192.168.3.");
845        assert_eq!(net.modifiable_part, "1");
846        assert_eq!(net.network_prefix, "30");
847

850            &IpNetwork::from_str("192.168.3.1/24").unwrap(),
851        );
852
853        assert_eq!(net.network_part, "192.168.5.");
854        assert_eq!(net.modifiable_part, "7");
855        assert_eq!(net.network_prefix, "24");
856
857        let net = split_ip(
858            &IpAddr::from_str("2001:0db8:85a3::8a2e:0370:7334").unwrap(),
859            &IpNetwork::from_str("2001:0db8:85a3::8a2e:0370:7334/64").unwrap(),
860        );
861
862        assert_eq!(net.network_part, "2001:0db8:85a3:0000:");
863        assert_eq!(net.modifiable_part, "0000:8a2e:0370:7334");
864        assert_eq!(net.network_prefix, "64");
865
866        let net = split_ip(
867            &IpAddr::from_str("2001:0db8::0010:8a2e:0370:aaaa").unwrap(),
868            &IpNetwork::from_str("2001:db8::10:8a2e:370:aaa8/125").unwrap(),
869        );
870
871        assert_eq!(net.network_part, "2001:0db8:0000:0000:0010:8a2e:0370:");
872        assert_eq!(net.modifiable_part, "aaaa");
873        assert_eq!(net.network_prefix, "125");
874    }
