36/// Verify the given username
37///
38/// To enable LDAP sync usernames need to avoid reserved characters.
39/// Username requirements:
40/// - 1 - 64 characters long
41/// - lowercase or uppercase latin alphabet letters (A-Z, a-z)
42/// - digits (0-9)
43/// - starts with non-special character
44/// - special characters: . - _
45/// - no whitespaces
46pub fn check_username(username: &str) -> Result<(), WebError> {
47    // check length
48    let length = username.len();
49    if !(1..64).contains(&length) {
50        return Err(WebError::Serialization(format!(
51            "Username ({username}) has incorrect length"
52        )));
53    }
54
55    // check first character is a letter or digit
56    if let Some(first_char) = username.chars().next() {
57        if !first_char.is_ascii_alphanumeric() {
58            return Err(WebError::Serialization(
59                "Username must not start with a special character".into(),
60            ));
61        }
62    }
63
64    // check if username contains only valid characters
65    if !username
66        .chars()
67        .all(|c| c.is_ascii_alphanumeric() || c == '.' || c == '-' || c == '_')
68    {
69        return Err(WebError::Serialization(
70            "Username contains invalid characters".into(),
71        ));
72    }
73
74    Ok(())
75}

77pub(crate) fn check_password_strength(password: &str) -> Result<(), WebError> {
78    if !(8..=128).contains(&password.len()) {
79        return Err(WebError::Serialization("Incorrect password length".into()));
80    }
81    if !password.chars().any(|c| c.is_ascii_punctuation()) {
82        return Err(WebError::Serialization(
83            "No special characters in password".into(),
84        ));
85    }
86    if !password.chars().any(|c| c.is_ascii_digit()) {
87        return Err(WebError::Serialization("No numbers in password".into()));
88    }
89    if !password.chars().any(|c| c.is_ascii_lowercase()) {
90        return Err(WebError::Serialization(
91            "No lowercase characters in password".into(),
92        ));
93    }
94    if !password.chars().any(|c| c.is_ascii_uppercase()) {
95        return Err(WebError::Serialization(
96            "No uppercase characters in password".into(),
97        ));
98    }
99    Ok(())
100}

275pub async fn add_user(
276    _role: AdminRole,
277    session: SessionInfo,
278    State(appstate): State<AppState>,
279    Json(user_data): Json<AddUserData>,
280) -> ApiResult {
281    let username = user_data.username.clone();
282    debug!("User {} adding user {username}", session.user.username);
283
284    // check username
285    if let Err(err) = check_username(&username) {
286        debug!("Username {username} rejected: {err}");
287        return Ok(ApiResponse {
288            json: json!({}),
289            status: StatusCode::BAD_REQUEST,
290        });
291    }
292    // check if email doesn't already exist
293    if User::find_by_email(&appstate.pool, &user_data.email)
294        .await?
295        .is_some()
296    {
297        debug!("User with email {} already exists", user_data.email);
298        return Ok(ApiResponse {
299            json: json!({}),
300            status: StatusCode::BAD_REQUEST,
301        });
302    }
303    let password = match &user_data.password {
304        Some(password) => {
305            // check password strength
306            if let Err(err) = check_password_strength(password) {
307                debug!("Password not strong enough: {err}");
308                return Ok(ApiResponse {
309                    json: json!({}),
310                    status: StatusCode::BAD_REQUEST,
311                });
312            }
313            Some(password.as_str())
314        }
315        None => None,
316    };
317
318    // create new user
319    let mut user = User::new(
320        user_data.username,
321        password,
322        user_data.last_name,
323        user_data.first_name,
324        user_data.email,
325        user_data.phone,
326    )
327    .save(&appstate.pool)
328    .await?;
329    update_counts(&appstate.pool).await?;
330
331    if let Some(password) = user_data.password {
332        ldap_add_user(&mut user, Some(&password), &appstate.pool).await;
333    }
334
335    let user_info = UserInfo::from_user(&appstate.pool, &user).await?;
336    appstate.trigger_action(AppEvent::UserCreated(user_info.clone()));
337    info!("User {} added user {username}", session.user.username);
338    if !user_info.enrolled {
339        warn!("User {username} hasn't been enrolled yet. Please proceed with enrollment.");
340    };
341    Ok(ApiResponse {
342        json: json!(&user_info),
343        status: StatusCode::CREATED,
344    })
345}

375pub async fn start_enrollment(
376    _role: AdminRole,
377    session: SessionInfo,
378    State(appstate): State<AppState>,
379    Path(username): Path<String>,
380    Json(data): Json<StartEnrollmentRequest>,
381) -> ApiResult {
382    debug!(
383        "User {} has started a new enrollment request.",
384        session.user.username
385    );
386
387    // validate request
388    if data.send_enrollment_notification && data.email.is_none() {
389        error!(
390            "Email notification is enabled for user {}, but email was not provided",
391            session.user.username
392        );
393        return Err(WebError::BadRequest(
394            "Email notification is enabled, but email was not provided".into(),
395        ));
396    }
397
398    debug!(
399        "Search for the user {} in database to get started with enrollment process.",
400        username
401    );
402    let Some(user) = User::find_by_username(&appstate.pool, &username).await? else {
403        error!("User {username} couldn't be found, enrollment aborted");
404        return Err(WebError::ObjectNotFound(format!(
405            "user {username} not found"
406        )));
407    };
408
409    debug!("Create a new database transaction to save a new enrollment token into the database.");
410    let mut transaction = appstate.pool.begin().await?;
411
412    let config = server_config();
413    let enrollment_token = user
414        .start_enrollment(
415            &mut transaction,
416            &session.user,
417            data.email,
418            config.enrollment_token_timeout.as_secs(),
419            config.enrollment_url.clone(),
420            data.send_enrollment_notification,
421            appstate.mail_tx.clone(),
422        )
423        .await?;
424
425    debug!("Try to commit transaction to save the enrollment token into the databse.");
426    transaction.commit().await?;
427    debug!("Transaction committed.");
428
429    info!(
430        "The enrollment process for {} has ended with success.",
431        session.user.username
432    );
433    debug!(
434        "Enrollment token {}, enrollment url {}",
435        enrollment_token,
436        config.enrollment_url.to_string()
437    );
438
439    Ok(ApiResponse {
440        json: json!({"enrollment_token": enrollment_token, "enrollment_url": config.enrollment_url.to_string()}),
441        status: StatusCode::CREATED,
442    })
443}

616pub async fn modify_user(
617    session: SessionInfo,
618    State(appstate): State<AppState>,
619    Path(username): Path<String>,
620    Json(mut user_info): Json<UserInfo>,
621) -> ApiResult {
622    debug!("User {} updating user {username}", session.user.username);
623    let mut user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
624    let old_username = user.username.clone();
625    if let Err(err) = check_username(&user_info.username) {
626        debug!("Username {} rejected: {err}", user_info.username);
627        return Ok(ApiResponse {
628            json: json!({}),
629            status: StatusCode::BAD_REQUEST,
630        });
631    }
632
633    let status_changing = user_info.is_active != user.is_active;
634
635    let mut transaction = appstate.pool.begin().await?;
636    let ldap_sync_allowed = user.ldap_sync_allowed(&mut *transaction).await?;
637
638    // remove authorized apps if needed
639    let request_app_ids: Vec<i64> = user_info
640        .authorized_apps
641        .iter()
642        .map(|app| app.oauth2client_id)
643        .collect();
644    let db_apps = user.oauth2authorizedapps(&mut *transaction).await?;
645    let removed_apps: Vec<i64> = db_apps
646        .iter()
647        .filter(|app| !request_app_ids.contains(&app.oauth2client_id))
648        .map(|app| app.oauth2client_id)
649        .collect();
650    if !removed_apps.is_empty() {
651        user.remove_oauth2_authorized_apps(&mut *transaction, &removed_apps)
652            .await?;
653    }
654    let mut group_diff = GroupDiff::default();
655    if session.is_admin {
656        // prevent admin from disabling himself
657        if session.user.username == username && !user_info.is_active {
658            debug!("Admin {username} attempted to disable himself");
659            return Ok(ApiResponse {
660                json: json!({}),
661                status: StatusCode::BAD_REQUEST,
662            });
663        }
664
665        // update VPN gateway config if user status or groups have changed
666        group_diff = user_info
667            .handle_user_groups(&mut transaction, &mut user)
668            .await?;
669        if group_diff.changed()
670            || user_info
671                .handle_status_change(&mut transaction, &mut user)
672                .await?
673        {
674            debug!(
675                "User {} changed {username} groups or status, syncing allowed network devices.",
676                session.user.username
677            );
678            user.sync_allowed_devices(&mut transaction, &appstate.wireguard_tx)
679                .await?;
680        };
681        user_info.into_user_all_fields(&mut user)?;
682    } else {
683        user_info.into_user_safe_fields(&mut user)?;
684    }
685    user.save(&mut *transaction).await?;
686    transaction.commit().await?;
687    let user_info = UserInfo::from_user(&appstate.pool, &user).await?;
688
689    if ldap_sync_allowed {
690        ldap_handle_user_modify(&old_username, &mut user, &appstate.pool).await;
691    }
692
693    user.maybe_update_rdn().await?;
694    user.save(&appstate.pool).await?;
695
696    ldap_update_user_state(&mut user, &appstate.pool).await;
697
698    if group_diff.changed() || status_changing {
699        if !group_diff.added.is_empty() {
700            ldap_add_user_to_groups(
701                &user,
702                group_diff
703                    .added
704                    .iter()
705                    .map(|g| g.as_str())
706                    .collect::<HashSet<&str>>(),
707                &appstate.pool,
708            )
709            .await;
710        };
711
712        if !group_diff.removed.is_empty() {
713            ldap_remove_user_from_groups(
714                &user,
715                group_diff
716                    .removed
717                    .iter()
718                    .map(|g| g.as_str())
719                    .collect::<HashSet<&str>>(),
720                &appstate.pool,
721            )
722            .await;
723        };
724    }
725
726    appstate.trigger_action(AppEvent::UserModified(user_info));
727
728    info!("User {} updated user {username}", session.user.username);
729    Ok(ApiResponse::default())
730}

757pub async fn delete_user(
758    _role: AdminRole,
759    State(appstate): State<AppState>,
760    Path(username): Path<String>,
761    session: SessionInfo,
762) -> ApiResult {
763    debug!("User {} deleting user {username}", session.user.username);
764    if session.user.username == username {
765        debug!("User {username} attempted to delete himself");
766        return Ok(ApiResponse {
767            json: json!({}),
768            status: StatusCode::BAD_REQUEST,
769        });
770    }
771    if let Some(user) = User::find_by_username(&appstate.pool, &username).await? {
772        // Get rid of all devices of the deleted user from networks first
773        debug!(
774            "User {} deleted user {username}, purging their network devices across all networks.",
775            session.user.username
776        );
777        let mut transaction = appstate.pool.begin().await?;
778        let user_for_ldap = if user.ldap_sync_allowed(&mut *transaction).await? {
779            Some(user.clone().as_noid())
780        } else {
781            None
782        };
783        user.delete_and_cleanup(&mut transaction, &appstate.wireguard_tx)
784            .await?;
785
786        appstate.trigger_action(AppEvent::UserDeleted(username.clone()));
787        transaction.commit().await?;
788        update_counts(&appstate.pool).await?;
789        if let Some(user_for_ldap) = user_for_ldap {
790            ldap_delete_user(&user_for_ldap, &appstate.pool).await;
791        }
792
793        info!("User {} deleted user {}", session.user.username, &username);
794        Ok(ApiResponse::default())
795    } else {
796        error!("User {username} not found");
797        Err(WebError::ObjectNotFound(format!(
798            "User {username} not found"
799        )))
800    }
