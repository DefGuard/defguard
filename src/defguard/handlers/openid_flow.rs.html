55impl From<&User<Id>> for StandardClaims<CoreGenderClaim> {
56    fn from(user: &User<Id>) -> StandardClaims<CoreGenderClaim> {
57        let mut name = LocalizedClaim::new();
58        name.insert(None, EndUserName::new(user.name()));
59        let mut given_name = LocalizedClaim::new();
60        given_name.insert(None, EndUserGivenName::new(user.first_name.clone()));
61        let mut given_name = LocalizedClaim::new();
62        given_name.insert(None, EndUserGivenName::new(user.first_name.clone()));
63        let mut family_name = LocalizedClaim::new();
64        family_name.insert(None, EndUserFamilyName::new(user.last_name.clone()));
65        let email = EndUserEmail::new(user.email.clone());
66        let phone_number = user.phone.clone().map(EndUserPhoneNumber::new);
67        let preferred_username = EndUserUsername::new(user.username.clone());
68
69        StandardClaims::new(SubjectIdentifier::new(user.username.clone()))
70            .set_name(Some(name))
71            .set_given_name(Some(given_name))
72            .set_family_name(Some(family_name))
73            .set_email(Some(email))
74            .set_phone_number(phone_number)
75            .set_preferred_username(Some(preferred_username))
76    }
77}

100/// Provide `OAuth2Client` when Basic Authorization header contains `client_id` and `client_secret`.
101impl<S> OptionalFromRequestParts<S> for OAuth2Client<Id>
102where
103    S: Send + Sync,
104    AppState: FromRef<S>,
105{
106    type Rejection = WebError;
107
108    async fn from_request_parts(
109        parts: &mut Parts,
110        state: &S,
111    ) -> Result<Option<Self>, Self::Rejection> {
112        if let Some(basic_auth) = parts.headers.get(AUTHORIZATION).and_then(|value| {
113            if let Ok(value) = value.to_str() {
114                if value.starts_with("Basic ") {
115                    return value.get(6..);
116                }
117            }
118            None
119        }) {
120            if let Ok(decoded) = BASE64_STANDARD.decode(basic_auth) {
121                if let Ok(auth_pair) = String::from_utf8(decoded) {
122                    if let Some((client_id, client_secret)) = auth_pair.split_once(':') {
123                        let appstate = AppState::from_ref(state);
124                        return OAuth2Client::find_by_auth(
125                            &appstate.pool,
126                            client_id,
127                            client_secret,
128                        )
129                        .await
130                        .map_err(Into::into);
131                    }
132                }
133            }
134            Err(WebError::Authorization("Invalid credentials".into()))
135        } else {
136            Ok(None)
137        }
138    }
139}

141/// List of values for "response_type" field.
142struct FieldResponseTypes(Vec<CoreResponseType>);
143
144impl Deref for FieldResponseTypes {
145    type Target = Vec<CoreResponseType>;
146
147    fn deref(&self) -> &Self::Target {
148        &self.0
149    }
150}
151
152impl DerefMut for FieldResponseTypes {
153    fn deref_mut(&mut self) -> &mut Self::Target {
154        &mut self.0
155    }
156}
157
158impl Serialize for FieldResponseTypes {
159    // serialize to a string with values separated by space
160    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
161        let types: Vec<&str> = self.iter().map(CoreResponseType::as_ref).collect();
162        serializer.serialize_str(&types.join(" "))
163    }
164}
165
166struct FieldResponseTypesVisitor;
167
168impl Visitor<'_> for FieldResponseTypesVisitor {
169    type Value = FieldResponseTypes;
170
171    fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
172        write!(
173            formatter,
174            "a string containing `code`, `id_token`, or `token`"
175        )
176    }
177
178    fn visit_str<E>(self, s: &str) -> Result<Self::Value, E>
179    where
180        E: DeError,
181    {
182        let mut response_types = FieldResponseTypes(Vec::new());
183        for value in s.split(' ') {
184            match value {
185                "code" => response_types.push(CoreResponseType::Code),
186                "id_token" => response_types.push(CoreResponseType::IdToken),
187                "token" => response_types.push(CoreResponseType::Token),
188                _ => return Err(DeError::invalid_value(Unexpected::Str(s), &self)),
189            }
190        }
191        Ok(response_types)
192    }
193}
194
195impl<'de> Deserialize<'de> for FieldResponseTypes {
196    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
197    where
198        D: Deserializer<'de>,
199    {
200        deserializer.deserialize_str(FieldResponseTypesVisitor)
201    }
202}

204/// Authentication Request
205/// See https://openid.net/specs/openid-connect-core-1_0.html#AuthRequest
206#[derive(Deserialize, Serialize)]
207pub struct AuthenticationRequest {
208    #[serde(default)]
209    #[serde(skip_serializing)]
210    allow: bool,
211    scope: String,
212    response_type: FieldResponseTypes,
213    client_id: String,
214    // client_secret: Option<String>,
215    redirect_uri: String,
216    state: Option<String>,
217    // response_mode: Option<String>,
218    nonce: Option<String>,
219    // display: Option<String>,
220    prompt: Option<String>,
221    // max_age: Option<String>,
222    // ui_locales: Option<String>,
223    // id_token_hint: Option<String>,
224    // login_hint: Option<String>,
225    // acr_values: Option<String>,
226    // PKCE
227    code_challenge: Option<String>,
228    code_challenge_method: Option<String>,
229}
230
231impl AuthenticationRequest {
232    fn validate_for_client(
233        &self,
234        oauth2client: &OAuth2Client<Id>,
235    ) -> Result<(), CoreAuthErrorResponseType> {
236        // check scope: it is valid if any requested scope exists in the `oauth2client`
237        if self
238            .scope
239            .split(' ')
240            .all(|scope| !oauth2client.scope.iter().any(|s| s == scope))
241        {
242            error!(
243                "Invalid scope for client {}: {}",
244                oauth2client.name, self.scope
245            );
246            return Err(CoreAuthErrorResponseType::InvalidScope);
247        }
248
249        // currently we support only "code" for `response_type`
250        if self.response_type.len() != 1 || !self.response_type.contains(&CoreResponseType::Code) {
251            error!(
252                "Invalid response_type for client {}, only 'code' supported",
253                oauth2client.name
254            );
255            return Err(CoreAuthErrorResponseType::InvalidRequest);
256        }
257
258        // assume `client_id` is the same here and in `oauth2client`
259
260        // check `redirect_uri` matches client config (ignoring trailing slashes)
261        let parsed_redirect_uris: Vec<String> = oauth2client
262            .redirect_uri
263            .iter()
264            .map(|uri| uri.trim_end_matches('/').into())
265            .collect();
266        if self
267            .redirect_uri
268            .split(' ')
269            .map(|uri| uri.trim_end_matches('/'))
270            .all(|uri| !parsed_redirect_uris.iter().any(|u| u == uri))
271        {
272            error!(
273                "Invalid redirect_uri for client {}: {} not in [{}]",
274                oauth2client.name,
275                self.redirect_uri,
276                oauth2client.redirect_uri.join(" "),
277            );
278            return Err(CoreAuthErrorResponseType::AccessDenied);
279        }
280
281        // check PKCE; currently, only SHA-256 method is supported
282        // TODO: support `plain` which is the default if not specified
283        if self.code_challenge.is_some() && self.code_challenge_method != Some("S256".to_string()) {
284            error!(
285                "Invalid PKCE method: {:?}, only S256 supported",
286                self.code_challenge_method
287                    .as_ref()
288                    .map_or("None", String::as_str),
289            );
290            return Err(CoreAuthErrorResponseType::InvalidRequest);
291        }
292
293        info!("Validation succeeded for client {}", oauth2client.name);
294
295        Ok(())
296    }
297}

299/// Helper function which creates redirect Uri with authorization code
300async fn generate_auth_code_redirect(
301    appstate: AppState,
302    data: AuthenticationRequest,
303    user_id: Id,
304) -> Result<String, WebError> {
305    let mut url =
306        Url::parse(&data.redirect_uri).map_err(|_| WebError::Http(StatusCode::BAD_REQUEST))?;
307    let auth_code = AuthCode::new(
308        user_id,
309        data.client_id,
310        data.redirect_uri,
311        data.scope,
312        data.nonce,
313        data.code_challenge,
314    )
315    .save(&appstate.pool)
316    .await?;
317
318    {
319        let mut query_pairs = url.query_pairs_mut();
320        query_pairs.append_pair("code", auth_code.code.as_str());
321        if let Some(state) = data.state {
322            query_pairs.append_pair("state", &state);
323        };
324    };
325
326    Ok(url.to_string())
327}

329/// Helper function to return redirection with status code 302.
330fn redirect_to<T: AsRef<str>>(
331    uri: T,
332    private_cookies: PrivateCookieJar,
333) -> (StatusCode, HeaderMap, PrivateCookieJar) {
334    let mut headers = HeaderMap::new();
335    headers.insert(
336        LOCATION,
337        HeaderValue::try_from(uri.as_ref()).expect("URI isn't a valid header value"),
338    );
339
340    (StatusCode::FOUND, headers, private_cookies)
341}

343/// Helper function to redirect unauthorized user to login page
344/// and store information about OpenID authorize url in cookie to redirect later
345fn login_redirect(
346    data: &AuthenticationRequest,
347    private_cookies: PrivateCookieJar,
348) -> (StatusCode, HeaderMap, PrivateCookieJar) {
349    let config = server_config();
350    let base_url = config.url.join("api/v1/oauth/authorize").unwrap();
351    let cookie = Cookie::build((
352        SIGN_IN_COOKIE_NAME,
353        format!(
354            "{base_url}?{}",
355            serde_urlencoded::to_string(data).unwrap_or_default()
356        ),
357    ))
358    .domain(
359        config
360            .cookie_domain
361            .clone()
362            .expect("Cookie domain not found"),
363    )
364    .path("/")
365    .secure(!config.cookie_insecure)
366    .same_site(SameSite::Lax)
367    .http_only(true)
368    .max_age(Duration::minutes(10));
369    redirect_to("/login", private_cookies.add(cookie))
370}

372/// Authorization Endpoint
373/// See https://openid.net/specs/openid-connect-core-1_0.html#AuthorizationEndpoint
374pub async fn authorization(
375    State(appstate): State<AppState>,
376    Query(data): Query<AuthenticationRequest>,
377    cookies: CookieJar,
378    private_cookies: PrivateCookieJar,
379) -> Result<(StatusCode, HeaderMap, PrivateCookieJar), WebError> {
380    let error;
381    if let Some(oauth2client) =
382        OAuth2Client::find_by_client_id(&appstate.pool, &data.client_id).await?
383    {
384        match data.validate_for_client(&oauth2client) {
385            Ok(()) => {
386                match &data.prompt {
387                    Some(s) if s == "consent" => {
388                        info!(
389                            "Redirecting user to consent form - client id {}",
390                            data.client_id
391                        );
392                        // FIXME: do not panic
393                        return Ok(redirect_to(
394                            format!("/consent?{}", serde_urlencoded::to_string(data).unwrap(),),
395                            private_cookies,
396                        ));
397                    }
398                    Some(s) if s == "none" => {
399                        error!("'none' prompt in client id {} request", data.client_id);
400                        error = CoreAuthErrorResponseType::LoginRequired;
401                    }
402                    _ => {
403                        return if let Some(session_cookie) = cookies.get(SESSION_COOKIE_NAME) {
404                            if let Ok(Some(session)) =
405                                Session::find_by_id(&appstate.pool, session_cookie.value()).await
406                            {
407                                // If session expired return login
408                                if session.expired() {
409                                    info!("Session {} for user id {} has expired, redirecting to login", session.id, session.user_id);
410                                    let _result = session.delete(&appstate.pool).await;
411                                    Ok(login_redirect(&data, private_cookies))
412                                } else {
413                                    let mut user =
414                                        User::find_by_id(&appstate.pool, session.user_id)
415                                            .await?
416                                            .ok_or(WebError::Authorization(
417                                                "User not found".into(),
418                                            ))?;
419
420                                    user.verify_mfa_state(&appstate.pool).await?;
421
422                                    // Session exists even if user hasn't completed MFA verification yet,
423                                    // thus we need to check if MFA is enabled and the verification is done.
424                                    if user.mfa_enabled
425                                        && session.state != SessionState::MultiFactorVerified
426                                    {
427                                        info!(
428                                            "MFA not verified for user id {}, redirecting to login",
429                                            session.user_id
430                                        );
431                                        return Ok(login_redirect(&data, private_cookies));
432                                    }
433
434                                    // If session is present check if app is in user authorized apps.
435                                    // If yes return auth code and state else redirect to consent form.
436                                    if let Some(app) =
437                                        OAuth2AuthorizedApp::find_by_user_and_oauth2client_id(
438                                            &appstate.pool,
439                                            session.user_id,
440                                            oauth2client.id,
441                                        )
442                                        .await?
443                                    {
444                                        info!(
445                                            "OAuth client id {} authorized by user id {}, returning auth code",
446                                            app.oauth2client_id, session.user_id
447                                        );
448                                        let private_cookies = private_cookies
449                                            .remove(Cookie::from(SIGN_IN_COOKIE_NAME));
450                                        let location = generate_auth_code_redirect(
451                                            appstate,
452                                            data,
453                                            session.user_id,
454                                        )
455                                        .await?;
456                                        Ok(redirect_to(location, private_cookies))
457                                    } else {
458                                        // If authorized app not found redirect to consent form
459                                        info!(
460                                            "OAuth client id {} not yet authorized by user id {}, redirecting to consent form",
461                                            oauth2client.id, session.user_id
462                                        );
463                                        Ok(redirect_to(
464                                            format!(
465                                                "/consent?{}",
466                                                serde_urlencoded::to_string(data).unwrap()
467                                            ),
468                                            private_cookies,
469                                        ))
470                                    }
471                                }
472                            } else {
473                                // If session is not present in db redirect to login
474                                info!(
475                                    "Session {} not found, redirecting to login page",
476                                    session_cookie.value()
477                                );
478                                Ok(login_redirect(&data, private_cookies))
479                            }
480
481                        // If no session cookie provided redirect to login
482                        } else {
483                            info!("Session cookie not provided, redirecting to login page");
484                            Ok(login_redirect(&data, private_cookies))
485                        };
486                    }
487                }
488            }
489            Err(err) => {
490                error!(
491                    "OIDC login validation failed for client {}: {err:?}",
492                    data.client_id
493                );
494                error = err;
495            }
496        }
497    } else {
498        error!("OIDC client id {} not found", data.client_id);
499        error = CoreAuthErrorResponseType::UnauthorizedClient;
500    }
501
502    let mut url =
503        Url::parse(&data.redirect_uri).map_err(|_| WebError::Http(StatusCode::BAD_REQUEST))?;
504
505    {
506        let mut query_pairs = url.query_pairs_mut();
507        query_pairs.append_pair("error", error.as_ref());
508        if let Some(state) = data.state {
509            query_pairs.append_pair("state", &state);
510        };
511    };
512
513    Ok(redirect_to(url, private_cookies))
514}

531/// Login Authorization Endpoint redirect with authorization code
532pub async fn secure_authorization(
533    session_info: SessionInfo,
534    State(appstate): State<AppState>,
535    Query(data): Query<AuthenticationRequest>,
536    private_cookies: PrivateCookieJar,
537) -> Result<(StatusCode, HeaderMap, PrivateCookieJar), WebError> {
538    let mut url =
539        Url::parse(&data.redirect_uri).map_err(|_| WebError::Http(StatusCode::BAD_REQUEST))?;
540    let error;
541    if data.allow {
542        if let Some(oauth2client) =
543            OAuth2Client::find_by_client_id(&appstate.pool, &data.client_id).await?
544        {
545            match data.validate_for_client(&oauth2client) {
546                Ok(()) => {
547                    if OAuth2AuthorizedApp::find_by_user_and_oauth2client_id(
548                        &appstate.pool,
549                        session_info.user.id,
550                        oauth2client.id,
551                    )
552                    .await?
553                    .is_none()
554                    {
555                        let app = OAuth2AuthorizedApp::new(session_info.user.id, oauth2client.id);
556                        app.save(&appstate.pool).await?;
557
558                        send_new_device_ocid_login_email(
559                            &session_info.user.email,
560                            oauth2client.name.to_string(),
561                            &appstate.mail_tx,
562                            &session_info.session,
563                        )
564                        .await?;
565                    }
566                    info!(
567                        "User {} allowed login with client {}",
568                        session_info.user.username, oauth2client.name
569                    );
570                    let private_cookies = private_cookies.remove(SIGN_IN_COOKIE_NAME);
571                    let location =
572                        generate_auth_code_redirect(appstate, data, session_info.user.id).await?;
573                    info!(
574                        "Redirecting user {} to {location}",
575                        session_info.user.username
576                    );
577                    return Ok(redirect_to(location, private_cookies));
578                }
579                Err(err) => {
580                    info!(
581                        "OIDC login validation failed for user {}, client {}",
582                        session_info.user.username, oauth2client.name
583                    );
584                    error = err;
585                }
586            }
587        } else {
588            error!(
589                "User {} tried to log in with non-existent OIDC client id {}",
590                session_info.user.username, data.client_id
591            );
592            error = CoreAuthErrorResponseType::UnauthorizedClient;
593        }
594    } else {
595        info!(
596            "User {} denied OIDC login with app id {}",
597            session_info.user.username, data.client_id
598        );
599        error = CoreAuthErrorResponseType::AccessDenied;
600    }
601
602    {
603        let mut query_pairs = url.query_pairs_mut();
604        query_pairs.append_pair("error", error.as_ref());
605        if let Some(state) = data.state {
606            query_pairs.append_pair("state", &state);
607        };
608    };
609
610    Ok(redirect_to(url, private_cookies))
611}

613/// https://openid.net/specs/openid-connect-core-1_0.html#TokenRequest
614#[derive(Deserialize)]
615pub struct TokenRequest {
616    grant_type: String,
617    // grant_type == "authorization_code"
618    code: Option<String>,
619    redirect_uri: Option<String>,
620    // grant_type == "refresh_token"
621    refresh_token: Option<String>,
622    // scope: Option<String>,
623    // Authorization
624    client_id: Option<String>,
625    client_secret: Option<String>,
626    // PKCE
627    code_verifier: Option<String>,
628}
629
630impl TokenRequest {
631    /// Verify Proof Key for Code Exchange (PKCE) https://www.rfc-editor.org/rfc/rfc7636
632    fn verify_pkce(&self, code_challenge: Option<&String>) -> bool {
633        if let Some(challenge) = code_challenge {
634            if let Some(verifier) = &self.code_verifier {
635                let pkce_challenge = PkceCodeChallenge::from_code_verifier_sha256(
636                    &PkceCodeVerifier::new(verifier.into()),
637                );
638                pkce_challenge.as_str() == challenge
639            } else {
640                false
641            }
642        } else {
643            true
644        }
645    }
646
647    fn authorization_code_flow<T>(
648        &self,
649        auth_code: &AuthCode<Id>,
650        token: &OAuth2Token,
651        claims: StandardClaims<CoreGenderClaim>,
652        base_url: &Url,
653        secret: T,
654        rsa_key: Option<CoreRsaPrivateSigningKey>,
655        group_claims: GroupClaims,
656    ) -> Result<DefguardTokenResponse, CoreErrorResponseType>
657    where
658        T: Into<Vec<u8>>,
659    {
660        // assume self.grant_type == "authorization_code"
661        if let (Some(code), Some(redirect_uri)) = (&self.code, &self.redirect_uri) {
662            if redirect_uri.trim_end_matches('/') != auth_code.redirect_uri.trim_end_matches('/') {
663                error!(
664                    "Redirect URIs don't match for client_id {}: {redirect_uri} != {}",
665                    self.client_id.as_ref().map_or("Unknown", String::as_str),
666                    auth_code.redirect_uri
667                );
668                return Err(CoreErrorResponseType::UnauthorizedClient);
669            }
670
671            if !self.verify_pkce(auth_code.code_challenge.as_ref()) {
672                error!(
673                    "PKCE verification failed for client id {}",
674                    self.client_id.as_ref().map_or("Unknown", String::as_str)
675                );
676                return Err(CoreErrorResponseType::InvalidRequest);
677            }
678
679            let access_token = AccessToken::new(token.access_token.clone());
680            // append ID token only when scope contains "openid"
681            let id_token = if token.scope.split(' ').any(|scope| scope == "openid") {
682                debug!("Scope contains openid, issuing JWT ID token");
683                let authorization_code = AuthorizationCode::new(code.into());
684                let issue_time = Utc::now();
685                let timeout: std::time::Duration = server_config().session_timeout.into();
686                let expiration = issue_time + timeout;
687                let id_token_claims = IdTokenClaims::new(
688                    IssuerUrl::from_url(base_url.clone()),
689                    vec![Audience::new(auth_code.client_id.clone())],
690                    expiration,
691                    issue_time,
692                    claims,
693                    group_claims,
694                )
695                .set_nonce(auth_code.nonce.clone().map(Nonce::new));
696
697                let id_token = match rsa_key {
698                    Some(key) => IdToken::new(
699                        id_token_claims,
700                        &key,
701                        CoreJwsSigningAlgorithm::RsaSsaPkcs1V15Sha256,
702                        Some(&access_token),
703                        Some(&authorization_code),
704                    ),
705                    None => IdToken::new(
706                        id_token_claims,
707                        &CoreHmacKey::new(secret),
708                        CoreJwsSigningAlgorithm::HmacSha256,
709                        Some(&access_token),
710                        Some(&authorization_code),
711                    ),
712                };
713                id_token.ok()
714            } else {
715                None
716            };
717
718            let mut token_response = DefguardTokenResponse::new(
719                access_token,
720                CoreTokenType::Bearer,
721                IdTokenFields::new(id_token, EmptyExtraTokenFields {}),
722            );
723            token_response.set_refresh_token(Some(RefreshToken::new(token.refresh_token.clone())));
724            Ok(token_response)
725        } else {
726            if self.code.is_none() {
727                error!("Request missing code param");
728            }
729            if self.redirect_uri.is_none() {
730                error!("Request missing redirect_uri param");
731            }
732            Err(CoreErrorResponseType::InvalidRequest)
733        }
734    }
735
736    fn refresh_token_flow(
737        token: &OAuth2Token,
738    ) -> StandardTokenResponse<EmptyExtraTokenFields, CoreTokenType> {
739        // assume self.grant_type == "refresh_token"
740        let access_token = AccessToken::new(token.access_token.clone());
741        let refresh_token = RefreshToken::new(token.refresh_token.clone());
742        let mut token_response = StandardTokenResponse::new(
743            access_token,
744            CoreTokenType::Bearer,
745            EmptyExtraTokenFields {},
746        );
747        token_response.set_refresh_token(Some(refresh_token));
748        token_response
749    }
750
751    async fn oauth2client(&self, pool: &PgPool) -> Option<OAuth2Client<Id>> {
752        if let (Some(client_id), Some(client_secret)) =
753            (self.client_id.as_ref(), self.client_secret.as_ref())
754        {
755            OAuth2Client::find_by_auth(pool, client_id, client_secret)
756                .await
757                .unwrap_or_default()
758            // .map_err(|_| CoreErrorResponseType::InvalidClient)
759        } else {
760            None
761        }
762    }
763}

765/// Token Endpoint
766/// https://openid.net/specs/openid-connect-core-1_0.html#TokenEndpoint
767/// https://openid.net/specs/openid-connect-core-1_0.html#RefreshTokens
768pub async fn token(
769    State(appstate): State<AppState>,
770    oauth2client: Option<OAuth2Client<Id>>,
771    Form(form): Form<TokenRequest>,
772) -> ApiResult {
773    // TODO: cleanup branches
774    match form.grant_type.as_str() {
775        "authorization_code" => {
776            debug!("Staring authorization_code flow");
777
778            // for logging
779            let form_client_id = match &form.client_id {
780                Some(id) => id.clone(),
781                None => String::from("N/A"),
782            };
783
784            if let Some(code) = &form.code {
785                if let Some(stored_auth_code) = AuthCode::find_code(&appstate.pool, code).await? {
786                    // copy data before removing used token
787                    let auth_code = stored_auth_code.clone();
788                    // remove authorization_code from DB so it cannot be reused
789                    debug!(
790                        "Removing used authorization_code {code}, client_id `{}`",
791                        form_client_id
792                    );
793                    stored_auth_code.consume(&appstate.pool).await?;
794                    if let Some(client) = oauth2client.or(form.oauth2client(&appstate.pool).await) {
795                        if let Some(user) =
796                            User::find_by_id(&appstate.pool, auth_code.user_id).await?
797                        {
798                            if let Some(authorized_app) =
799                                OAuth2AuthorizedApp::find_by_user_and_oauth2client_id(
800                                    &appstate.pool,
801                                    user.id,
802                                    client.id,
803                                )
804                                .await?
805                            {
806                                debug!(
807                                    "Issuing new token for user {} client {}",
808                                    user.username, client.name
809                                );
810                                // Remove existing token in case same client asks for new token
811                                if let Some(token) = OAuth2Token::find_by_authorized_app_id(
812                                    &appstate.pool,
813                                    authorized_app.id,
814                                )
815                                .await?
816                                {
817                                    token.delete(&appstate.pool).await?;
818                                }
819                                let token = OAuth2Token::new(
820                                    authorized_app.id,
821                                    auth_code.redirect_uri.clone(),
822                                    auth_code.scope.clone(),
823                                );
824                                let group_claims = if auth_code.scope.contains("groups") {
825                                    get_group_claims(&appstate.pool, &user).await?
826                                } else {
827                                    GroupClaims { groups: None }
828                                };
829                                let config = server_config();
830                                match form.authorization_code_flow(
831                                    &auth_code,
832                                    &token,
833                                    (&user).into(),
834                                    &config.url,
835                                    client.client_secret,
836                                    config.openid_key(),
837                                    group_claims,
838                                ) {
839                                    Ok(response) => {
840                                        token.save(&appstate.pool).await?;
841                                        info!(
842                                            "Issued new token for user {} client {}",
843                                            user.username, client.name
844                                        );
845                                        return Ok(ApiResponse {
846                                            json: json!(response),
847                                            status: StatusCode::OK,
848                                        });
849                                    }
850                                    Err(err) => {
851                                        error!(
852                                            "Error issuing new token for user {} client {}: {}",
853                                            user.username, client.name, err
854                                        );
855                                        let response =
856                                            StandardErrorResponse::<CoreErrorResponseType>::new(
857                                                err, None, None,
858                                            );
859                                        return Ok(ApiResponse {
860                                            json: json!(response),
861                                            status: StatusCode::BAD_REQUEST,
862                                        });
863                                    }
864                                }
865                            }
866                            error!("Can't issue token - authorized app not found for user {}, client {}", user.username, client.name);
867                        } else {
868                            error!("User id {} not found", auth_code.user_id);
869                        }
870                    } else {
871                        error!("OAuth client id `{form_client_id}` not found");
872                    }
873                } else {
874                    error!("OAuth auth code not found");
875                }
876            } else {
877                error!("No code provided in request for client id `{form_client_id}`",);
878            }
879        }
880        "refresh_token" => {
881            debug!("Starting refresh_token flow");
882            if let Some(refresh_token) = form.refresh_token {
883                if let Ok(Some(mut token)) =
884                    OAuth2Token::find_refresh_token(&appstate.pool, &refresh_token).await
885                {
886                    token.refresh_and_save(&appstate.pool).await?;
887                    let response = TokenRequest::refresh_token_flow(&token);
888                    token.save(&appstate.pool).await?;
889                    return Ok(ApiResponse {
890                        json: json!(response),
891                        status: StatusCode::OK,
892                    });
893                }
894            }
895        }
896        _ => (), // TODO: Err(CoreErrorResponseType::UnsupportedGrantType),
897    }
898    let err = CoreErrorResponseType::UnsupportedGrantType;
899    let response = StandardErrorResponse::<CoreErrorResponseType>::new(err, None, None);
900    Ok(ApiResponse {
901        json: json!(response),
902        status: StatusCode::BAD_REQUEST,
903    })
904}

915// Must be served under /.well-known/openid-configuration
916pub async fn openid_configuration() -> ApiResult {
917    let config = server_config();
918    let provider_metadata = CoreProviderMetadata::new(
919        IssuerUrl::from_url(config.url.clone()),
920        AuthUrl::from_url(config.url.join("api/v1/oauth/authorize").unwrap()),
921        JsonWebKeySetUrl::from_url(config.url.join("api/v1/oauth/discovery/keys").unwrap()),
922        vec![ResponseTypes::new(vec![CoreResponseType::Code])],
923        vec![CoreSubjectIdentifierType::Public],
924        vec![
925            CoreJwsSigningAlgorithm::HmacSha256,           // required
926            CoreJwsSigningAlgorithm::RsaSsaPkcs1V15Sha256, // recommended
927        ],
928        EmptyAdditionalProviderMetadata {},
929    )
930    .set_token_endpoint(Some(TokenUrl::from_url(
931        config.url.join("api/v1/oauth/token").unwrap(),
932    )))
933    .set_scopes_supported(Some(vec![
934        Scope::new("openid".into()),
935        Scope::new("profile".into()),
936        Scope::new("email".into()),
937        Scope::new("phone".into()),
938        Scope::new("groups".into()),
939    ]))
940    .set_claims_supported(Some(vec![
941        CoreClaimName::new("iss".into()),
942        CoreClaimName::new("sub".into()),
943        CoreClaimName::new("aud".into()),
944        CoreClaimName::new("exp".into()),
945        CoreClaimName::new("iat".into()),
946        CoreClaimName::new("name".into()),
947        CoreClaimName::new("given_name".into()),
948        CoreClaimName::new("family_name".into()),
949        CoreClaimName::new("email".into()),
950        CoreClaimName::new("phone_number".into()),
951        CoreClaimName::new("groups".into()),
952    ]))
953    .set_grant_types_supported(Some(vec![
954        CoreGrantType::AuthorizationCode,
955        CoreGrantType::RefreshToken,
956    ]))
957    .set_userinfo_endpoint(Some(UserInfoUrl::from_url(
958        config.url.join("api/v1/oauth/userinfo").unwrap(),
959    )));
960
961    Ok(ApiResponse {
962        json: json!(provider_metadata),
963        status: StatusCode::OK,
964    })
