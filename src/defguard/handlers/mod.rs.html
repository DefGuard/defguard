60impl From<WebError> for ApiResponse {
61    fn from(web_error: WebError) -> ApiResponse {
62        match web_error {
63            WebError::ObjectNotFound(msg) => {
64                ApiResponse::new(json!({ "msg": msg }), StatusCode::NOT_FOUND)
65            }
66            WebError::Authorization(msg) => {
67                error!(msg);
68                ApiResponse::new(json!({ "msg": msg }), StatusCode::UNAUTHORIZED)
69            }
70            WebError::Forbidden(msg) => {
71                error!(msg);
72                ApiResponse::new(json!({ "msg": msg }), StatusCode::FORBIDDEN)
73            }
74            WebError::DbError(_)
75            | WebError::Grpc(_)
76            | WebError::Ldap(_)
77            | WebError::WebauthnRegistration(_)
78            | WebError::Serialization(_)
79            | WebError::ModelError(_)
80            | WebError::ServerConfigMissing
81            | WebError::EmailMfa(_)
82            | WebError::ClientIpError
83            | WebError::FirewallError(_) => {
84                error!("{web_error}");
85                ApiResponse::new(
86                    json!({"msg": "Internal server error"}),
87                    StatusCode::INTERNAL_SERVER_ERROR,
88                )
89            }
90            WebError::AclError(err) => match err {
91                AclError::ParseIntError(_)
92                | AclError::IpNetworkError(_)
93                | AclError::AddrParseError(_)
94                | AclError::InvalidRelationError(_)
95                | AclError::InvalidPortsFormat(_) => ApiResponse::new(
96                    json!({"msg": "Unprocessable entity"}),
97                    StatusCode::UNPROCESSABLE_ENTITY,
98                ),
99                AclError::InvalidIpRangeError(err) => ApiResponse::new(
100                    json!({"msg": format!("Invalid IP range: {err}")}),
101                    StatusCode::UNPROCESSABLE_ENTITY,
102                ),
103                AclError::RuleNotFoundError(id) => ApiResponse::new(
104                    json!({"msg": format!("Rule {id} not found")}),
105                    StatusCode::NOT_FOUND,
106                ),
107                AclError::RuleAlreadyAppliedError(id) => ApiResponse::new(
108                    json!({"msg": format!("Rule {id} already applied")}),
109                    StatusCode::BAD_REQUEST,
110                ),
111                AclError::AliasNotFoundError(id) => ApiResponse::new(
112                    json!({"msg": format!("Alias {id} not found")}),
113                    StatusCode::NOT_FOUND,
114                ),
115                AclError::AliasAlreadyAppliedError(id) => ApiResponse::new(
116                    json!({"msg": format!("Alias {id} already applied")}),
117                    StatusCode::BAD_REQUEST,
118                ),
119                AclError::AliasUsedByRulesError(id) => ApiResponse::new(
120                    json!({"msg": format!("Alias {id} is used by some existing ACL rules")}),
121                    StatusCode::BAD_REQUEST,
122                ),
123                AclError::DbError(_) | AclError::FirewallError(_) => {
124                    error!("{err}");
125                    ApiResponse::new(
126                        json!({"msg": "Internal server error"}),
127                        StatusCode::INTERNAL_SERVER_ERROR,
128                    )
129                }
130                AclError::PortOutOfRangeError(port) => ApiResponse::new(
131                    json!({"msg": format!("Port out of range: {port}")}),
132                    StatusCode::UNPROCESSABLE_ENTITY,
133                ),
134                AclError::CannotModifyDeletedRuleError(id) => ApiResponse::new(
135                    json!({"msg": format!("Cannot modify deleted ACL rule {id}")}),
136                    StatusCode::BAD_REQUEST,
137                ),
138                AclError::CannotUseModifiedAliasInRuleError(alias_ids) => ApiResponse::new(
139                    json!({"msg": format!("Cannot use modified alias in ACL rule {alias_ids:?}")}),
140                    StatusCode::BAD_REQUEST,
141                ),
142            },
143            WebError::Http(status) => {
144                error!("{status}");
145                ApiResponse::new(
146                    json!({ "msg": status.canonical_reason().unwrap_or_default() }),
147                    status,
148                )
149            }
150            WebError::TooManyLoginAttempts(_) => ApiResponse::new(
151                json!({ "msg": "Too many login attempts" }),
152                StatusCode::TOO_MANY_REQUESTS,
153            ),
154            WebError::IncorrectUsername(msg)
155            | WebError::PubkeyValidation(msg)
156            | WebError::PubkeyExists(msg)
157            | WebError::BadRequest(msg) => {
158                error!(msg);
159                ApiResponse::new(json!({ "msg": msg }), StatusCode::BAD_REQUEST)
160            }
161            WebError::TemplateError(err) => {
162                error!("Template error: {err}");
163                ApiResponse::new(
164                    json!({"msg": "Internal server error"}),
165                    StatusCode::INTERNAL_SERVER_ERROR,
166                )
167            }
168            WebError::LicenseError(err) => match err {
169                LicenseError::DecodeError(msg) | LicenseError::InvalidLicense(msg) => {
170                    warn!(msg);
171                    ApiResponse::new(json!({ "msg": msg }), StatusCode::BAD_REQUEST)
172                }
173                LicenseError::SignatureMismatch => {
174                    let msg = "License signature doesn't match its content";
175                    warn!(msg);
176                    ApiResponse::new(json!({ "msg": msg }), StatusCode::BAD_REQUEST)
177                }
178                LicenseError::InvalidSignature => {
179                    let msg = "License signature is malformed and couldn't be read";
180                    warn!(msg);
181                    ApiResponse::new(json!({ "msg": msg }), StatusCode::BAD_REQUEST)
182                }
183                LicenseError::LicenseNotFound => {
184                    let msg = "License not found";
185                    warn!(msg);
186                    ApiResponse::new(json!({ "msg": msg }), StatusCode::NOT_FOUND)
187                }
188                _ => {
189                    error!("License error: {err}");
190                    ApiResponse::new(
191                        json!({"msg": "Internal server error"}),
192                        StatusCode::FORBIDDEN,
193                    )
194                }
195            },
196        }
197    }
198}

200impl IntoResponse for WebError {
201    fn into_response(self) -> Response {
202        let api_response = ApiResponse::from(self);
203        api_response.into_response()
204    }

207impl IntoResponse for ApiResponse {
208    fn into_response(self) -> Response {
209        let mut response = Json(self.json).into_response();
210        response.headers_mut().insert(
211            HeaderName::from_static("x-defguard-version"),
212            HeaderValue::from_static(VERSION),
213        );
214        *response.status_mut() = self.status;
215        response
216    }
217}

402/// Try to fetch [`User`] if the username is of the currently logged in user, or
403/// the logged in user is an admin.
404pub async fn user_for_admin_or_self(
405    pool: &PgPool,
406    session: &SessionInfo,
407    username: &str,
408) -> Result<User<Id>, WebError> {
409    if session.user.username == username || session.is_admin {
410        debug!(
411            "The user meets one or both of these conditions: \
412            1) the user from the current session has admin privileges, \
413            2) the user performs this operation on themself."
414        );
415        if let Some(user) = User::find_by_username(pool, username).await? {
416            debug!("User {} has been found in database.", user.username);
417            Ok(user)
418        } else {
419            debug!("User with {username} does not exist in database.");
420            Err(WebError::ObjectNotFound(format!(
421                "user {username} not found"
422            )))
423        }
424    } else {
425        debug!(
426            "User from the current session doesn't have enough privileges to do this operation."
427        );
428        Err(WebError::Forbidden("requires privileged access".into()))
429    }
430}

432/// Try to fetch [`Device'] if the device.id is of the currently logged in user, or
433/// the logged in user is an admin.
434#[cfg(feature = "wireguard")]
435pub async fn device_for_admin_or_self<'e, E: sqlx::PgExecutor<'e>>(
436    executor: E,
437    session: &SessionInfo,
438    id: Id,
439) -> Result<Device<Id>, WebError> {
440    let fetch = if session.is_admin {
441        Device::find_by_id(executor, id).await
442    } else {
443        Device::find_by_id_and_username(executor, id, &session.user.username).await
444    }?;
445
446    match fetch {
447        Some(device) => Ok(device),
448        None => Err(WebError::ObjectNotFound(format!(
449            "device id {id} not found"
450        ))),
451    }
