33impl AuthenticationKeyInfo {
34    pub async fn find_by_user_id<'e, E>(executor: E, user_id: Id) -> Result<Vec<Self>, SqlxError>
35    where
36        E: PgExecutor<'e>,
37    {
38        let q_res = query!(
39            "SELECT k.id key_id, k.name, k.key_type \"key_type: AuthenticationKeyType\", \
40            k.key, k.user_id, k.yubikey_id, \
41            y.name \"yubikey_name: Option<String>\", y.serial \"serial: Option<String>\" \
42            FROM \"authentication_key\" k \
43            LEFT JOIN \"yubikey\" y ON k.yubikey_id = y.id \
44            WHERE k.user_id = $1",
45            user_id
46        )
47        .fetch_all(executor)
48        .await?;
49        let res = q_res
50            .iter()
51            .map(|q| Self {
52                id: q.key_id,
53                key: q.key.clone(),
54                key_type: q.key_type.clone(),
55                user_id: q.user_id,
56                name: q.name.clone(),
57                yubikey_id: q.yubikey_id,
58                yubikey_name: q.yubikey_name.clone(),
59                yubikey_serial: q.serial.clone(),
60            })
61            .collect();
62
63        Ok(res)
64    }
65}

67async fn add_user_ssh_keys_to_list(pool: &PgPool, user: &User<Id>, ssh_keys: &mut Vec<String>) {
68    let keys_result =
69        AuthenticationKey::find_by_user_id(pool, user.id, Some(AuthenticationKeyType::Ssh)).await;
70
71    if let Ok(authentication_keys) = keys_result {
72        let mut keys: Vec<String> = authentication_keys
73            .into_iter()
74            .map(|item| item.key)
75            .collect();
76        ssh_keys.append(&mut keys);
77    }
78}

81pub struct SshKeysRequestParams {
82    username: Option<String>,
83    group: Option<String>,
84}
85
86/// Fetch public SSH keys for user
87///
88/// Meant to be used with `AuthorizedKeysCommand` config option in `sshd`.
89/// Should always return a response to partially mitigate user enumeration.
90/// Optional query params `username` and `group` are used for filtering users.
91/// If no params are specified an empty response is returned.
92pub async fn get_authorized_keys(
93    params: Query<SshKeysRequestParams>,
94    State(appstate): State<AppState>,
95) -> Result<String, WebError> {
96    info!("Fetching public SSH keys for {:?}", params);
97    let mut ssh_keys: Vec<String> = Vec::new();
98
99    // check if group filter was specified
100    match &params.group {
101        Some(group_name) => {
102            // fetch group
103            if let Some(group) = Group::find_by_name(&appstate.pool, group_name).await? {
104                // check if user filter was specified
105                if let Some(username) = &params.username {
106                    debug!("Fetching SSH keys for user {username} in group {group_name}");
107                    // fetch user
108                    if let Some(user) = User::find_by_username(&appstate.pool, username).await? {
109                        // check if user belongs to specified group
110                        let members = group.member_usernames(&appstate.pool).await?;
111                        if members.contains(&user.username) {
112                            add_user_ssh_keys_to_list(&appstate.pool, &user, &mut ssh_keys).await;
113                        } else {
114                            debug!("User {username} is not a member of group {group_name}",);
115                        }
116                    } else {
117                        debug!("Specified user does not exist");
118                    }
119                } else {
120                    debug!("Fetching SSH keys for all users in group {group_name}");
121                    // fetch all users in group
122                    let users = group.members(&appstate.pool).await?;
123                    for user in users {
124                        add_user_ssh_keys_to_list(&appstate.pool, &user, &mut ssh_keys).await;
125                    }
126                }
127            } else {
128                debug!("Specified group does not exist");
129            }
130        }
131        None => {
132            // check if user filter was specified
133            if let Some(username) = &params.username {
134                debug!("Fetching SSH keys for user {username}");
135                // fetch user
136                if let Some(user) = User::find_by_username(&appstate.pool, username).await? {
137                    add_user_ssh_keys_to_list(&appstate.pool, &user, &mut ssh_keys).await;
138                } else {
139                    debug!("Specified user does not exist");
140                }
141            }
142        }
143    }
144
145    // concatenate all keys into a response
146    Ok(ssh_keys.join("\n"))
147}

152    name: String,
153    key_type: AuthenticationKeyType,

156pub async fn add_authentication_key(
157    State(appstate): State<AppState>,
158    session: SessionInfo,
159    Path(username): Path<String>,
160    Json(data): Json<AddAuthenticationKeyData>,
161) -> ApiResult {
162    debug!(
163        "Adding authentication key of type {:?} for user {username}",
164        data.key_type,
165    );
166
167    // authorize request
168    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
169
170    let trimmed_key = data.key.trim_end_matches(['\n', '\r']);
171
172    // verify key
173    match data.key_type {
174        AuthenticationKeyType::Ssh => {
175            let parsed = trimmed_key.parse::<PublicKey>();
176            if parsed.is_err() {
177                error!("User {username} tried to insert invalid SSH key: {data:?}");
178                return Err(WebError::BadRequest("SSH key failed verification.".into()));
179            }
180        }
181        // FIXME: verify GPG key
182        AuthenticationKeyType::Gpg => {}
183    }
184
185    // check if exists
186    let exists_res = query!(
187        "SELECT COUNT(1) FROM \"authentication_key\" WHERE user_id = $1 AND key = $2",
188        user.id,
189        trimmed_key,
190    )
191    .fetch_one(&appstate.pool)
192    .await?;
193    if exists_res.count == Some(1) {
194        error!("User {username} tried to insert existing key: {data:?}");
195        return Err(WebError::BadRequest("Key already exists.".into()));
196    }
197
198    AuthenticationKey::new(
199        user.id,
200        trimmed_key.to_string(),
201        Some(data.name.clone()),
202        data.key_type.clone(),
203        None,
204    )
205    .save(&appstate.pool)
206    .await?;
207
208    info!(
209        "Added new key \"{}\" of type {:?} for user {username}",
210        data.name, data.key_type
211    );
212
213    Ok(ApiResponse {
214        json: json!({}),
215        status: StatusCode::CREATED,
216    })
217}

234pub async fn delete_authentication_key(
235    State(appstate): State<AppState>,
236    session: SessionInfo,
237    Path((username, key_id)): Path<(String, i64)>,
238) -> ApiResult {
239    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
240    if let Some(key) = AuthenticationKey::find_by_id(&appstate.pool, key_id).await? {
241        if !session.is_admin && user.id != key.user_id {
242            return Err(WebError::Forbidden(String::new()));
243        }
244        key.delete(&appstate.pool).await?;
245    } else {
246        error!("Key with id {} not found", key_id);
247        return Err(WebError::BadRequest("Key not found".into()));
248    }
249
250    Ok(ApiResponse {
251        json: json!({}),
252        status: StatusCode::OK,
253    })
254}

256#[derive(Debug, Deserialize, Clone)]
257pub struct RenameRequest {
258    name: String,
259}
260
261pub async fn rename_authentication_key(
262    State(appstate): State<AppState>,
263    session: SessionInfo,
264    Path((username, key_id)): Path<(String, i64)>,
265    Json(data): Json<RenameRequest>,
266) -> ApiResult {
267    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
268    if let Some(mut key) = AuthenticationKey::find_by_id(&appstate.pool, key_id).await? {
269        if key.yubikey_id.is_some() {
270            warn!(
271                "User {} tried to rename authentication key instead of yubikey",
272                username
273            );
274            return Err(WebError::BadRequest("Rename yubikey instead.".into()));
275        }
276        if !session.is_admin && user.id != key.user_id {
277            warn!(
278                "User {} tried to rename key ({}) of another user with id {}",
279                username, key_id, key.user_id
280            );
281            return Err(WebError::Forbidden(String::new()));
282        }
283        key.name = Some(data.name);
284        key.save(&appstate.pool).await?;
285    } else {
286        error!(
287            "User {} tried to rename non-existing key with id {}",
288            username, key_id
289        );
290        return Err(WebError::ObjectNotFound(String::new()));
291    }
292
293    Ok(ApiResponse {
294        json: json!({}),
295        status: StatusCode::OK,
296    })
