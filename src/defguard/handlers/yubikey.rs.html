8use super::{user_for_admin_or_self, ApiResponse, ApiResult};

11pub async fn delete_yubikey(
12    State(appstate): State<AppState>,
13    session: SessionInfo,
14    Path((username, key_id)): Path<(String, i64)>,
15) -> ApiResult {
16    debug!("Deleting yubikey {key_id} by {:?}", &session.user.id);
17    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
18    let Some(yubikey) = YubiKey::find_by_id(&appstate.pool, key_id).await? else {
19        error!("Yubikey with id {key_id} not found");
20        return Err(WebError::ObjectNotFound("YubiKey not found".into()));
21    };
22    if !session.is_admin && yubikey.user_id != user.id {
23        warn!(
24            "User {} tried to delete yubikey {key_id} of user {} without being an admin.",
25            user.id, yubikey.user_id
26        );
27        return Err(WebError::Forbidden("Not allowed to delete YubiKey".into()));
28    }
29    yubikey.delete(&appstate.pool).await?;
30    info!("Yubikey {key_id} deleted by user {}", user.id);
31    Ok(ApiResponse {
32        json: json!({}),
33        status: StatusCode::OK,
34    })
35}

37#[derive(Debug, Deserialize, Clone)]
38pub struct RenameRequest {
39    name: String,
40}
41
42pub async fn rename_yubikey(
43    State(appstate): State<AppState>,
44    session: SessionInfo,
45    Path((username, key_id)): Path<(String, i64)>,
46    Json(data): Json<RenameRequest>,
47) -> ApiResult {
48    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
49    debug!("User {} attempts to rename yubikey {}", user.id, key_id);
50    let Some(mut yubikey) = YubiKey::find_by_id(&appstate.pool, key_id).await? else {
51        error!("Yubikey with id {key_id} not found");
52        return Err(WebError::ObjectNotFound("YubiKey not found".into()));
53    };
54    if !session.is_admin && yubikey.user_id != user.id {
55        warn!(
56            "User {}, tried to rename yubikey {key_id} of user {} without being an admin.",
57            user.id, yubikey.user_id
58        );
59        return Err(WebError::Forbidden(String::new()));
60    }
61    yubikey.name = data.name;
62    yubikey.save(&appstate.pool).await?;
63    info!("Yubikey {} renamed by user {}", yubikey.id, user.id);
64    Ok(ApiResponse {
65        json: json!(yubikey),
66        status: StatusCode::OK,
67    })
