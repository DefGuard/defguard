45/// Parse a string with comma-separated IP addresses.
46/// Invalid addresses will be silently ignored.
47pub(crate) fn parse_address_list(ips: &str) -> Vec<IpNetwork> {
48    ips.split(',')
49        .filter_map(|ip| ip.trim().parse().ok())
50        .collect()
51}
52
53/// Parse a string with comma-separated IP network addresses.
54/// Host bits will be stripped.
55/// Invalid addresses will be silently ignored.
56pub(crate) fn parse_network_address_list(ips: &str) -> Vec<IpNetwork> {
57    ips.split(',')
58        .filter_map(|ip| ip.trim().parse().ok())
59        .filter_map(|ip: IpNetwork| {
60            let network_address = ip.network();
61            let network_mask = ip.mask();
62            IpNetwork::with_netmask(network_address, network_mask).ok()
63        })
64        .collect()
65}

127    _role: AdminRole,
128    State(appstate): State<AppState>,

132    let network_name = data.name.clone();

134        "User {} creating WireGuard network {network_name}",

137    let allowed_ips = data.parse_allowed_ips();
138    let network = WireguardNetwork::new(

153    let mut transaction = appstate.pool.begin().await?;
154    let network = network.save(&mut *transaction).await?;
155    network
156        .set_allowed_groups(&mut transaction, data.allowed_groups)
157        .await?;
158
159    // generate IP addresses for existing devices
160    network.add_all_allowed_devices(&mut transaction).await?;
161    info!("Assigning IPs for existing devices in network {network}");
162
163    appstate.send_wireguard_event(GatewayEvent::NetworkCreated(network.id, network.clone()));
164
165    transaction.commit().await?;
166
167    info!(
168        "User {} created WireGuard network {network_name}",
169        session.user.username
170    );
171    update_counts(&appstate.pool).await?;

173    Ok(ApiResponse {
174        json: json!(network),

440pub(crate) async fn import_network(
441    _role: AdminRole,
442    State(appstate): State<AppState>,
443    Json(data): Json<ImportNetworkData>,
444) -> ApiResult {
445    debug!("Importing network from config file");
446    let (mut network, imported_devices) =
447        parse_wireguard_config(&data.config).map_err(|error| {
448            error!("{error}");
449            WebError::Http(StatusCode::UNPROCESSABLE_ENTITY)
450        })?;
451    network.name = data.name;
452    network.endpoint = data.endpoint;
453
454    let mut transaction = appstate.pool.begin().await?;
455    let network = network.save(&mut *transaction).await?;
456    network
457        .set_allowed_groups(&mut transaction, data.allowed_groups)
458        .await?;
459
460    info!("New network {network} created");
461    appstate.send_wireguard_event(GatewayEvent::NetworkCreated(network.id, network.clone()));
462
463    let reserved_ips: Vec<IpAddr> = imported_devices
464        .iter()
465        .map(|dev| dev.wireguard_ip)
466        .collect();
467    let (devices, gateway_events) = network
468        .handle_imported_devices(&mut transaction, imported_devices)
469        .await?;
470    appstate.send_multiple_wireguard_events(gateway_events);
471
472    // assign IPs for other existing devices
473    debug!("Assigning IPs in imported network for remaining existing devices");
474    let gateway_events = network
475        .sync_allowed_devices(&mut transaction, Some(&reserved_ips))
476        .await?;
477    appstate.send_multiple_wireguard_events(gateway_events);
478    debug!("Assigned IPs in imported network for remaining existing devices");
479
480    transaction.commit().await?;
481
482    info!("Imported network {network} with {} devices", devices.len());
483
484    update_counts(&appstate.pool).await?;
485
486    Ok(ApiResponse {
487        json: json!(ImportedNetworkData { network, devices }),
488        status: StatusCode::CREATED,
489    })
490}

493pub(crate) async fn add_user_devices(
494    _role: AdminRole,
495    session: SessionInfo,
496    State(appstate): State<AppState>,
497    Path(network_id): Path<i64>,
498    Json(request_data): Json<MappedDevices>,
499) -> ApiResult {
500    let mapped_devices = request_data.devices.clone();
501    let user = session.user;
502    let device_count = mapped_devices.len();
503
504    debug!(
505        "User {} mapping {device_count} devices for network {network_id}",
506        user.username,
507    );
508
509    // finish early if no devices were provided in request
510    if mapped_devices.is_empty() {
511        debug!("No devices provided in request, skipping mapping");
512        return Ok(ApiResponse {
513            json: json!({}),
514            status: StatusCode::NO_CONTENT,
515        });
516    }
517
518    if let Some(network) = WireguardNetwork::find_by_id(&appstate.pool, network_id).await? {
519        // wrap loop in transaction to abort if a device is invalid
520        let mut transaction = appstate.pool.begin().await?;
521        let events = network
522            .handle_mapped_devices(&mut transaction, mapped_devices)
523            .await?;
524        appstate.send_multiple_wireguard_events(events);
525        transaction.commit().await?;
526
527        info!(
528            "User {} mapped {device_count} devices for {network_id} network",
529            user.username,
530        );
531        update_counts(&appstate.pool).await?;
532
533        Ok(ApiResponse {
534            json: json!({}),
535            status: StatusCode::CREATED,
536        })
537    } else {
538        error!("Failed to map devices, network {network_id} not found");
539        Err(WebError::ObjectNotFound(format!(
540            "Network {network_id} not found"
541        )))
542    }
543}

607pub(crate) async fn add_device(
608    _can_manage_devices: CanManageDevices,
609    session: SessionInfo,
610    State(appstate): State<AppState>,
611    // Alias, because otherwise `axum` reports conflicting routes.
612    Path(username): Path<String>,
613    Json(add_device): Json<AddDevice>,
614) -> ApiResult {
615    let device_name = add_device.name.clone();
616    debug!(
617        "User {} adding device {device_name} for user {username}",
618        session.user.username,
619    );
620
621    let user = user_for_admin_or_self(&appstate.pool, &session, &username).await?;
622
623    // Let admins manage devices for disabled users
624    if !user.is_active && !session.is_admin {
625        info!(
626            "User {} tried to add a device for a disabled user {username}",
627            session.user.username
628        );
629
630        return Err(WebError::Forbidden("User is disabled.".into()));
631    }
632
633    let networks = WireguardNetwork::all(&appstate.pool).await?;
634    if networks.is_empty() {
635        error!("Failed to add device {device_name}, no networks found");
636        return Ok(ApiResponse {
637            json: json!({}),
638            status: StatusCode::BAD_REQUEST,
639        });
640    }
641
642    Device::validate_pubkey(&add_device.wireguard_pubkey).map_err(WebError::PubkeyValidation)?;
643
644    // Make sure there is no device with the same pubkey, such state may lead to unexpected issues
645    if Device::find_by_pubkey(&appstate.pool, &add_device.wireguard_pubkey)
646        .await?
647        .is_some()
648    {
649        return Err(WebError::PubkeyExists(format!(
650            "Failed to add device {device_name}, identical pubkey ({}) already exists",
651            add_device.wireguard_pubkey
652        )));
653    }
654
655    // save device
656    let mut transaction = appstate.pool.begin().await?;
657    let device = Device::new(
658        add_device.name,
659        add_device.wireguard_pubkey,
660        user.id,
661        DeviceType::User,
662        None,
663        true,
664    )
665    .save(&mut *transaction)
666    .await?;
667
668    let (network_info, configs) = device.add_to_all_networks(&mut transaction).await?;
669
670    // prepare a list of gateway events to be sent
671    let mut events = Vec::new();
672
673    // get all locations affected by device being added
674    let mut affected_location_ids = HashSet::new();
675    for network_info_item in network_info.clone() {
676        affected_location_ids.insert(network_info_item.network_id);
677    }
678
679    // send firewall config updates to affected locations
680    // if they have ACL enabled & enterprise features are active
681    for location_id in affected_location_ids {
682        if let Some(location) = WireguardNetwork::find_by_id(&mut *transaction, location_id).await?
683        {
684            if let Some(firewall_config) =
685                location.try_get_firewall_config(&mut transaction).await?
686            {
687                debug!("Sending firewall config update for location {location} affected by adding new user {username} devices");
688                events.push(GatewayEvent::FirewallConfigChanged(
689                    location_id,
690                    firewall_config,
691                ));
692            }
693        }
694    }
695
696    // add peer on relevant gateways
697    events.push(GatewayEvent::DeviceCreated(DeviceInfo {
698        device: device.clone(),
699        network_info: network_info.clone(),
700    }));
701
702    appstate.send_multiple_wireguard_events(events);
703
704    transaction.commit().await?;
705
706    let template_locations: Vec<TemplateLocation> = configs
707        .iter()
708        .map(|c| TemplateLocation {
709            name: c.network_name.clone(),
710            assigned_ip: c.address.to_string(),
711        })
712        .collect();
713
714    // hide session info if triggered by admin for other user
715    let (session_ip, session_device_info) = if session.is_admin && session.user != user {
716        (None, None)
717    } else {
718        (
719            Some(session.session.ip_address.as_str()),
720            session.session.device_info.clone(),
721        )
722    };
723    send_new_device_added_email(
724        &device.name,
725        &device.wireguard_pubkey,
726        &template_locations,
727        &user.email,
728        &appstate.mail_tx,
729        session_ip,
730        session_device_info.as_deref(),
731    )?;
732
733    info!(
734        "User {} added device {device_name} for user {username}",
735        session.user.username
736    );
737
738    let result = AddDeviceResult { configs, device };
739
740    update_counts(&appstate.pool).await?;
741
742    Ok(ApiResponse {
743        json: json!(result),
744        status: StatusCode::CREATED,
745    })
746}

784pub(crate) async fn modify_device(
785    _can_manage_devices: CanManageDevices,
786    session: SessionInfo,
787    Path(device_id): Path<i64>,
788    State(appstate): State<AppState>,
789    Json(data): Json<ModifyDevice>,
790) -> ApiResult {
791    debug!("User {} updating device {device_id}", session.user.username);
792    let mut device = device_for_admin_or_self(&appstate.pool, &session, device_id).await?;
793    let networks = WireguardNetwork::all(&appstate.pool).await?;
794
795    if networks.is_empty() {
796        error!("Failed to update device {device_id}, no networks found");
797        return Ok(ApiResponse {
798            json: json!({}),
799            status: StatusCode::BAD_REQUEST,
800        });
801    }
802
803    // check pubkeys
804    for network in &networks {
805        if network.pubkey == data.wireguard_pubkey {
806            error!("Failed to update device {device_id}, device's pubkey must be different from server's pubkey");
807            return Ok(ApiResponse {
808                json: json!({"msg": "device's pubkey must be different from server's pubkey"}),
809                status: StatusCode::BAD_REQUEST,
810            });
811        }
812    }
813
814    // update device info
815    device.update_from(data);
816    device.save(&appstate.pool).await?;
817
818    // send update to gateway's
819    let mut network_info = Vec::new();
820    for network in &networks {
821        let wireguard_network_device =
822            WireguardNetworkDevice::find(&appstate.pool, device.id, network.id).await?;
823        if let Some(wireguard_network_device) = wireguard_network_device {
824            let device_network_info = DeviceNetworkInfo {
825                network_id: network.id,
826                device_wireguard_ip: wireguard_network_device.wireguard_ip,
827                preshared_key: wireguard_network_device.preshared_key,
828                is_authorized: wireguard_network_device.is_authorized,
829            };
830            network_info.push(device_network_info);
831        }
832    }
833    appstate.send_wireguard_event(GatewayEvent::DeviceModified(DeviceInfo {
834        device: device.clone(),
835        network_info,
836    }));
837
838    info!("User {} updated device {device_id}", session.user.username);
839    Ok(ApiResponse {
840        json: json!(device),
841        status: StatusCode::OK,
842    })
843}

1105/// Returns appropriate aggregation level depending on the `from` date param
1106/// If `from` is >= than 6 hours ago, returns `Hour` aggregation
1107/// Otherwise returns `Minute` aggregation
1108fn get_aggregation(from: NaiveDateTime) -> Result<DateTimeAggregation, StatusCode> {
1109    // Use hourly aggregation for longer periods
1110    let aggregation = match Utc::now().naive_utc() - from {
1111        duration if duration >= TimeDelta::hours(6) => Ok(DateTimeAggregation::Hour),
1112        duration if duration < TimeDelta::zero() => Err(StatusCode::BAD_REQUEST),
1113        _ => Ok(DateTimeAggregation::Minute),
1114    }?;
1115    Ok(aggregation)
1116}

1123impl QueryFrom {
1124    /// If `datetime` is Some, parses the date string, otherwise returns `DateTime` one hour ago.

1126        Ok(match &self.from {
1127            Some(from) => DateTime::<Utc>::from_str(from).map_err(|_| StatusCode::BAD_REQUEST)?,
1128            None => Utc::now() - TimeDelta::hours(1),
1129        })
1130    }
