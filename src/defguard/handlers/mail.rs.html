50/// Handles logging the error and returns ApiResponse that contains it
51fn internal_error(to: &str, subject: &str, error: &impl Display) -> ApiResponse {
52    error!("Error sending mail to {to}, subject: {subject}, error: {error}");
53    ApiResponse {
54        json: json!({
55            "error": error.to_string(),
56        }),
57        status: StatusCode::INTERNAL_SERVER_ERROR,
58    }
59}

61pub async fn test_mail(
62    _admin: AdminRole,
63    session: SessionInfo,
64    State(appstate): State<AppState>,
65    Json(data): Json<TestMail>,
66) -> ApiResult {
67    debug!(
68        "User {} sending test mail to {}",
69        session.user.username, data.to
70    );
71
72    let (tx, mut rx) = unbounded_channel();
73    let mail = Mail {
74        to: data.to.clone(),
75        subject: TEST_MAIL_SUBJECT.to_string(),
76        content: templates::test_mail(Some(&session.session))?,
77        attachments: Vec::new(),
78        result_tx: Some(tx),
79    };
80    let (to, subject) = (mail.to.clone(), mail.subject.clone());
81    match appstate.mail_tx.send(mail) {
82        Ok(()) => match rx.recv().await {
83            Some(Ok(_)) => {
84                info!(
85                    "User {} sent test mail to {}",
86                    session.user.username, data.to
87                );
88                Ok(ApiResponse {
89                    json: json!({}),
90                    status: StatusCode::OK,
91                })
92            }
93            Some(Err(err)) => Ok(internal_error(&to, &subject, &err)),
94            None => Ok(internal_error(
95                &to,
96                &subject,
97                &String::from("None received"),
98            )),
99        },
100        Err(err) => Ok(internal_error(&to, &subject, &err)),
101    }
102}

104async fn read_logs() -> String {
105    let Some(path) = &server_config().log_file else {
106        return "Log file not configured".to_string();
107    };
108
109    match read_to_string(path).await {
110        Ok(logs) => logs,
111        Err(err) => {
112            error!("Error dumping app logs: {err}");
113            format!("Error dumping app logs: {err}")
114        }
115    }
116}

118pub async fn send_support_data(
119    _admin: AdminRole,
120    session: SessionInfo,
121    State(appstate): State<AppState>,
122) -> ApiResult {
123    debug!(
124        "User {} sending support mail to {SUPPORT_EMAIL_ADDRESS}",
125        session.user.username
126    );
127    let config = dump_config(&appstate.pool).await;
128    let config =
129        serde_json::to_string_pretty(&config).unwrap_or("Json formatting error".to_string());
130    let config = Attachment {
131        filename: format!("defguard-support-data-{}.json", Utc::now()),
132        content: config.into(),
133        content_type: ContentType::TEXT_PLAIN,
134    };
135    let logs = read_logs().await;
136    let logs = Attachment {
137        filename: format!("defguard-logs-{}.txt", Utc::now()),
138        content: logs.into(),
139        content_type: ContentType::TEXT_PLAIN,
140    };
141    let (tx, mut rx) = unbounded_channel();
142    let mail = Mail {
143        to: SUPPORT_EMAIL_ADDRESS.to_string(),
144        subject: SUPPORT_EMAIL_SUBJECT.to_string(),
145        content: support_data_mail()?,
146        attachments: vec![config, logs],
147        result_tx: Some(tx),
148    };
149    let (to, subject) = (mail.to.clone(), mail.subject.clone());
150    match appstate.mail_tx.send(mail) {
151        Ok(()) => match rx.recv().await {
152            Some(Ok(_)) => {
153                info!(
154                    "User {} sent support mail to {SUPPORT_EMAIL_ADDRESS}",
155                    session.user.username
156                );
157                Ok(ApiResponse {
158                    json: json!({}),
159                    status: StatusCode::OK,
160                })
161            }
162            Some(Err(err)) => Ok(internal_error(&to, &subject, &err)),
163            None => Ok(internal_error(
164                &to,
165                &subject,
166                &String::from("None received"),
167            )),
168        },
169        Err(err) => Ok(internal_error(&to, &subject, &err)),
170    }
171}
