12/// Unwraps the result returning a JSON representation of value or error
13fn unwrap_json<S: Serialize, D: Display>(result: Result<S, D>) -> Value {
14    match result {
15        Ok(value) => to_value(value).expect("conversion to JSON failed"),
16        Err(err) => json!({"error": err.to_string()}),
17    }
18}
19
20/// Dumps all data that could be used for debugging.
21pub async fn dump_config(db: &PgPool) -> Value {
22    // App settings DB records
23    let settings = match Settings::get(db).await {
24        Ok(Some(mut settings)) => {
25            settings.smtp_password = None;
26            json!(settings)
27        }
28        Ok(None) => json!({"error": "Settings not found"}),
29        Err(err) => json!({"error": err.to_string()}),
30    };
31    // Networks
32    let (networks, devices) = match WireguardNetwork::all(db).await {
33        Ok(networks) => {
34            // Devices for each network
35            let mut devices = HashMap::<Id, Value>::new();
36            for network in &networks {
37                devices.insert(
38                    network.id,
39                    unwrap_json(WireguardNetworkDevice::all_for_network(db, network.id).await),
40                );
41            }
42            (
43                to_value(networks).expect("JSON serialization error"),
44                to_value(devices).expect("JSON serialization error"),
45            )
46        }
47        Err(err) => (json!({"error": err.to_string()}), Value::Null),
48    };
49    let users_diagnostic_data = unwrap_json(User::all_without_sensitive_data(db).await);
50
51    json!({
52        "settings": settings,
53        "networks": networks,
54        "version": VERSION,
55        "devices": devices,
56        "users": users_diagnostic_data,
57        "config": server_config(),
58    })
59}
