22// Times in seconds
23const UTILITY_THREAD_MAIN_SLEEP_TIME: u64 = 5;
24const COUNT_UPDATE_INTERVAL: u64 = 60 * 60;
25const UPDATES_CHECK_INTERVAL: u64 = 60 * 60 * 6;
26const EXPIRED_ACL_RULES_CHECK_INTERVAL: u64 = 60 * 5;
27const ENTERPRISE_STATUS_CHECK_INTERVAL: u64 = 60 * 5;

29#[instrument(skip_all)]
30pub async fn run_utility_thread(
31    pool: &PgPool,
32    wireguard_tx: Sender<GatewayEvent>,
33) -> Result<(), anyhow::Error> {
34    let mut last_count_update = Instant::now();
35    let mut last_directory_sync = Instant::now();
36    let mut last_updates_check = Instant::now();
37    let mut last_ldap_sync = Instant::now();
38    let mut last_expired_acl_rules_check = Instant::now();
39    let mut last_enterprise_status_check = Instant::now();
40
41    // helper variable which stores previous enterprise features status
42    let mut enterprise_enabled = is_enterprise_enabled();
43
44    let directory_sync_task = || async {
45        if let Err(e) = do_directory_sync(pool, &wireguard_tx)
46            .instrument(info_span!("directory_sync_task"))
47            .await
48        {
49            error!("There was an error while performing directory sync job: {e:?}",);
50        }
51    };
52
53    let count_update_task = || async {
54        if let Err(e) = do_count_update(pool)
55            .instrument(info_span!("count_update_task"))
56            .await
57        {
58            error!("There was an error while performing count update job: {e:?}");
59        }
60    };
61
62    let updates_check_task = || async {
63        if let Err(e) = do_new_version_check()
64            .instrument(info_span!("updates_check_task"))
65            .await
66        {
67            error!("There was an error while checking for new Defguard version: {e:?}");
68        }
69    };
70
71    let ldap_sync_task = || async {
72        if let Err(e) = do_ldap_sync(pool)
73            .instrument(info_span!("ldap_sync_task"))
74            .await
75        {
76            error!("There was an error while performing LDAP sync job: {e}");
77        }
78    };
79
80    let expired_acl_rules_task = || async {
81        if let Err(err) = expired_acl_rules_check(pool, wireguard_tx.clone())
82            .instrument(info_span!("expired_acl_rules_task"))
83            .await
84        {
85            error!("Failed to check expired ACL rules: {err}");
86        };
87    };
88
89    directory_sync_task().await;
90    count_update_task().await;
91    updates_check_task().await;
92    ldap_sync_task().await;
93    expired_acl_rules_task().await;
94
95    loop {
96        sleep(Duration::from_secs(UTILITY_THREAD_MAIN_SLEEP_TIME)).await;
97
98        // Count update job for updating device/user/network counts
99        if last_count_update.elapsed().as_secs() >= COUNT_UPDATE_INTERVAL {
100            count_update_task().await;
101            last_count_update = Instant::now();
102        }
103
104        // Directory sync job for syncing with the directory service
105        if last_directory_sync.elapsed().as_secs() >= get_directory_sync_interval(pool).await {
106            directory_sync_task().await;
107            last_directory_sync = Instant::now();
108        }
109
110        // Check for new Defguard version
111        if last_updates_check.elapsed().as_secs() >= UPDATES_CHECK_INTERVAL {
112            updates_check_task().await;
113            last_updates_check = Instant::now();
114        }
115
116        // Perform LDAP sync
117        if last_ldap_sync.elapsed().as_secs() >= get_ldap_sync_interval() {
118            ldap_sync_task().await;
119            last_ldap_sync = Instant::now();
120        }
121
122        // Mark expired ACL rules
123        if last_expired_acl_rules_check.elapsed().as_secs() >= EXPIRED_ACL_RULES_CHECK_INTERVAL {
124            expired_acl_rules_task().await;
125            last_expired_acl_rules_check = Instant::now();
126        }
127
128        // Check if enterprise features got enabled or disabled
129        if last_enterprise_status_check.elapsed().as_secs() >= ENTERPRISE_STATUS_CHECK_INTERVAL {
130            let new_enterprise_enabled = is_enterprise_enabled();
131            if let Err(err) = enterprise_status_check(
132                pool,
133                wireguard_tx.clone(),
134                enterprise_enabled,
135                new_enterprise_enabled,
136            )
137            .instrument(info_span!("enterprise_status_check"))
138            .await
139            {
140                error!("Failed to check enterprise status: {err}");
141            } else {
142                // update status
143                enterprise_enabled = new_enterprise_enabled;
144            };
145            last_enterprise_status_check = Instant::now();
146        }
147    }
148}

150/// Check if enterprise status has changed and perform any necessary actions
151async fn enterprise_status_check(
152    pool: &PgPool,
153    wireguard_tx: Sender<GatewayEvent>,
154    current_enterprise_enabled: bool,
155    new_enterprise_enabled: bool,
156) -> Result<(), anyhow::Error> {
157    if new_enterprise_enabled != current_enterprise_enabled {
158        debug!("Enterprise feature status changed from {current_enterprise_enabled} to {new_enterprise_enabled}");
159
160        // fetch all ACL-enabled networks
161        let locations: Vec<WireguardNetwork<Id>> = WireguardNetwork::all(pool)
162            .await?
163            .into_iter()
164            .filter(|location| location.acl_enabled)
165            .collect();
166
167        if new_enterprise_enabled {
168            // handle switch from disabled -> enabled
169            debug!("Re-enabling gateway firewall configuration for ACL-enabled locations");
170            let mut conn = pool.acquire().await?;
171            for location in locations {
172                debug!("Re-enabling gateway firewall configuration for location {location:?}");
173                let firewall_config = location
174                    .try_get_firewall_config(&mut conn)
175                    .await?
176                    .expect("ACL-enabled location must have firewall config");
177
178                wireguard_tx.send(GatewayEvent::FirewallConfigChanged(
179                    location.id,
180                    firewall_config,
181                ))?;
182            }
183        } else {
184            // handle switch from enabled -> disabled
185            debug!("Disabling gateway firewall configuration for ACL-enabled locations");
186            for location in locations {
187                debug!("Disabling gateway firewall configuration for location {location:?}");
188                wireguard_tx.send(GatewayEvent::FirewallDisabled(location.id))?;
189            }
190        }
191    };
192    Ok(())
193}

195/// Find newly expired ACL rules and update their status.
196async fn expired_acl_rules_check(
197    pool: &PgPool,
198    wireguard_tx: Sender<GatewayEvent>,
199) -> Result<(), anyhow::Error> {
200    // mark relevant rules as expired
201    let updated_rules = query_as!(
202            AclRule::<Id>,
203            "UPDATE aclrule SET state = 'expired'::aclrule_state \
204            WHERE state = 'applied'::aclrule_state AND expires < NOW() \
205            RETURNING id, parent_id, state AS \"state: RuleState\", name, allow_all_users, deny_all_users, \
206                allow_all_network_devices, deny_all_network_devices, all_networks, \
207                destination, ports, protocols, enabled, expires"
208        )
209        .fetch_all(pool)
210        .await?;
211
212    // send firewall config updates to locations which have been affected by updated
213    // rules
214    debug!(
215        "Marked {} ACL rules as expired. Sending firewall config updates to affected locations.",
216        updated_rules.len()
217    );
218
219    // find affected locations
220    let mut affected_locations = HashSet::new();
221    for rule in updated_rules {
222        let locations = rule.get_networks(pool).await?;
223        for location in locations {
224            affected_locations.insert(location);
225        }
226    }
227
228    let affected_locations: Vec<WireguardNetwork<Id>> = affected_locations.into_iter().collect();
229    debug!(
230            "{} locations affected by expired ACL rules. Sending gateway firewall update events for each location",
231            affected_locations.len()
232        );
233
234    let mut conn = pool.acquire().await?;
235    for location in affected_locations {
236        match location.try_get_firewall_config(&mut conn).await? {
237            Some(firewall_config) => {
238                debug!("Sending firewall update event for location {location}");
239                wireguard_tx.send(GatewayEvent::FirewallConfigChanged(
240                    location.id,
241                    firewall_config,
242                ))?;
243            }
244            None => {
245                debug!("No firewall config generated for location {location}. Not sending a gateway event")
246            }
247        }
248    }
249
250    Ok(())
251}
