6#[derive(Debug, PartialEq)]
7pub enum HexError {
8    InvalidCharacter(u8),
9    InvalidStringLength(usize),
10}
11
12impl Error for HexError {}
13
14impl Display for HexError {
15    fn fmt(&self, f: &mut Formatter) -> FmtResult {
16        match self {
17            Self::InvalidCharacter(char) => {
18                write!(f, "Invalid character {char}")
19            }
20            Self::InvalidStringLength(length) => write!(f, "Invalid string length {length}"),
21        }
22    }
23}

25pub fn hex_decode<T: AsRef<[u8]>>(hex: T) -> Result<Vec<u8>, HexError> {
26    let mut hex = hex.as_ref();
27    let mut length = hex.len();
28    if length == 0 || length % 2 != 0 {
29        return Err(HexError::InvalidStringLength(length));
30    }
31
32    if length > 2 && hex[0] == b'0' && (hex[1] == b'x' || hex[1] == b'X') {
33        length -= 2;
34        hex = &hex[2..];
35    }
36
37    let hex_value = |char: u8| -> Result<u8, HexError> {
38        match char {
39            b'A'..=b'F' => Ok(char - b'A' + 10),
40            b'a'..=b'f' => Ok(char - b'a' + 10),
41            b'0'..=b'9' => Ok(char - b'0'),
42            _ => Err(HexError::InvalidCharacter(char)),
43        }
44    };
45
46    let mut bytes = Vec::with_capacity(length / 2);
47    for chunk in hex.chunks(2) {
48        let msd = hex_value(chunk[0])?;
49        let lsd = hex_value(chunk[1])?;
50        bytes.push((msd << 4) | lsd);
51    }
52
53    Ok(bytes)
54}

57pub fn to_lower_hex(bytes: &[u8]) -> String {
58    let mut hex = String::with_capacity(bytes.len() + 2);
59    let to_char = |nibble: u8| -> char {
60        (match nibble {
61            0..=9 => b'0' + nibble,
62            _ => nibble + b'a' - 10,
63        }) as char
64    };
65    for byte in bytes {
66        hex.push(to_char(*byte >> 4));
67        hex.push(to_char(*byte & 0xf));
68    }
69    hex
70}
