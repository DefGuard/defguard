41/// Subset of Settings object representing SMTP configuration
42struct SmtpSettings {
43    pub server: String,
44    pub port: u16,
45    pub encryption: SmtpEncryption,
46    pub user: String,
47    pub password: String,
48    pub sender: String,
49}
50
51impl SmtpSettings {
52    /// Retrieves `Settings` from database and builds `SmtpSettings`.
53    pub async fn get(db: &PgPool) -> Result<Self, MailError> {
54        Self::from_settings(Self::get_settings(db).await?)
55    }
56
57    /// Constructs `SmtpSettings` from `Settings`. Returns error if `SmtpSettings` are incomplete.
58    pub fn from_settings(settings: Settings) -> Result<SmtpSettings, MailError> {
59        if let (Some(server), Some(port), encryption, Some(user), Some(password), Some(sender)) = (
60            settings.smtp_server,
61            settings.smtp_port,
62            settings.smtp_encryption,
63            settings.smtp_user,
64            settings.smtp_password,
65            settings.smtp_sender,
66        ) {
67            let port = port.try_into().map_err(|_| MailError::InvalidPort(port))?;
68            Ok(Self {
69                server,
70                port,
71                encryption,
72                user,
73                password: password.expose_secret().to_string(),
74                sender,
75            })
76        } else {
77            Err(MailError::SmtpNotConfigured)
78        }
79    }
80
81    /// Retrieves Settings object from database
82    async fn get_settings(db: &PgPool) -> Result<Settings, MailError> {
83        Settings::get(db).await?.ok_or(MailError::EmptySettings)
84    }
85}

96#[derive(Debug)]
97pub struct Attachment {
98    pub filename: String,
99    pub content: Vec<u8>,
100    pub content_type: ContentType,
101}
102
103impl From<Attachment> for SinglePart {
104    fn from(attachment: Attachment) -> Self {
105        lettre::message::Attachment::new(attachment.filename)
106            .body(attachment.content, attachment.content_type)
107    }
108}
109
110impl Mail {
111    /// Converts Mail to lettre Message
112    fn into_message(self, from: &str) -> Result<Message, MailError> {
113        let builder = Message::builder()
114            .from(Self::mailbox(from)?)
115            .to(Self::mailbox(&self.to)?)
116            .subject(self.subject.clone());
117        match self.attachments {
118            attachments if attachments.is_empty() => Ok(builder
119                .header(ContentType::TEXT_HTML)
120                .body(self.content.clone())?),
121            attachments => {
122                let mut multipart = MultiPart::mixed().singlepart(SinglePart::html(self.content));
123                for attachment in attachments {
124                    multipart = multipart.singlepart(attachment.into());
125                }
126                Ok(builder.multipart(multipart)?)
127            }
128        }
129    }
130
131    /// Builds Mailbox structure from string representing email address
132    fn mailbox(address: &str) -> Result<Mailbox, MailError> {
133        if let Some((user, domain)) = address.split_once('@') {
134            if !(user.is_empty() || domain.is_empty()) {
135                return Ok(Mailbox::new(None, Address::new(user, domain)?));
136            }
137        }
138        Err(AddressError::MissingParts)?
139    }
140}

142struct MailHandler {
143    rx: UnboundedReceiver<Mail>,
144    db: PgPool,
145}
146
147impl MailHandler {
148    pub fn new(rx: UnboundedReceiver<Mail>, db: PgPool) -> Self {
149        Self { rx, db }
150    }
151
152    pub fn send_result(
153        tx: Option<UnboundedSender<Result<Response, MailError>>>,
154        result: Result<Response, MailError>,
155    ) {
156        if let Some(tx) = tx {
157            if tx.send(result).is_ok() {
158                debug!("SMTP result sent back to caller");
159            } else {
160                error!("Error sending SMTP result back to caller");
161            }
162        }
163    }
164
165    /// Listens on rx channel for messages and sends them via SMTP.
166    pub async fn run(mut self) {
167        while let Some(mail) = self.rx.recv().await {
168            let (to, subject) = (mail.to.clone(), mail.subject.clone());
169            debug!("Sending mail to: {to}, subject: {subject}");
170            let settings = match SmtpSettings::get(&self.db).await {
171                Ok(settings) => settings,
172                Err(MailError::SmtpNotConfigured) => {
173                    warn!("SMTP not configured, email sending skipped");
174                    continue;
175                }
176                Err(err) => {
177                    error!("Error retrieving SMTP settings: {err}");
178                    continue;
179                }
180            };
181
182            // Construct lettre Message
183            let result_tx = mail.result_tx.clone();
184            let message: Message = match mail.into_message(&settings.sender) {
185                Ok(message) => message,
186                Err(err) => {
187                    error!("Failed to build message to: {to}, subject: {subject}, error: {err}");
188                    continue;
189                }
190            };
191            // Build mailer and send the message
192            match Self::mailer(settings) {
193                Ok(mailer) => match mailer.send(message).await {
194                    Ok(response) => {
195                        Self::send_result(result_tx, Ok(response.clone()));
196                        info!("Mail sent successfully to: {to}, subject: {subject}, response: {response:?}");
197                    }
198                    Err(err) => {
199                        error!("Mail sending failed to: {to}, subject: {subject}, error: {err}");
200                        Self::send_result(result_tx, Err(MailError::SmtpError(err)));
201                    }
202                },
203                Err(MailError::SmtpNotConfigured) => {
204                    warn!("SMTP not configured, onboarding email sending skipped");
205                    Self::send_result(result_tx, Err(MailError::SmtpNotConfigured));
206                }
207                Err(err) => {
208                    error!("Error building mailer: {err}");
209                    Self::send_result(result_tx, Err(err));
210                }
211            }
212        }
213    }
214
215    /// Builds mailer object with specified configuration
216    fn mailer(settings: SmtpSettings) -> Result<AsyncSmtpTransport<Tokio1Executor>, MailError> {
217        let builder = match settings.encryption {
218            SmtpEncryption::None => {
219                AsyncSmtpTransport::<Tokio1Executor>::builder_dangerous(settings.server)
220            }
221            SmtpEncryption::StartTls => {
222                AsyncSmtpTransport::<Tokio1Executor>::starttls_relay(&settings.server)?
223            }
224            SmtpEncryption::ImplicitTls => {
225                AsyncSmtpTransport::<Tokio1Executor>::relay(&settings.server)?
226            }
227        }
228        .port(settings.port)
229        .timeout(Some(Duration::from_secs(SMTP_TIMEOUT_SECONDS)));
230
231        // Skip credentials if any of them is empty
232        let builder = if settings.user.is_empty() || settings.password.is_empty() {
233            debug!("SMTP credentials were not provided, skipping username/password authentication");
234            builder
235        } else {
236            builder.credentials(Credentials::new(settings.user, settings.password))
237        };
238
239        Ok(builder.build())
240    }
241}

246    MailHandler::new(rx, db).run().await;
