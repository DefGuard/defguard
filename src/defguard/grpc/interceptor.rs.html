1use tonic::{service::Interceptor, Status};
2
3use crate::auth::{Claims, ClaimsType};
4
5/// Auth interceptor used by gRPC services. Verifies JWT token sent
6/// in gRPC metadata under "authorization" key.
7#[derive(Clone)]
8pub struct JwtInterceptor {
9    claims_type: ClaimsType,
10}
11
12impl JwtInterceptor {
13    #[must_use]
14    pub fn new(claims_type: ClaimsType) -> Self {
15        Self { claims_type }
16    }
17}
18
19impl Interceptor for JwtInterceptor {
20    fn call(&mut self, mut req: tonic::Request<()>) -> Result<tonic::Request<()>, Status> {
21        // This is only used for logging purposes, so no proper error handling
22        let hostname = req
23            .metadata()
24            .get("hostname")
25            .map_or("UNKNOWN", |h| h.to_str().unwrap_or("UNKNOWN"));
26
27        let token = match req.metadata().get("authorization") {
28            Some(token) => token.to_str().map_err(|err| {
29                warn!(
30                    "Failed to parse authorization header during handling gRPC request from \
31                    hostname {hostname}. If you recognize this hostname, there may be an issue \
32                    with the token used for authorization. Cause of the failed parsing: {err:?}"
33                );
34                Status::unauthenticated("Invalid token")
35            })?,
36            None => return Err(Status::unauthenticated("Missing authorization header")),
37        };
38
39        match Claims::from_jwt(self.claims_type, token) {
40            Ok(claims) => {
41                let request_metadata = req.metadata_mut();
42
43                if let ClaimsType::Gateway = self.claims_type {
44                    request_metadata.insert(
45                        "gateway_network_id",
46                        claims
47                            .client_id
48                            .parse()
49                            .map_err(|_| Status::unknown("Network ID parsing error"))?,
50                    );
51                }
52
53                // FIXME: can we push whole Claims object into metadata?
54                request_metadata.insert(
55                    "username",
56                    claims
57                        .sub
58                        .parse()
59                        .map_err(|_| Status::unknown("Username parsing error"))?,
60                );
61                debug!("Authorization successful for user {}", claims.sub);
62                Ok(req)
63            }
64            Err(err) => {
65                warn!(
66                    "Failed to authorize a gRPC request from hostname {hostname}. If you recognize \
67                    this hostname, there may be an issue with the token used for authorization. \
68                    Cause of the failed authorization: {err:?}"
69                );
70                Err(Status::unauthenticated("Invalid token"))
71            }
72        }
73    }
74}
