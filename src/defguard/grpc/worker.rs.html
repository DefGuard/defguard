22impl WorkerInfo {
23    /// Create new `Worker` instance.
24    #[must_use]
25    pub fn new() -> Self {
26        Self {
27            last_seen: Instant::now(),
28            ip: IpAddr::V4(Ipv4Addr::UNSPECIFIED),
29            jobs: Vec::new(),
30        }
31    }
32
33    /// Update connectivity timer.
34    pub fn refresh_status(&mut self) {
35        self.last_seen = Instant::now();
36    }
37
38    /// Connectivity status.
39    #[must_use]
40    pub fn connected(&self) -> bool {
41        self.last_seen.elapsed().as_secs() < 2
42    }
43
44    /// Return first availale Job.
45    #[must_use]
46    pub fn get_job(&self) -> Option<&Job> {
47        self.jobs.first()
48    }
49
50    /// Set worker ip
51    pub fn set_ip(&mut self, ip: IpAddr) {
52        self.ip = ip;
53    }
54
55    /// Add Job.
56    pub fn add_job(&mut self, job: Job) {
57        self.jobs.push(job);
58    }
59
60    /// Remove Job with given id.
61    pub fn remove_job_with_id(&mut self, job_id: u32) -> Option<Job> {
62        if let Some(index) = self.jobs.iter().position(|job| job.id == job_id) {
63            Some(self.jobs.remove(index))
64        } else {
65            None
66        }
67    }
68}

76impl WorkerState {
77    /// Return initial state.
78    #[must_use]
79    pub fn new(webhook_tx: UnboundedSender<AppEvent>) -> Self {
80        Self {
81            current_job_id: 1,
82            workers: HashMap::new(),
83            job_status: HashMap::new(),
84            webhook_tx,
85        }
86    }

88    /// Return `true` on success.
89    pub fn register_worker(&mut self, id: String) -> bool {
90        if let Entry::Vacant(entry) = self.workers.entry(id) {
91            entry.insert(WorkerInfo::new());
92            true
93        } else {
94            false
95        }
96    }
97
98    /// Create a new job.
99    /// Return job id.
100    pub fn create_job(
101        &mut self,
102        worker_id: &str,
103        first_name: String,
104        last_name: String,
105        email: String,
106        username: String,
107    ) -> u32 {
108        if let Some(worker) = self.workers.get_mut(worker_id) {
109            let id = self.current_job_id;
110            self.current_job_id = id.wrapping_add(1);
111            worker.add_job(Job {
112                id,
113                first_name,
114                last_name,
115                email,
116                username,
117            });
118            id
119        } else {
120            0
121        }
122    }
123
124    /// Remove a job for a given worker.
125    pub fn remove_job(&mut self, id: &str, job_id: u32) -> Option<Job> {
126        if let Some(worker) = self.workers.get_mut(id) {
127            worker.refresh_status();
128            worker.remove_job_with_id(job_id)
129        } else {
130            None
131        }
132    }
133
134    /// Return the first available job.
135    pub fn get_job(&mut self, id: &str, ip: IpAddr) -> Option<&Job> {
136        if let Some(worker) = self.workers.get_mut(id) {
137            worker.refresh_status();
138            worker.set_ip(ip);
139            worker.get_job()
140        } else {
141            None
142        }
143    }

149            let workers = WorkerDetail {

194#[tonic::async_trait]
195impl worker_service_server::WorkerService for WorkerServer {
196    async fn register_worker(&self, request: Request<Worker>) -> Result<Response<()>, Status> {
197        let message = request.into_inner();
198        let mut state = self.state.lock().unwrap();
199        if state.register_worker(String::from(&message.id)) {
200            debug!("Added worker with id: {}", message.id);
201            Ok(Response::new(()))
202        } else {
203            Err(Status::already_exists("Worker already registered"))
204        }
205    }
206
207    async fn get_job(&self, request: Request<Worker>) -> Result<Response<GetJobResponse>, Status> {
208        let ip = request
209            .remote_addr()
210            .map_or(IpAddr::V4(Ipv4Addr::UNSPECIFIED), |addr| addr.ip());
211        let message = request.into_inner();
212        let mut state = self.state.lock().unwrap();
213        if let Some(job) = state.get_job(&message.id, ip) {
214            Ok(Response::new(GetJobResponse {
215                first_name: job.first_name.clone(),
216                last_name: job.last_name.clone(),
217                email: job.email.clone(),
218                job_id: job.id,
219            }))
220        } else {
221            Err(Status::not_found("No more jobs"))
222        }
223    }
224
225    async fn set_job_done(&self, request: Request<JobStatus>) -> Result<Response<()>, Status> {
226        let message = request.into_inner();
227        info!(
228            "Marking job {} on worker {} as done.",
229            message.job_id, message.id
230        );
231        // Mutex manipulation is done explicitly in a separate block to avoid compiler errors
232        // https://github.com/rust-lang/rust/issues/57478
233        let username: Option<String> = {
234            let mut state = self.state.lock().unwrap();
235            // Remove job from worker
236            let job = state.remove_job(&message.id, message.job_id);
237            if let Some(job_done) = job {
238                state.set_job_status(message.clone(), job_done.username.clone());
239                if message.success {
240                    state
241                        .webhook_tx
242                        .send(AppEvent::HWKeyProvision(HWKeyUserData {
243                            username: job_done.username.clone(),
244                            email: job_done.email.clone(),
245                            ssh_key: message.ssh_key.clone(),
246                            pgp_key: message.public_key.clone(),
247                            serial: message.yubikey_serial.clone(),
248                        }))
249                        .expect("Failed to send event.");
250                }
251                Some(job_done.username)
252            } else {
253                None
254            }
255        };
256
257        if let Some(username) = username {
258            if message.success {
259                match User::find_by_username(&self.pool, &username).await {
260                    // TODO: Create respectable Authentication KEYS and Add yubikey entry to DB table "yubikey"
261                    Ok(Some(user)) => {
262                        // create yubikey
263                        // FIXME: pass name from user input this is temporary solution
264                        let yubi_count_res = query!(
265                            "SELECT COUNT(*) FROM \"yubikey\" WHERE user_id = $1",
266                            user.id
267                        )
268                        .fetch_one(&self.pool)
269                        .await
270                        .map_err(|_| Status::internal("Failed to count keys"))?;
271                        // FIXME: names may collide
272                        let name = match yubi_count_res.count {
273                            Some(count) => {
274                                let name = format!("YubiKey {}", count + 1);
275                                name
276                            }
277                            None => "YubiKey".to_string(),
278                        };
279                        let new_yubi = YubiKey::new(name, message.yubikey_serial, user.id)
280                            .save(&self.pool)
281                            .await
282                            .map_err(|_| Status::internal("Failed to save YubiKey"))?;
283                        let key_id = new_yubi.id;
284                        let ssh = AuthenticationKey::new(
285                            user.id,
286                            message.ssh_key,
287                            None,
288                            AuthenticationKeyType::Ssh,
289                            Some(key_id),
290                        );
291                        let gpg = AuthenticationKey::new(
292                            user.id,
293                            message.public_key,
294                            None,
295                            AuthenticationKeyType::Gpg,
296                            Some(key_id),
297                        );
298                        ssh.save(&self.pool)
299                            .await
300                            .map_err(|_| Status::internal("Failed to save auth key"))?;
301                        gpg.save(&self.pool)
302                            .await
303                            .map_err(|_| Status::internal("Failed to save auth key"))?;
304                    }
305                    Ok(None) => info!("User {username} not found"),
306                    Err(err) => error!("Error {err}"),
307                }
308            }
309        }
310
311        Ok(Response::new(()))
312    }
