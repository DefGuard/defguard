22const CLIENT_SESSION_TIMEOUT: u64 = 60 * 5; // 10 minutes
23
24struct ClientLoginSession {
25    method: MfaMethod,
26    location: WireguardNetwork<Id>,
27    device: Device<Id>,
28    user: User<Id>,
29}

31pub(super) struct ClientMfaServer {
32    pool: PgPool,
33    mail_tx: UnboundedSender<Mail>,
34    wireguard_tx: Sender<GatewayEvent>,
35    sessions: HashMap<String, ClientLoginSession>,
36}

38impl ClientMfaServer {
39    #[must_use]
40    pub fn new(
41        pool: PgPool,
42        mail_tx: UnboundedSender<Mail>,
43        wireguard_tx: Sender<GatewayEvent>,
44    ) -> Self {
45        Self {
46            pool,
47            mail_tx,
48            wireguard_tx,
49            sessions: HashMap::new(),
50        }
51    }

53    fn generate_token(pubkey: &str) -> Result<String, Status> {
54        Claims::new(
55            ClaimsType::DesktopClient,
56            String::new(),
57            pubkey.into(),
58            CLIENT_SESSION_TIMEOUT,
59        )
60        .to_jwt()
61        .map_err(|err| {
62            error!("Failed to generate JWT token: {err:?}");
63            Status::internal("unexpected error")
64        })
65    }
66
67    /// Validate JWT and extract client pubkey
68    fn parse_token(token: &str) -> Result<String, Status> {
69        let claims = Claims::from_jwt(ClaimsType::DesktopClient, token).map_err(|err| {
70            error!("Failed to parse JWT token: {err:?}");
71            Status::invalid_argument("invalid token")
72        })?;
73        Ok(claims.client_id)
74    }

76    #[instrument(skip_all)]
77    pub async fn start_client_mfa_login(
78        &mut self,
79        request: ClientMfaStartRequest,
80    ) -> Result<ClientMfaStartResponse, Status> {
81        debug!("Starting desktop client login: {request:?}");
82        // fetch location
83        let Ok(Some(location)) =
84            WireguardNetwork::find_by_id(&self.pool, request.location_id).await
85        else {
86            error!("Failed to find location with ID {}", request.location_id);
87            return Err(Status::invalid_argument("location not found"));
88        };
89
90        // fetch device
91        let Ok(Some(device)) = Device::find_by_pubkey(&self.pool, &request.pubkey).await else {
92            error!("Failed to find device with pubkey {}", request.pubkey);
93            return Err(Status::invalid_argument("device not found"));
94        };
95
96        // fetch user
97        let Ok(Some(mut user)) = User::find_by_id(&self.pool, device.user_id).await else {
98            error!("Failed to find user with ID {}", device.user_id);
99            return Err(Status::invalid_argument("user not found"));
100        };
101        let user_info = UserInfo::from_user(&self.pool, &user).await.map_err(|_| {
102            error!("Failed to fetch user info for {}", user.username);
103            Status::internal("unexpected error")
104        })?;
105
106        // validate user is allowed to connect to a given location
107        let mut transaction = self.pool.begin().await.map_err(|_| {
108            error!("Failed to begin transaction");
109            Status::internal("unexpected error")
110        })?;
111        let allowed_groups = location
112            .get_allowed_groups(&mut transaction)
113            .await
114            .map_err(|err| {
115                error!("Failed to fetch allowed groups for location {location}: {err:?}");
116                Status::internal("unexpected error")
117            })?;
118        if let Some(groups) = allowed_groups {
119            // check if user belongs to one of allowed groups
120            if !groups
121                .iter()
122                .any(|allowed_group| user_info.groups.contains(allowed_group))
123            {
124                error!(
125                    "User {} not allowed to connect to location {location} because he doesn't belong to any of the allowed groups.
126                    User groups: {:?}, allowed groups: {:?}",
127                    user.username, user_info.groups, groups
128                );
129                return Err(Status::unauthenticated("unauthorized"));
130            }
131        }
132
133        user.verify_mfa_state(&self.pool).await.map_err(|err| {
134            error!(
135                "Failed to verify MFA state for user {}: {err:?}",
136                user.username
137            );
138            Status::internal("unexpected error")
139        })?;
140
141        // check if selected method is enabled
142        let method = MfaMethod::try_from(request.method).map_err(|err| {
143            error!("Invalid MFA method selected ({}): {err}", request.method);
144            Status::invalid_argument("invalid MFA method selected")
145        })?;
146        match method {
147            MfaMethod::Totp => {
148                if !user.totp_enabled {
149                    error!("TOTP not enabled for user {}", user.username);
150                    return Err(Status::invalid_argument(
151                        "selected MFA method not available",
152                    ));
153                }
154            }
155            MfaMethod::Email => {
156                if !user.email_mfa_enabled {
157                    error!("Email MFA not enabled for user {}", user.username);
158                    return Err(Status::invalid_argument(
159                        "selected MFA method not available",
160                    ));
161                }
162                // send email code
163                send_email_mfa_code_email(&user, &self.mail_tx, None).map_err(|err| {
164                    error!(
165                        "Failed to send email MFA code for user {}: {err:?}",
166                        user.username
167                    );
168                    Status::internal("unexpected error")
169                })?;
170            }
171        };
172
173        // generate auth token
174        let token = Self::generate_token(&request.pubkey)?;
175
176        info!(
177            "Desktop client MFA login started for {} at location {}",
178            user.username, location.name
179        );
180
181        // store login session
182        self.sessions.insert(
183            request.pubkey,
184            ClientLoginSession {
185                method,
186                location,
187                device,
188                user,
189            },
190        );
191
192        Ok(ClientMfaStartResponse { token })
193    }

195    #[instrument(skip_all)]
196    pub async fn finish_client_mfa_login(
197        &mut self,
198        request: ClientMfaFinishRequest,
199    ) -> Result<ClientMfaFinishResponse, Status> {
200        debug!("Finishing desktop client login: {request:?}");
201        // get pubkey from token
202        let pubkey = Self::parse_token(&request.token)?;
203
204        // fetch login session
205        let Some(session) = self.sessions.get(&pubkey) else {
206            error!("Client login session not found");
207            return Err(Status::invalid_argument("login session not found"));
208        };
209        let ClientLoginSession {
210            method,
211            device,
212            location,
213            user,
214        } = session;
215
216        // validate code
217        match method {
218            MfaMethod::Totp => {
219                if !user.verify_totp_code(&request.code.to_string()) {
220                    error!("Provided TOTP code is not valid");
221                    return Err(Status::unauthenticated("unauthorized"));
222                }
223            }
224            MfaMethod::Email => {
225                if !user.verify_email_mfa_code(&request.code.to_string()) {
226                    error!("Provided email code is not valid");
227                    return Err(Status::unauthenticated("unauthorized"));
228                }
229            }
230        };
231
232        // begin transaction
233        let mut transaction = self.pool.begin().await.map_err(|_| {
234            error!("Failed to begin transaction");
235            Status::internal("unexpected error")
236        })?;
237
238        // fetch device config for the location
239        let Ok(Some(mut network_device)) =
240            WireguardNetworkDevice::find(&mut *transaction, device.id, location.id).await
241        else {
242            error!("Failed to fetch network config for device {device} and location {location}");
243            return Err(Status::internal("unexpected error"));
244        };
245
246        // generate PSK
247        let key = WireguardNetwork::genkey();
248        network_device.preshared_key = Some(key.public.clone());
249
250        // authorize device for given location
251        network_device.is_authorized = true;
252        network_device.authorized_at = Some(Utc::now().naive_utc());
253
254        // save updated network config
255        network_device
256            .update(&mut *transaction)
257            .await
258            .map_err(|err| {
259                error!("Failed to update device network config {network_device:?}: {err:?}");
260                Status::internal("unexpected error")
261            })?;
262
263        // send gateway event
264        debug!("Sending `peer_create` message to gateway");
265        let device_info = DeviceInfo {
266            device: device.clone(),
267            network_info: vec![DeviceNetworkInfo {
268                network_id: location.id,
269                device_wireguard_ip: network_device.wireguard_ip,
270                preshared_key: network_device.preshared_key,
271                is_authorized: network_device.is_authorized,
272            }],
273        };
274        let event = GatewayEvent::DeviceCreated(device_info);
275        self.wireguard_tx.send(event).map_err(|err| {
276            error!("Error sending WireGuard event: {err}");
277            Status::internal("unexpected error")
278        })?;
279
280        info!(
281            "Desktop client login finished for {} at location {}",
282            user.username, location.name
283        );
284
285        // remove login session from map
286        self.sessions.remove(&pubkey);
287
288        // commit transaction
289        transaction.commit().await.map_err(|_| {
290            error!("Failed to commit transaction while finishing desktop client login.");
291            Status::internal("unexpected error")
292        })?;
293
294        Ok(ClientMfaFinishResponse {
295            preshared_key: key.public,
296        })
297    }
