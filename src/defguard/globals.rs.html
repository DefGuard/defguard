1#[macro_export]
2/// Specify a global value that can be accessed from anywhere in the application.
3/// Positional arguments:
4/// - `$name`: The name of the global value. This will be the name of the variable that holds the value.
5/// - `$type`: The type of the global value.
6/// - `$init`: The initial value of the global value.
7/// - `$set_fn`: The name of the function that will be used to set the global value.
8/// - `$get_fn`: The name of the function that will be used to get the global value.
9///
10/// The macro will also automatically generate boilerplate code for unit tests to work correctly.
11macro_rules! global_value {
12    ($name:ident, $type:ty, $init:expr, $set_fn:ident, $get_fn:ident) => {
13        use std::sync::{RwLock, RwLockReadGuard};
14
15        static $name: RwLock<$type> = RwLock::new($init);
16
17        pub fn $set_fn(value: $type) {
18            *$name.write().expect("Failed to acquire lock on the mutex.") = value;
19        }
20
21        pub fn $get_fn() -> RwLockReadGuard<'static, $type> {
22            $name.read().expect("Failed to acquire lock on the mutex.")
23        }
24    };
25}
