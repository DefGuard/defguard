25#[must_use]
26pub(crate) fn get_user_agent_device(user_agent_client: &Client) -> String {
27    let device_type = user_agent_client
28        .device
29        .model
30        .as_ref()
31        .map_or("unknown model", Borrow::borrow);
32
33    let mut device_version = String::new();
34    if let Some(major) = &user_agent_client.os.major {
35        device_version.push_str(major);
36
37        if let Some(minor) = &user_agent_client.os.minor {
38            device_version.push('.');
39            device_version.push_str(minor);
40
41            if let Some(patch) = &user_agent_client.os.patch {
42                device_version.push('.');
43                device_version.push_str(patch);
44            }
45        }
46    }
47
48    let mut device_os = user_agent_client.os.family.to_string();
49    device_os.push(' ');
50    device_os.push_str(&device_version);
51    device_os.push_str(", ");
52    device_os.push_str(&user_agent_client.user_agent.family);
53
54    format!("{device_type}, OS: {device_os}")
55}

57fn get_user_agent_device_login_data(
58    user_id: Id,
59    ip_address: String,
60    event_type: String,
61    user_agent_client: &Client,
62) -> DeviceLoginEvent {
63    let model = user_agent_client
64        .device
65        .model
66        .as_ref()
67        .map(ToString::to_string);
68
69    let brand = user_agent_client
70        .device
71        .brand
72        .as_ref()
73        .map(ToString::to_string);
74
75    let family = user_agent_client.device.family.to_string();
76    let os_family = user_agent_client.os.family.to_string();
77    let browser = user_agent_client.user_agent.family.to_string();
78
79    DeviceLoginEvent::new(
80        user_id, ip_address, model, family, brand, os_family, browser, event_type,
81    )
82}

84pub(crate) async fn check_new_device_login(
85    pool: &PgPool,
86    mail_tx: &UnboundedSender<Mail>,
87    session: &Session,
88    user: &User<Id>,
89    ip_address: String,
90    event_type: String,
91    agent: Client<'_>,
92) -> Result<(), TemplateError> {
93    let device_login_event =
94        get_user_agent_device_login_data(user.id, ip_address, event_type, &agent);
95
96    if let Ok(Some(created_device_login_event)) = device_login_event
97        .check_if_device_already_logged_in(pool)
98        .await
99    {
100        send_new_device_login_email(
101            &user.email,
102            mail_tx,
103            session,
104            created_device_login_event.created,
105        )
106        .await?;
107    }
108
109    Ok(())
110}
