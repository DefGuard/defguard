278    impl Modify for SecurityAddon {
279        fn modify(&self, openapi: &mut utoipa::openapi::OpenApi) {
280            if let Some(components) = openapi.components.as_mut() {
281                // session cookie auth
282                components.add_security_scheme(
283                    "cookie",
284                    SecurityScheme::ApiKey(ApiKey::Cookie(ApiKeyValue::new(SESSION_COOKIE_NAME))),
285                );
286                // API token auth
287                components.add_security_scheme(
288                    "api_token",
289                    SecurityScheme::Http(HttpBuilder::new().scheme(HttpAuthScheme::Bearer).build()),
290                );
291            }
292        }
293    }

596/// Runs core web server exposing REST API.
597#[instrument(skip_all)]
598pub async fn run_web_server(
599    worker_state: Arc<Mutex<WorkerState>>,
600    gateway_state: Arc<Mutex<GatewayMap>>,
601    webhook_tx: UnboundedSender<AppEvent>,
602    webhook_rx: UnboundedReceiver<AppEvent>,
603    wireguard_tx: Sender<GatewayEvent>,
604    mail_tx: UnboundedSender<Mail>,
605    pool: PgPool,
606    failed_logins: Arc<Mutex<FailedLoginMap>>,
607) -> Result<(), anyhow::Error> {
608    let webapp = build_webapp(
609        webhook_tx,
610        webhook_rx,
611        wireguard_tx,
612        mail_tx,
613        worker_state,
614        gateway_state,
615        pool,
616        failed_logins,
617    );
618    info!("Started web services");
619    let addr = SocketAddr::new(IpAddr::V4(Ipv4Addr::UNSPECIFIED), server_config().http_port);
620    let listener = TcpListener::bind(&addr).await?;
621    serve(
622        listener,
623        webapp.into_make_service_with_connect_info::<SocketAddr>(),
624    )
625    .await
626    .map_err(|err| anyhow!("Web server can't be started {err}"))
627}

735/// Create a new VPN location.
736/// Meant to be used to automate setting up a new defguard instance.
737/// Does not handle assigning device IPs, since no device should exist at this point.
738pub async fn init_vpn_location(
739    pool: &PgPool,
740    args: &InitVpnLocationArgs,
741) -> Result<String, anyhow::Error> {
742    // check if a VPN location exists already
743    let networks = WireguardNetwork::all(pool).await?;
744    if !networks.is_empty() {
745        return Err(anyhow!(
746            "Failed to initialize first VPN location. A location already exists."
747        ));
748    };
749
750    // create a new network
751    let network = WireguardNetwork::new(
752        args.name.clone(),
753        vec![args.address],
754        args.port,
755        args.endpoint.clone(),
756        args.dns.clone(),
757        args.allowed_ips.clone(),
758        false,
759        DEFAULT_KEEPALIVE_INTERVAL,
760        DEFAULT_DISCONNECT_THRESHOLD,
761        false,
762        false,
763    )?
764    .save(pool)
765    .await?;
766
767    // generate gateway token
768    let token = Claims::new(
769        ClaimsType::Gateway,
770        format!("DEFGUARD-NETWORK-{}", network.id),
771        network.id.to_string(),
772        u32::MAX.into(),
773    )
774    .to_jwt()?;
775
776    Ok(token)
777}
