1use std::env;
2
3use chrono::NaiveDate;
4use semver::Version;
5
6use crate::global_value;
7
8const PRODUCT_NAME: &str = "Defguard";
9const UPDATES_URL: &str = "https://pkgs.defguard.net/api/update/check";
10const VERSION: &str = env!("CARGO_PKG_VERSION");
11
12#[derive(Deserialize, Debug, Serialize)]
13#[cfg_attr(test, derive(Clone))]
14pub struct Update {
15    version: String,
16    release_date: NaiveDate,
17    release_notes_url: String,
18    update_url: String,
19    critical: bool,
20    notes: String,
21}
22
23global_value!(NEW_UPDATE, Option<Update>, None, set_update, get_update);
24
25async fn fetch_update() -> Result<Update, anyhow::Error> {
26    let body = serde_json::json!({
27        "product": PRODUCT_NAME,
28        "client_version": VERSION,
29        "operating_system": env::consts::OS,
30    });
31    let response = reqwest::Client::new()
32        .post(UPDATES_URL)
33        .json(&body)
34        .send()
35        .await?;
36    Ok(response.json::<Update>().await?)
37}
38
39pub(crate) async fn do_new_version_check() -> Result<(), anyhow::Error> {
40    debug!("Checking for new version of Defguard ...");
41    let update = fetch_update().await?;
42    let current_version = Version::parse(VERSION)?;
43    let new_version = Version::parse(&update.version)?;
44    if new_version > current_version {
45        if update.critical {
46            warn!("There is a new critical Defguard update available: {} (Released on {}). It's recommended to update as soon as possible.",
47                  update.version, update.release_date);
48        } else {
49            info!(
50                "There is a new Defguard version available: {} (Released on {})",
51                update.version, update.release_date
52            );
53        }
54        set_update(Some(update));
55    } else {
56        debug!("New version check done. You are using the latest version of Defguard.");
57    }
58    Ok(())
59}
