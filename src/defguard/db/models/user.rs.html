172fn hash_password(password: &str) -> Result<String, HashError> {
173    let salt = SaltString::generate(&mut OsRng);
174    Ok(Argon2::default()
175        .hash_password(password.as_bytes(), &salt)?
176        .to_string())
177}

226    pub(crate) fn verify_password(&self, password: &str) -> Result<(), HashError> {
227        debug!("Checking if password matches for user {}", self.username);
228        if let Some(hash) = &self.password_hash {
229            let parsed_hash = PasswordHash::new(hash)?;
230            Argon2::default().verify_password(password.as_bytes(), &parsed_hash)
231        } else {
232            info!("User {} has no password set", self.username);
233            Err(HashError::Password)
234        }
235    }
236
237    #[must_use]
238    pub(crate) fn has_password(&self) -> bool {
239        self.password_hash.is_some()
240    }
241
242    #[must_use]
243    pub(crate) fn name(&self) -> String {
244        format!("{} {}", self.first_name, self.last_name)
245    }

247    /// Check if user is enrolled.
248    /// We assume the user is enrolled if they have a password set
249    /// or they have logged in using an external OIDC.
250    #[must_use]
251    pub(crate) fn is_enrolled(&self) -> bool {
252        self.password_hash.is_some() || self.openid_sub.is_some() || self.from_ldap
253    }
254
255    #[must_use]
256    pub(crate) fn ldap_rdn_value(&self) -> &str {
257        if let Some(ldap_rdn) = &self.ldap_rdn {
258            ldap_rdn
259        } else {
260            warn!(
261                "LDAP RDN is not set for user {}. Using username as a fallback.",
262                self.username
263            );
264            &self.username
265        }
266    }

333    /// - TOTP is enabled

335    async fn check_mfa_enabled<'e, E>(&self, executor: E) -> Result<bool, SqlxError>
336    where
337        E: PgExecutor<'e>,
338    {
339        // short-cut

341            return Ok(true);
342        }

344        query_scalar!(
345            "SELECT totp_enabled OR email_mfa_enabled \
346            OR count(webauthn.id) > 0 \"bool!\" FROM \"user\" \

348            WHERE \"user\".id = $1 GROUP BY totp_enabled, email_mfa_enabled;",
349            self.id
350        )

352        .await
353    }

355    /// Verify the state of mfa flags are correct.
356    /// Recovers from invalid mfa_method
357    /// Use this function after removing any of the authentication factors.
358    pub async fn verify_mfa_state(&mut self, pool: &PgPool) -> Result<(), WebError> {
359        if let Some(info) = MFAInfo::for_user(pool, self).await? {
360            let factors_present = info.mfa_available();
361            if self.mfa_enabled != factors_present {
362                // store correct value for MFA flag in the DB
363                if self.mfa_enabled {
364                    // last factor was removed so we have to disable MFA
365                    self.disable_mfa(pool).await?;
366                } else {
367                    // first factor was added so MFA needs to be enabled
368                    query!(
369                        "UPDATE \"user\" SET mfa_enabled = $2 WHERE id = $1",
370                        self.id,
371                        factors_present
372                    )
373                    .execute(pool)
374                    .await?;
375                };
376
377                if !factors_present && self.mfa_method != MFAMethod::None {
378                    debug!(
379                        "MFA for user {} disabled, updating MFA method to None",
380                        self.username
381                    );
382                    self.set_mfa_method(pool, MFAMethod::None).await?;
383                }
384
385                self.mfa_enabled = factors_present;
386            }
387
388            // set correct value for default method
389            if factors_present {
390                match info.list_available_methods() {
391                    None => {
392                        error!("Incorrect MFA info state for user {}", self.username);
393                        return Err(WebError::Http(StatusCode::INTERNAL_SERVER_ERROR));
394                    }
395                    Some(methods) => {
396                        info!(
397                            "Checking if {:?} in in available methods {methods:?}, {}",
398                            info.mfa_method,
399                            methods.contains(&info.mfa_method)
400                        );
401                        if !methods.contains(&info.mfa_method) {
402                            // FIXME: do not panic
403                            self.set_mfa_method(pool, methods.into_iter().next().unwrap())
404                                .await?;
405                        }
406                    }
407                }
408            }
409        };
410        Ok(())
411    }

426    /// Update gateway state based on this user device access rights
427    pub async fn sync_allowed_devices(
428        &self,
429        conn: &mut PgConnection,
430        wg_tx: &Sender<GatewayEvent>,
431    ) -> Result<(), WebError> {
432        debug!("Syncing allowed devices of user {}", self.username);
433        let networks = WireguardNetwork::all(&mut *conn).await?;
434        for network in networks {
435            let gateway_events = network
436                .sync_allowed_devices_for_user(&mut *conn, self, None)
437                .await?;
438            // check if any peers were updated
439            if !gateway_events.is_empty() {
440                // send peer update events
441                send_multiple_wireguard_events(gateway_events, wg_tx);
442
443                // send firewall config update if ACLs & enterprise features are enabled
444                if let Some(firewall_config) = network.try_get_firewall_config(&mut *conn).await? {
445                    send_wireguard_event(
446                        GatewayEvent::FirewallConfigChanged(network.id, firewall_config),
447                        wg_tx,
448                    );
449                }
450            }
451        }
452        info!("Allowed devices of user {} synced", self.username);
453        Ok(())
454    }

456    /// Deletes the user and cleans up his devices from gateways
457    pub async fn delete_and_cleanup(
458        self,
459        conn: &mut PgConnection,
460        wg_tx: &Sender<GatewayEvent>,
461    ) -> Result<(), WebError> {
462        let username = self.username.clone();
463        debug!("Deleting user {username}, removing his devices from gateways and updating ldap...",);
464        let devices = self.devices(&mut *conn).await?;
465        let mut events = Vec::new();
466
467        // get all locations affected by devices being deleted
468        let mut affected_location_ids = HashSet::new();
469
470        for device in devices {
471            let device_info = DeviceInfo::from_device(&mut *conn, device).await?;
472            for network_info in &device_info.network_info {
473                affected_location_ids.insert(network_info.network_id);
474            }
475            events.push(GatewayEvent::DeviceDeleted(device_info));
476        }
477
478        self.delete(&mut *conn).await?;
479        update_counts(&mut *conn).await?;
480
481        // send firewall config updates to affected locations
482        // if they have ACL enabled & enterprise features are active
483        for location_id in affected_location_ids {
484            if let Some(location) = WireguardNetwork::find_by_id(&mut *conn, location_id).await? {
485                if let Some(firewall_config) = location.try_get_firewall_config(&mut *conn).await? {
486                    debug!("Sending firewall config update for location {location} affected by deleting user {username} devices");
487                    events.push(GatewayEvent::FirewallConfigChanged(
488                        location_id,
489                        firewall_config,
490                    ));
491                }
492            }
493        }
494
495        send_multiple_wireguard_events(events, wg_tx);
496        info!(
497            "The user {} has been deleted and his devices removed from gateways.",
498            &username
499        );
500        Ok(())
501    }

511    /// Get recovery codes. If recovery codes exist, this function returns `None`.
512    /// That way recovery codes are returned only once - when MFA is turned on.
513    pub async fn get_recovery_codes<'e, E>(
514        &mut self,
515        executor: E,
516    ) -> Result<Option<Vec<String>>, SqlxError>
517    where
518        E: PgExecutor<'e>,
519    {
520        if !self.recovery_codes.is_empty() {
521            return Ok(None);
522        }
523
524        for _ in 0..RECOVERY_CODES_COUNT {
525            let code = gen_alphanumeric(16);
526            self.recovery_codes.push(code);
527        }
528        query!(
529            "UPDATE \"user\" SET recovery_codes = $2 WHERE id = $1",
530            self.id,
531            &self.recovery_codes
532        )
533        .execute(executor)
534        .await?;
535
536        Ok(Some(self.recovery_codes.clone()))
537    }

697    pub fn verify_totp_code(&self, code: &str) -> bool {

713    /// Generate MFA code for email verification.
714    ///
715    /// NOTE: This code will be valid for two time frames. See comment for verify_email_mfa_code().
716    pub fn generate_email_mfa_code(&self) -> Result<String, WebError> {
717        if let Some(email_mfa_secret) = &self.email_mfa_secret {
718            let timeout = &server_config().mfa_code_timeout;
719            if let Ok(timestamp) = SystemTime::now().duration_since(SystemTime::UNIX_EPOCH) {
720                let code = totp_custom::<Sha1>(
721                    timeout.as_secs(),
722                    EMAIL_CODE_DIGITS,
723                    email_mfa_secret,
724                    timestamp.as_secs(),
725                );
726                Ok(code)
727            } else {
728                Err(WebError::EmailMfa("SystemTime before UNIX epoch".into()))
729            }
730        } else {
731            Err(WebError::EmailMfa(format!(
732                "Email MFA secret not configured for user {}",
733                self.username
734            )))
735        }
736    }

738    /// Check if email MFA `code` is valid.
739    ///
740    /// IMPORTANT: because current implementation uses TOTP for email verification,
741    /// allow the code for the previous time frame. This approach pretends the code is valid
742    /// for a certain *period of time* (as opposed to a TOTP code which is valid for a certain time *frame*).
743    ///
744    /// ```text
745    /// |<---- frame #0 ---->|<---- frame #1 ---->|<---- frame #2 ---->|
746    /// |................[*]email sent.................................|
747    /// |......................[*]email code verified..................|
748    /// ```
749    #[must_use]
750    pub fn verify_email_mfa_code(&self, code: &str) -> bool {
751        if let Some(email_mfa_secret) = &self.email_mfa_secret {
752            let timeout = server_config().mfa_code_timeout.as_secs();
753            if let Ok(timestamp) = SystemTime::now().duration_since(SystemTime::UNIX_EPOCH) {
754                let expected_code = totp_custom::<Sha1>(
755                    timeout,
756                    EMAIL_CODE_DIGITS,
757                    email_mfa_secret,
758                    timestamp.as_secs(),
759                );
760                if code == expected_code {
761                    return true;
762                }
763                debug!(
764                    "Email MFA verification TOTP code for user {} doesn't fit current time \
765                    frame, checking the previous one. \
766                    Expected: {expected_code}, got: {code}",
767                    self.username
768                );
769
770                let previous_code = totp_custom::<Sha1>(
771                    timeout,
772                    EMAIL_CODE_DIGITS,
773                    email_mfa_secret,
774                    timestamp.as_secs() - timeout,
775                );
776
777                if code == previous_code {
778                    return true;
779                }
780                debug!(
781                    "Email MFA verification TOTP code for user {} doesn't fit previous time frame, \
782                    expected: {previous_code}, got: {code}",
783                    self.username
784                );
785                return false;
786            }
787            debug!(
788                "Couldn't calculate current timestamp when verifying email MFA code for user {}",
789                self.username
790            );
791        } else {
792            debug!("Email MFA secret not configured for user {}", self.username);
793        }
794        false
795    }

797    /// Verify recovery code. If it is valid, consume it, so it can't be used again.
798    pub(crate) async fn verify_recovery_code(
799        &mut self,
800        pool: &PgPool,
801        code: &str,
802    ) -> Result<bool, SqlxError> {
803        if let Some(index) = self.recovery_codes.iter().position(|c| c == code) {
804            // Note: swap_remove() should be faster than remove().
805            self.recovery_codes.swap_remove(index);
806
807            query!(
808                "UPDATE \"user\" SET recovery_codes = $2 WHERE id = $1",
809                self.id,
810                &self.recovery_codes
811            )
812            .execute(pool)
813            .await?;
814
815            Ok(true)
816        } else {
817            Ok(false)
818        }
819    }

1042    /// Create admin user if one doesn't exist yet
1043    pub async fn init_admin_user(
1044        pool: &PgPool,
1045        default_admin_pass: &str,
1046    ) -> Result<(), anyhow::Error> {
1047        debug!("Checking if some admin user already exists and creating one if not...");
1048        let admins = User::find_admins(pool).await?;
1049        if admins.is_empty() {
1050            let admin_groups = Group::find_by_permission(pool, Permission::IsAdmin).await?;
1051            if admin_groups.is_empty() {
1052                return Err(anyhow::anyhow!(
1053                    "No admin group and users found, or they are all disabled. \
1054                    You'll need to create and assign the admin group manually, \
1055                    as there must be at least one active admin user."
1056                ));
1057            }
1058
1059            // create admin user
1060            let password_hash = hash_password(default_admin_pass)?;
1061            let result = query_scalar!(
1062                "INSERT INTO \"user\" (username, password_hash, last_name, first_name, email, ldap_rdn) \
1063                VALUES ('admin', $1, 'Administrator', 'DefGuard', 'admin@defguard', 'admin') \
1064                ON CONFLICT DO NOTHING \
1065                RETURNING id",
1066                password_hash
1067            )
1068            .fetch_optional(pool)
1069            .await?;
1070
1071            // if new user was created add them to admin group, first one you find
1072            // the groups are sorted by ID desceding, so it will often be the 1st one = the default admin group
1073            if let Some(new_user_id) = result {
1074                let admin_group_id = admin_groups
1075                    .first()
1076                    .ok_or(anyhow::anyhow!(
1077                        "No admin group found, can't create admin user"
1078                    ))?
1079                    .id;
1080                info!("New admin user has been created, adding to Admin group...");
1081                query("INSERT INTO group_user (group_id, user_id) VALUES ($1, $2)")
1082                    .bind(admin_group_id)
1083                    .bind(new_user_id)
1084                    .execute(pool)
1085                    .await?;
1086                info!("Admin user has been created as there was no other admin user");
1087            } else {
1088                return Err(anyhow::anyhow!(
1089                    "A conflict occurred while trying to add a missing admin. \
1090                    There is already a user with username 'admin' but he is not an admin or he is disabled. \
1091                    You will need to assign someone the admin group manually or enable this admin user, \
1092                    as there must be at least one active admin."
1093                ));
1094            }
1095        } else {
1096            debug!("Admin users already exists, skipping creation of the default admin user");
1097        }
1098        Ok(())
1099    }
