11/// Initializes global `SETTINGS` struct at program startup
12pub async fn initialize_current_settings(pool: &PgPool) -> Result<(), sqlx::Error> {
13    debug!("Initializing global settings strut");
14    if let Some(settings) = Settings::get(pool).await? {
15        set_settings(Some(settings));
16    } else {
17        debug!(
18            "Settings not found in DB. Using default values to initialize global settings struct"
19        );
20        set_settings(Some(Settings::default()));
21    }
22    Ok(())
23}

25/// Helper function which stores updated `Settings` in the DB and also updates the global `SETTINGS` struct
26pub async fn update_current_settings<'e, E: sqlx::PgExecutor<'e>>(
27    executor: E,
28    new_settings: Settings,
29) -> Result<(), sqlx::Error> {
30    debug!("Updating current settings to: {new_settings:?}");
31    new_settings.save(executor).await?;
32    set_settings(Some(new_settings));
33    Ok(())
34}

162    /// Checks if given settings are correct
163    pub fn validate(&self) -> Result<(), SettingsValidationError> {
164        debug!("Validating settings: {self:?}");
165        // check if gateway disconnect notifications can be enabled, since it requires SMTP to be configured
166        if self.gateway_disconnect_notifications_enabled && !self.smtp_configured() {
167            warn!("Cannot enable gateway disconnect notifications. SMTP is not configured.");
168            return Err(SettingsValidationError::CannotEnableGatewayNotifications);
169        };
170
171        Ok(())
172    }

293    // Set default values for settings if not set yet.
294    // This is only relevant to a subset of settings which are nullable
295    // and we want to initialize their values.
296    pub async fn init_defaults(pool: &PgPool) -> Result<(), sqlx::Error> {
297        info!("Initializing default settings");
298
299        let default_settings = HashMap::from([
300            ("enrollment_welcome_message", defaults::WELCOME_MESSAGE),
301            ("enrollment_welcome_email", defaults::WELCOME_MESSAGE),
302            (
303                "enrollment_welcome_email_subject",
304                defaults::WELCOME_EMAIL_SUBJECT,
305            ),
306        ]);
307
308        for (field, value) in default_settings {
309            let query_string = format!("UPDATE settings SET {field} = $1 WHERE {field} IS NULL");
310            query(&query_string).bind(value).execute(pool).await?;
311        }
312
313        Ok(())
314    }

316    /// Check if all required SMTP options are configured.
317    /// User & password can be empty for no-auth servers.
318    ///
319    /// Meant to be used to check if sending emails is enabled in current instance.
320    #[must_use]
321    pub fn smtp_configured(&self) -> bool {
322        self.smtp_server.is_some()
323            && self.smtp_port.is_some()
324            && self.smtp_sender.is_some()
325            && self.smtp_server != Some(String::new())
326            && self.smtp_sender != Some(String::new())
327    }

329    pub fn ldap_using_username_as_rdn(&self) -> bool {
330        self.ldap_user_rdn_attr
331            .as_deref()
332            .is_none_or(|rdn| rdn.is_empty() || Some(rdn) == self.ldap_username_attr.as_deref())
333    }

422    #[test]
423    fn test_smtp_config() {
424        let mut settings = Settings::default();
425        assert!(!settings.smtp_configured());
426
427        // incomplete SMTP config
428        settings.smtp_server = Some("localhost".into());
429        settings.smtp_port = Some(587);
430        assert!(!settings.smtp_configured());
431
432        // no-auth SMTP config
433        settings.smtp_sender = Some("no-reply@defguard.net".into());
434        assert!(settings.smtp_configured());
435
436        // add non-default encryption
437        settings.smtp_encryption = SmtpEncryption::StartTls;
438        assert!(settings.smtp_configured());
439
440        // add auth info
441        settings.smtp_user = Some("smtp_user".into());
442        settings.smtp_password = Some(SecretStringWrapper::from_str("hunter2").unwrap());
443        assert!(settings.smtp_configured());
444    }
