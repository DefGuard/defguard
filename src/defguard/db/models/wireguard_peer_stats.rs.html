25impl WireguardPeerStats {
26    /// Delete stats older than a configured threshold.
27    /// This is done to prevent unnecessary table growth.
28    /// At least one record is retained for each device and network combination,
29    /// even when older than set threshold.
30    pub(crate) async fn purge_old_stats(
31        pool: &PgPool,
32        stats_purge_threshold: Duration,
33    ) -> Result<(), sqlx::Error> {
34        let start = Utc::now();
35        info!(
36            "Purging stats older than {}",
37            format_duration(stats_purge_threshold)
38        );
39
40        let threshold = (Utc::now()
41            - TimeDelta::from_std(stats_purge_threshold).expect("Failed to parse duration"))
42        .naive_utc();
43        let result = query!(
44            "DELETE FROM wireguard_peer_stats \
45            WHERE collected_at < $1 \
46            AND (device_id, network, collected_at) NOT IN ( \
47                SELECT device_id, network, MAX(collected_at) \
48                FROM wireguard_peer_stats \
49                GROUP BY device_id, network)",
50            threshold
51        )
52        .execute(pool)
53        .await?;
54
55        let end = Utc::now();
56        let rows_count = result.rows_affected();
57
58        info!("Removed {rows_count} old records from wireguard_peer_stats",);
59
60        // Store successful stats purge in database.
61        Self::record_stats_purge(pool, start, end, threshold, rows_count as i64).await?;
62
63        Ok(())
64    }
65
66    // Check how much time has elapsed since last recorded stats purge
67    pub async fn time_since_last_purge<'e, E>(executor: E) -> Result<Option<Duration>, sqlx::Error>
68    where
69        E: PgExecutor<'e>,
70    {
71        debug!("Checking time since last stats purge");
72
73        let timestamp = query_scalar!("SELECT MAX(started_at) FROM wireguard_stats_purge")
74            .fetch_one(executor)
75            .await?;
76
77        match timestamp {
78            Some(timestamp) => {
79                let time_since = Utc::now().signed_duration_since(timestamp.and_utc());
80                let time_since = time_since.to_std().expect("Failed to parse duration");
81                debug!(
82                    "Time since last stats purge: {}",
83                    format_duration(time_since)
84                );
85                Ok(Some(time_since))
86            }
87            None => Ok(None),
88        }
89    }
90
91    async fn record_stats_purge<'e, E>(
92        executor: E,
93        start: DateTime<Utc>,
94        end: DateTime<Utc>,
95        removal_threshold: NaiveDateTime,
96        records_removed: i64,
97    ) -> Result<(), sqlx::Error>
98    where
99        E: PgExecutor<'e>,
100    {
101        debug!("Recording successful stats purge in database");
102        query!("INSERT INTO wireguard_stats_purge (started_at, finished_at, removal_threshold, records_removed) VALUES ($1, $2, $3, $4)",
103        start.naive_utc(), end.naive_utc(), removal_threshold, records_removed).execute(executor).await?;
104
105        Ok(())
106    }
107}

109impl WireguardPeerStats<Id> {
110    pub(crate) async fn fetch_latest(
111        conn: &PgPool,
112        device_id: Id,
113        network_id: Id,
114    ) -> Result<Option<Self>, sqlx::Error> {
115        let stats = query_as!(
116            Self,
117            "SELECT id, device_id \"device_id!\", collected_at \"collected_at!\", \
118            network \"network!\", endpoint, upload \"upload!\", download \"download!\", \
119            latest_handshake \"latest_handshake!\", allowed_ips \
120            FROM wireguard_peer_stats \
121            WHERE device_id = $1 AND network = $2 \
122            ORDER BY collected_at DESC LIMIT 1",
123            device_id,
124            network_id,
125        )
126        .fetch_optional(conn)
127        .await?;
128
129        Ok(stats)
130    }
131
132    /// Remove port part from `endpoint`.
133    /// IPv4: a.b.c.d:p -> a.b.c.d
134    /// IPv6: [x::y:z]:p -> x::y:z
135    pub(crate) fn endpoint_without_port(&self) -> Option<String> {
136        self.endpoint.as_ref().and_then(|endpoint| {
137            let mut addr = endpoint.rsplit_once(':')?.0;
138            // Strip square brackets.
139            if addr.starts_with('[') && addr.ends_with(']') {
140                let end = addr.len() - 1;
141                addr = &addr[1..end];
142            }
143            Some(addr.to_owned())
144        })
145    }
146
147    /// Trim `allowed_ips` returning the first one without CIDR.
148    pub(crate) fn trim_allowed_ips(&self) -> Option<String> {
149        self.allowed_ips
150            .as_ref()
151            .and_then(|ips| Some(ips.split_once('/')?.0.to_owned()))
152    }
