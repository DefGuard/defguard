1use chrono::{NaiveDateTime, TimeDelta, Utc};
2use sqlx::{query, query_as, Error as SqlxError, PgExecutor, PgPool, Type};
3use webauthn_rs::prelude::{PasskeyAuthentication, PasskeyRegistration};
4
5use crate::{db::Id, random::gen_alphanumeric, server_config};

7#[derive(Clone, PartialEq, Type)]
8#[repr(i16)]
9pub enum SessionState {
10    NotVerified,
11    PasswordVerified,
12    MultiFactorVerified,
13    ApiTokenVerified,
14}
15
16// Representation of a Defguard server user session
17// derived from session cookies
18#[derive(Clone)]
19pub struct Session {
20    pub id: String,
21    pub user_id: Id,
22    pub state: SessionState,
23    pub created: NaiveDateTime,
24    pub expires: NaiveDateTime,
25    pub webauthn_challenge: Option<Vec<u8>>,
26    pub ip_address: String,
27    pub device_info: Option<String>,
28}
29
30impl Session {
31    #[must_use]
32    pub fn new(
33        user_id: Id,
34        state: SessionState,
35        ip_address: String,
36        device_info: Option<String>,
37    ) -> Self {
38        let now = Utc::now();
39        let timeout = server_config().session_timeout;
40        Self {
41            id: gen_alphanumeric(24),
42            user_id,
43            state,
44            created: now.naive_utc(),
45            expires: (now + TimeDelta::seconds(timeout.as_secs() as i64)).naive_utc(),
46            webauthn_challenge: None,
47            ip_address,
48            device_info,
49        }
50    }
51
52    #[must_use]
53    pub fn expired(&self) -> bool {
54        self.expires < Utc::now().naive_utc()
55    }
56
57    pub async fn find_by_id(pool: &PgPool, id: &str) -> Result<Option<Self>, SqlxError> {
58        query_as!(
59            Self,
60            "SELECT id, user_id, state \"state: SessionState\", created, expires, webauthn_challenge, \
61            ip_address, device_info FROM session WHERE id = $1",
62            id
63        )
64        .fetch_optional(pool)
65        .await
66    }
67
68    pub async fn save(&self, pool: &PgPool) -> Result<(), SqlxError> {
69        query!(
70            "INSERT INTO session (id, user_id, state, created, expires, webauthn_challenge, ip_address, device_info) \
71            VALUES ($1, $2, $3, $4, $5, $6, $7, $8)",
72            self.id,
73            self.user_id,
74            self.state.clone() as i16,
75            self.created,
76            self.expires,
77            self.webauthn_challenge,
78            self.ip_address,
79            self.device_info,
80        )
81        .execute(pool)
82        .await?;
83
84        Ok(())
85    }
86
87    pub async fn set_state(&mut self, pool: &PgPool, state: SessionState) -> Result<(), SqlxError> {
88        query!(
89            "UPDATE session SET state = $1 WHERE id = $2",
90            state.clone() as i16,
91            self.id
92        )
93        .execute(pool)
94        .await?;
95        self.state = state;
96
97        Ok(())
98    }
99
100    #[must_use]
101    pub fn get_passkey_registration(&self) -> Option<PasskeyRegistration> {
102        self.webauthn_challenge
103            .as_ref()
104            .and_then(|challenge| serde_cbor::from_slice(challenge).ok())
105    }
106
107    #[must_use]
108    pub fn get_passkey_authentication(&self) -> Option<PasskeyAuthentication> {
109        self.webauthn_challenge
110            .as_ref()
111            .and_then(|challenge| serde_cbor::from_slice(challenge).ok())
112    }
113
114    pub async fn set_passkey_authentication<'e, E>(
115        &mut self,
116        executor: E,
117        passkey_auth: &PasskeyAuthentication,
118    ) -> Result<(), SqlxError>
119    where
120        E: PgExecutor<'e>,
121    {
122        if let Ok(webauthn_challenge) = serde_cbor::to_vec(passkey_auth) {
123            query!(
124                "UPDATE session SET webauthn_challenge = $1 WHERE id = $2",
125                webauthn_challenge,
126                self.id
127            )
128            .execute(executor)
129            .await?;
130            self.webauthn_challenge = Some(webauthn_challenge);
131        }
132
133        Ok(())
134    }
