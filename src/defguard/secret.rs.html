11/// Wrapper for secrecy `SecretString` struct which implements sqlx traits.
12#[derive(Clone, Debug, Deserialize)]
13pub struct SecretStringWrapper(SecretString);
14
15impl SecretStringWrapper {
16    #[must_use]
17    pub fn expose_secret(&self) -> &str {
18        self.0.expose_secret()
19    }
20}
21
22impl FromStr for SecretStringWrapper {
23    type Err = Infallible;
24
25    fn from_str(src: &str) -> Result<Self, Self::Err> {
26        Ok(Self(SecretString::from(src)))
27    }
28}
29
30impl Serialize for SecretStringWrapper {
31    fn serialize<S>(&self, ser: S) -> Result<S::Ok, S::Error>
32    where
33        S: serde::Serializer,
34    {
35        ser.serialize_str(self.0.expose_secret())
36    }
37}
38
39impl Decode<'_, Postgres> for SecretStringWrapper {
40    fn decode(value: PgValueRef<'_>) -> Result<Self, Box<dyn Error + Send + Sync>> {
41        <String as Decode<'_, Postgres>>::decode(value).map(|v| Self(SecretString::from(v)))
42    }
43}
44
45impl Encode<'_, Postgres> for SecretStringWrapper {
46    fn encode_by_ref(
47        &self,
48        buf: &mut PgArgumentBuffer,
49    ) -> Result<IsNull, Box<dyn Error + Sync + Send>> {
50        <&str as Encode<Postgres>>::encode_by_ref(&self.0.expose_secret(), buf)
51    }
52}
53
54impl Type<Postgres> for SecretStringWrapper {
55    fn type_info() -> PgTypeInfo {

57    }
58

61    }
62}
63
64impl PartialEq for SecretStringWrapper {
