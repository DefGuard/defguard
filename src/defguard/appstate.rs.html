37impl AppState {
38    pub(crate) fn trigger_action(&self, event: AppEvent) {
39        let event_name = event.name().to_owned();
40        match self.tx.send(event) {
41            Ok(()) => info!("Sent trigger {event_name}"),
42            Err(err) => error!("Error sending trigger {event_name}: {err}"),
43        }
44    }
45
46    /// Handle webhook events
47    async fn handle_triggers(pool: PgPool, mut rx: UnboundedReceiver<AppEvent>) {
48        let reqwest_client = Client::builder().user_agent("reqwest").build().unwrap();
49        while let Some(msg) = rx.recv().await {
50            debug!("WebHook triggered");
51            debug!("Retrieving webhooks");
52            if let Ok(webhooks) = WebHook::all_enabled(&pool, &msg).await {
53                debug!("Found webhooks: {webhooks:?}");
54                let (payload, event) = match msg {
55                    AppEvent::UserCreated(user) => (json!(user), "user_created"),
56                    AppEvent::UserModified(user) => (json!(user), "user_modified"),
57                    AppEvent::UserDeleted(username) => {
58                        (json!({ "username": username }), "user_deleted")
59                    }
60                    AppEvent::HWKeyProvision(data) => (json!(data), "user_keys"),
61                };
62                for webhook in webhooks {
63                    match reqwest_client
64                        .post(&webhook.url)
65                        .bearer_auth(&webhook.token)
66                        .header("x-defguard-event", event)
67                        .json(&payload)
68                        .send()
69                        .await
70                    {
71                        Ok(res) => {
72                            info!("Trigger sent to {}, status {}", webhook.url, res.status());
73                        }
74                        Err(err) => {
75                            error!("Error sending trigger to {}: {err}", webhook.url);
76                        }
77                    }
78                }
79            }
80        }
81    }

83    /// Sends given `GatewayEvent` to be handled by gateway GRPC server.

89    /// Sends multiple events to be handled by gateway GRPC server.

94
95    /// Create application state
96    pub fn new(
97        pool: PgPool,
98        tx: UnboundedSender<AppEvent>,
99        rx: UnboundedReceiver<AppEvent>,
100        wireguard_tx: Sender<GatewayEvent>,
101        mail_tx: UnboundedSender<Mail>,
102        failed_logins: Arc<Mutex<FailedLoginMap>>,
103    ) -> Self {
104        spawn(Self::handle_triggers(pool.clone(), rx));
105
106        let config = server_config();
107        let webauthn_builder = WebauthnBuilder::new(
108            config
109                .webauthn_rp_id
110                .as_ref()
111                .expect("Webauth RP ID configuration is required"),
112            &config.url,
113        )
114        .expect("Invalid WebAuthn configuration");
115        let webauthn = Arc::new(
116            webauthn_builder
117                .build()
118                .expect("Invalid WebAuthn configuration"),
119        );
120
121        let key = Key::from(config.secret_key.expose_secret().as_bytes());
122
123        Self {
124            pool,
125            tx,
126            wireguard_tx,
127            mail_tx,
128            webauthn,
129            failed_logins,
130            key,
131        }
132    }
133}
