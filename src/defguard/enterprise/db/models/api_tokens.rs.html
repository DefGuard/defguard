1use chrono::NaiveDateTime;
2use model_derive::Model;
3use sqlx::{query_as, Error as SqlxError, PgExecutor};
4
5use crate::db::{Id, NoId};
6
7#[derive(Deserialize, Model, Serialize)]
8#[table(api_token)]
9pub struct ApiToken<I = NoId> {
10    id: I,
11    pub user_id: Id,
12    pub created_at: NaiveDateTime,
13    pub name: String,
14    pub token_hash: String,
15}
16
17impl ApiToken {
18    pub fn new(user_id: Id, created_at: NaiveDateTime, name: String, token_string: &str) -> Self {
19        let token_hash = Self::hash_token(token_string);
20        Self {
21            id: NoId,
22            user_id,
23            created_at,
24            name,
25            token_hash,
26        }
27    }
28
29    /// Generates an SHA256 hash which can be stored in a database based on a token string.
30    fn hash_token(token_string: &str) -> String {
31        sha256::digest(token_string)
32    }
33}
34
35impl ApiToken<Id> {
36    pub async fn find_by_user_id<'e, E>(executor: E, user_id: Id) -> Result<Vec<Self>, SqlxError>
37    where
38        E: PgExecutor<'e>,
39    {
40        query_as!(
41            Self,
42            "SELECT id, user_id, created_at, name, token_hash \
43                    FROM api_token WHERE user_id = $1 ORDER BY id",
44            user_id
45        )
46        .fetch_all(executor)
47        .await
48    }
49
50    pub async fn try_find_by_auth_token<'e, E>(
51        executor: E,
52        auth_token: &str,
53    ) -> Result<Option<Self>, SqlxError>
54    where
55        E: PgExecutor<'e>,
56    {
57        let token_hash = ApiToken::hash_token(auth_token);
58        let maybe_token = query_as!(
59            Self,
60            "SELECT id, user_id, created_at, name, token_hash \
61                    FROM api_token WHERE token_hash = $1",
62            token_hash
63        )
64        .fetch_optional(executor)
65        .await?;
66        Ok(maybe_token)
67    }
68}
