11/// Calculate salted SHA1 hash from given password in SSHA password storage scheme.
12#[must_use]
13pub fn salted_sha1_hash(password: &str) -> String {
14    // random bytes
15    let mut salt = [0u8; 4];
16    OsRng.fill_bytes(&mut salt);
17
18    let mut pass = Vec::from(password);
19    pass.extend_from_slice(&salt);
20
21    let checksum = Sha1::digest(pass);
22    let checksum = checksum.concat(GenericArray::from(salt));
23
24    format!(
25        "{{SSHA}}{}",
26        base64::prelude::BASE64_STANDARD.encode(checksum)
27    )
28}

30/// Calculate Windows NT-HASH; used for `sambaNTPassword`.
31#[must_use]
32pub fn nthash(password: &str) -> String {
33    let password_utf16_le: Vec<u8> = password
34        .encode_utf16()
35        .flat_map(|c| IntoIterator::into_iter(c.to_le_bytes()))
36        .collect();
37    to_lower_hex(&Md4::digest(password_utf16_le))
38}

40/// Calculated AD password used for `unicodePwd`.
41#[must_use]
42pub fn unicode_pwd(password: &str) -> Vec<u8> {
43    let quoted = format!("\"{password}\"");
44    let utf16_bytes: Vec<u8> = quoted
45        .encode_utf16()
46        .flat_map(|c| c.to_le_bytes())
47        .collect();
48
49    utf16_bytes
50}
