18    pub(crate) async fn create() -> Result<super::LDAPConnection, LdapError> {
19        let settings = Settings::get_current_settings();
20        let config = super::LDAPConfig::try_from(settings.clone())?;
21        let url = settings.ldap_url.ok_or(LdapError::MissingSettings(
22            "LDAP URL is required for LDAP configuration to work".to_string(),
23        ))?;
24        let password = settings
25            .ldap_bind_password
26            .ok_or(LdapError::MissingSettings("LDAP bind password".to_string()))?;
27        let conn_settings = LdapConnSettings::new()
28            .set_starttls(settings.ldap_use_starttls)
29            .set_no_tls_verify(!settings.ldap_tls_verify_cert)
30            .set_conn_timeout(Duration::from_secs(8));
31        let (conn, mut ldap) = LdapConnAsync::with_settings(conn_settings, &url).await?;
32        drive!(conn);
33        info!("Connected to LDAP: {url}");
34        ldap.simple_bind(&config.ldap_bind_username, password.expose_secret())
35            .await?
36            .success()?;
37
38        Ok(Self { config, ldap, url })
39    }

137    /// Updates LDAP object with specified distinguished name and attributes.
138    pub(super) async fn modify<S>(
139        &mut self,
140        old_dn: &str,
141        new_dn: &str,
142        mods: Vec<Mod<S>>,
143    ) -> Result<(), LdapError>
144    where
145        S: AsRef<[u8]> + Eq + Hash,
146    {
147        self.ldap.modify(old_dn, mods).await?;
148        if old_dn != new_dn {
149            if let Some((new_rdn, _rest)) = new_dn.split_once(',') {
150                self.ldap.modifydn(old_dn, new_rdn, true, None).await?;
151            } else {
152                warn!("Failed to rename LDAP object {old_dn} to {new_dn}, new DN is invalid");
153            }
154        }
155        info!("Modified LDAP object {old_dn}");
156
157        Ok(())
158    }

170    /// Returns a map of group names to a set of members
171    pub(super) async fn get_ldap_group_memberships<'a>(
172        &mut self,
173        all_ldap_users: &'a [User],
174    ) -> Result<HashMap<String, HashSet<&'a User>>, LdapError> {
175        debug!("Retrieving LDAP group memberships");
176        let mut membership_entries = self.list_group_memberships().await?;
177        let mut memberships: HashMap<String, HashSet<&User>> = HashMap::new();
178        // rdn: user map
179        let rdn_map = all_ldap_users
180            .iter()
181            .map(|u| (u.ldap_rdn_value(), u))
182            .collect::<HashMap<_, _>>();
183
184        for entry in membership_entries.iter_mut() {
185            let groupname = entry
186                .attrs
187                .remove(&self.config.ldap_groupname_attr)
188                .and_then(|mut v| v.pop());
189
190            if let Some(groupname) = groupname {
191                if let Some(members) = entry.attrs.get(&self.config.ldap_group_member_attr) {
192                    let members = members
193                        .iter()
194                        .filter_map(|v| {
195                            extract_rdn_value(v).and_then(|v| {
196                                if let Some(user) = rdn_map.get(v.as_str()) {
197                                    Some(*user)
198                                } else {
199                                    debug!(
200                                        "LDAP group {groupname} contains member {v} that does not belong to the filtered LDAP users list, skipping"
201                                    );
202                                    None
203                                }
204
205                            })
206                        })
207                        .collect::<HashSet<_>>();
208                    memberships.insert(groupname, members);
209                } else {
210                    warn!("LDAP group {groupname} missing group member attribute, skipping");
211                }
212            } else {
213                warn!("Group entry {entry:?} missing groupname attribute, skipping");
214            }
215        }
216
217        Ok(memberships)
218    }

220    pub(super) async fn is_member_of(
221        &mut self,
222        user_dn: &str,
223        groupname: &str,
224    ) -> Result<bool, LdapError> {
225        debug!("Checking if user {user_dn} is member of group {groupname}");
226        let filter = format!(
227            "(&(objectClass={})({}={})({}={}))",
228            self.config.ldap_group_obj_class,
229            self.config.ldap_groupname_attr,
230            groupname,
231            self.config.ldap_group_member_attr,
232            user_dn
233        );
234        debug!(
235            "Using the following filter for group search: {filter} and base: {}",
236            self.config.ldap_group_search_base
237        );
238        let (rs, res) = self
239            .ldap
240            .search(
241                &self.config.ldap_group_search_base,
242                Scope::Subtree,
243                filter.as_str(),
244                vec!["*"],
245            )
246            .await?
247            .success()?;
248        debug!("LDAP group membership search result: {res:?}");
249        Ok(!rs.is_empty())
250    }

252    pub(super) async fn get_group_members(
253        &mut self,
254        groupname: &str,
255    ) -> Result<Vec<String>, LdapError> {
256        debug!("Searching for group memberships for group {}", groupname);
257        let filter = format!(
258            "(&(objectClass={})({}={}))",
259            self.config.ldap_group_obj_class, self.config.ldap_groupname_attr, groupname
260        );
261        debug!(
262            "Using the following filter for group search: {filter} and base: {}",
263            self.config.ldap_group_search_base
264        );
265        let mut search_stream = self
266            .ldap
267            .streaming_search_with(
268                PagedResults::new(500),
269                &self.config.ldap_group_search_base,
270                Scope::Subtree,
271                filter.as_str(),
272                vec![&self.config.ldap_group_member_attr],
273            )
274            .await?;
275
276        let mut member_entries = Vec::new();
277        while let Some(entry) = search_stream.next().await? {
278            member_entries.push(SearchEntry::construct(entry));
279        }
280
281        let members = member_entries
282            .first()
283            .and_then(|entry| {
284                let member_entries = entry.attrs.get(&self.config.ldap_group_member_attr);
285                member_entries.map(|v| {
286                    v.iter()
287                        .filter_map(|v| extract_rdn_value(v))
288                        .collect::<Vec<_>>()
289                })
290            })
291            .unwrap_or_default();
292        debug!(
293            "Performed LDAP group memberships search for group {}",
294            groupname
295        );
296
297        Ok(members)
298    }

300    pub(super) async fn list_users(&mut self) -> Result<Vec<SearchEntry>, LdapError> {
301        let filter = if !self.config.ldap_sync_groups.is_empty() {
302            debug!(
303                "LDAP sync groups are defined, filtering users by those groups: {:?}",
304                self.config.ldap_sync_groups
305            );
306            let mut group_filters = vec![];
307            for group in self.config.ldap_sync_groups.iter() {
308                let group_dn = self.config.group_dn(group);
309                group_filters.push(format!("({}={})", self.config.ldap_member_attr, group_dn));
310            }
311            debug!(
312                "Using the following group filters for user search: {:?}",
313                group_filters
314            );
315            format!(
316                "(&(objectClass={})(|{}))",
317                self.config.ldap_user_obj_class,
318                group_filters.join("")
319            )
320        } else {
321            debug!("No LDAP sync groups defined, searching for all users in the base DN");
322            format!("(objectClass={})", self.config.ldap_user_obj_class)
323        };
324
325        debug!(
326            "Using the following filter for user search: {filter} and base: {}",
327            self.config.ldap_user_search_base
328        );
329
330        let mut search_stream = self
331            .ldap
332            .streaming_search_with(
333                PagedResults::new(500),
334                &self.config.ldap_user_search_base,
335                Scope::Subtree,
336                &filter,
337                vec!["*", &self.config.ldap_member_attr],
338            )
339            .await?;
340
341        let mut entries = vec![];
342        while let Some(entry) = search_stream.next().await? {
343            entries.push(SearchEntry::construct(entry));
344        }
345
346        debug!("Performed LDAP user search");
347
348        Ok(entries)
349    }
