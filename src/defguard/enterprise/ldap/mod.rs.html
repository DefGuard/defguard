27pub(crate) async fn do_ldap_sync(pool: &PgPool) -> Result<(), LdapError> {
28    debug!("Starting LDAP sync, if enabled");
29    let mut settings = Settings::get_current_settings();
30
31    // Mark as out of sync only if we can't propagate changes to LDAP, as it
32    // doesn't matter for the sync status if we can't pull changes.
33    if !settings.ldap_enabled {
34        debug!("LDAP is disabled, not performing LDAP sync");
35        if get_ldap_sync_status() == SyncStatus::InSync {
36            set_ldap_sync_status(SyncStatus::OutOfSync, pool).await?;
37        }
38        return Ok(());
39    }
40
41    if !settings.ldap_sync_enabled {
42        debug!("LDAP sync is disabled, not performing LDAP sync");
43        return Ok(());
44    }
45
46    if !is_enterprise_enabled() {
47        info!("Enterprise features are disabled, not performing LDAP sync and automatically disabling it");
48        settings.ldap_sync_enabled = false;
49        update_current_settings(pool, settings).await?;
50        return Err(LdapError::EnterpriseDisabled("LDAP sync".to_string()));
51    }
52
53    if is_ldap_desynced() {
54        info!("LDAP is considered to be desynced, doing a full sync");
55    } else {
56        info!("Ldap is not considered to be desynced, doing an incremental sync");
57        debug!("Because of incremental sync, LDAP authority will be selected to pull changes from LDAP");
58    }
59
60    let mut ldap_connection = match LDAPConnection::create().await {
61        Ok(connection) => connection,
62        Err(err) => {
63            set_ldap_sync_status(SyncStatus::OutOfSync, pool).await?;
64            return Err(err);
65        }
66    };
67
68    if let Err(err) = ldap_connection.sync(pool, is_ldap_desynced()).await {
69        set_ldap_sync_status(SyncStatus::OutOfSync, pool).await?;
70        return Err(err);
71    } else {
72        set_ldap_sync_status(SyncStatus::InSync, pool).await?;
73    };
74
75    let _ = update_counts(pool).await;
76
77    info!("LDAP sync completed");
78
79    Ok(())
80}

82/// Convenience function to run a function that performs an LDAP operation and handle the result
83/// appropriately, setting the LDAP sync status to Desynced if an error is encountered.
84pub(crate) async fn with_ldap_status<T, F>(pool: &PgPool, f: F) -> Result<T, LdapError>
85where
86    F: Future<Output = Result<T, LdapError>>,
87{
88    let settings = Settings::get_current_settings();
89    if !is_enterprise_enabled() {
90        info!("Enterprise features are disabled, not performing LDAP operation");
91        set_ldap_sync_status(SyncStatus::OutOfSync, pool).await?;
92        return Err(LdapError::EnterpriseDisabled("LDAP".to_string()));
93    }
94
95    if !settings.ldap_enabled {
96        debug!("LDAP is disabled, not performing LDAP operation");
97        set_ldap_sync_status(SyncStatus::OutOfSync, pool).await?;
98        return Err(LdapError::MissingSettings("LDAP is disabled".into()));
99    }
100
101    if settings.ldap_sync_enabled && get_ldap_sync_status() == SyncStatus::OutOfSync {
102        warn!("LDAP is considered to be desynced, not performing LDAP operation");
103        return Err(LdapError::Desynced);
104    }
105
106    match f.await {
107        Ok(result) => Ok(result),
108        Err(e) => {
109            warn!("Encountered an error while performing LDAP operation: {e:?}");
110            if let Err(status_err) = set_ldap_sync_status(SyncStatus::OutOfSync, pool).await {
111                warn!("Failed to update LDAP sync status: {:?}", status_err);
112            }
113
114            Err(e)
115        }
116    }
117}

300    /// Updates user state in LDAP based on the following rules:
301    /// - If the user is disabled in Defguard, he will be removed from LDAP
302    /// - If there are no sync groups defined or the user is in them but doesn't exist yet in LDAP, he will be added to LDAP and assigned to his groups
303    /// - If the user is not in sync groups but is present in LDAP, he will be removed from LDAP
304    ///
305    /// Make sure to call this every time one of the above conditions changes (e.g. group addition, user disabling)
306    pub(crate) async fn update_users_state(
307        &mut self,
308        users: Vec<&mut User<Id>>,
309        pool: &PgPool,
310    ) -> Result<(), LdapError> {
311        debug!("Updating users state in LDAP");
312        let transaction = pool.begin().await?;
313
314        for user in users {
315            let user_exists_in_ldap = self.user_exists(user).await?;
316            let user_groups = user.member_of_names(pool).await?;
317            let user_sync_allowed = user.ldap_sync_allowed(pool).await?;
318
319            // User is disabled in Defguard or he is not in the defined sync groups
320            // If they exist in LDAP, remove them
321            if !user.is_active && user_exists_in_ldap {
322                debug!("User {user} is disabled in Defguard, removing from LDAP");
323                self.delete_user(user).await?;
324                continue;
325            }
326
327            // No sync groups defined or user already belongs to them,
328            // Add the user if they don't exist in LDAP already but are active in Defguard
329            if user_sync_allowed && !user_exists_in_ldap {
330                debug!("User {user} is not in LDAP, adding to LDAP");
331                self.add_user(user, None, pool).await?;
332                for group in user_groups {
333                    self.add_user_to_group(user, &group).await?;
334                }
335                continue;
336            }
337
338            // We may bring user into the synchronization scope, sync his data (email, groups, etc.) based on
339            // the authority
340            if user_sync_allowed && user_exists_in_ldap {
341                debug!(
342                    "User {user} is in LDAP and is allowed to be synced, synchronizing his data"
343                );
344                self.sync_user_data(user, pool).await?;
345                debug!("User {user} data synchronized");
346                continue;
347            }
348        }
349
350        transaction.commit().await?;
351
352        Ok(())
353    }

355    /// Checks if user belongs to one of the defined sync groups in the LDAP server.
356    async fn user_in_ldap_sync_groups<I>(&mut self, user: &User<I>) -> Result<bool, LdapError> {
357        debug!("Checking if user {} is in LDAP sync groups", user.username);
358
359        // Sync groups empty, we should sync all users
360        if self.config.ldap_sync_groups.is_empty() {
361            debug!("Sync groups were not defined, user {user} will be synced");
362            return Ok(true);
363        }
364
365        let dn = self.config.user_dn(user.ldap_rdn_value());
366
367        if !self.user_exists(user).await? {
368            debug!("User {user} does not exist, not syncing user");
369            return Ok(false);
370        }
371
372        let user_groups_entries = self.get_user_groups(&dn).await?;
373        let user_groups_names = user_groups_entries
374            .iter()
375            .filter_map(|entry| {
376                entry
377                    .attrs
378                    .get(&self.config.ldap_groupname_attr)
379                    .and_then(|v| v.first())
380            })
381            .collect::<Vec<_>>();
382
383        debug!(
384            "User groups: {user_groups_names:?}, sync groups: {:?}",
385            self.config.ldap_sync_groups
386        );
387
388        if user_groups_names
389            .into_iter()
390            .any(|group| self.config.ldap_sync_groups.contains(group))
391        {
392            debug!("User {user} is in sync groups, syncing user");
393            Ok(true)
394        } else {
395            debug!("User {user} is not in sync groups, not syncing user");
396            Ok(false)
397        }
398    }

516    /// Adds user to LDAP.
517    pub async fn add_user(
518        &mut self,
519        user: &mut User<Id>,
520        password: Option<&str>,
521        pool: &PgPool,
522    ) -> Result<(), LdapError> {
523        debug!("Adding LDAP user {}", user.username);
524        let user_rdn = user.ldap_rdn_value();
525        let dn = self.config.user_dn(user_rdn);
526        let password_is_random = password.is_none();
527        let password = if let Some(password) = password {
528            debug!("Using provided password for user {}", user.username);
529            password.to_string()
530        } else {
531            // ldap may not accept no password, this is a workaround when we don't have access to the
532            // user's password
533            debug!(
534                "Generating random password for user {}, as no password has been specified",
535                user.username
536            );
537            let random_password = rand::thread_rng()
538                .sample_iter(&rand::distributions::Alphanumeric)
539                .take(32)
540                .map(char::from)
541                .collect::<String>();
542
543            debug!("Generated random password for user {}", user.username);
544            random_password
545        };
546        let ssha_password = hash::salted_sha1_hash(&password);
547        let nt_password = hash::nthash(&password);
548        let user_obj_classes = self.config.get_all_user_obj_classes();
549        let username_attr = self.config.ldap_username_attr.clone();
550        let rdn_attr = self.config.get_rdn_attr().to_string();
551        if !self.is_username_available(&user.username).await
552            || !self.is_rdn_available(user_rdn).await
553        {
554            return Err(LdapError::ObjectAlreadyExists(format!(
555                "User with username {} or RDN {user_rdn} already exists",
556                user.username
557            )));
558        }
559        self.add(
560            &dn,
561            user.as_ldap_attrs(
562                &ssha_password,
563                &nt_password,
564                user_obj_classes.iter().map(|s| s.as_str()).collect(),
565                self.config.ldap_uses_ad,
566                &username_attr,
567                &rdn_attr,
568            ),
569        )
570        .await?;
571        if self.config.ldap_uses_ad {
572            self.set_password(user, &password).await?;
573            self.activate_ad_user(user_rdn).await?;
574        }
575        if password_is_random {
576            user.ldap_pass_randomized = true;
577            user.save(pool).await?;
578        }
579        info!("Added LDAP user {}", user.username);
580        Ok(())
581    }

583    /// Modifies existing LDAP user.
584    pub async fn modify_user(
585        &mut self,
586        old_username: &str,
587        user: &User<Id>,
588    ) -> Result<(), LdapError> {
589        debug!("Modifying user {old_username} in LDAP");
590        // If we're using the username as the RDN, also update the RDN value on user if his username has been changed
591        let old_rdn = if self.config.using_username_as_rdn() {
592            old_username
593        } else {
594            user.ldap_rdn_value()
595        };
596        let new_rdn = if self.config.using_username_as_rdn() {
597            user.username.as_str()
598        } else {
599            user.ldap_rdn_value()
600        };
601        if !self.user_exists_by_rdn(old_rdn).await? {
602            return Err(LdapError::ObjectNotFound(format!(
603                "User {old_username} not found in LDAP, cannot modify",
604            )));
605        }
606        let old_dn = self.config.user_dn(old_rdn);
607        let new_dn = self.config.user_dn(new_rdn);
608        let config = self.config.clone();
609        let mods = user.as_ldap_mod(&config);
610        self.modify(&old_dn, &new_dn, mods).await?;
611        info!("Modified user {old_username} in LDAP");
612
613        Ok(())
614    }

631    pub async fn delete_user<I>(&mut self, user: &User<I>) -> Result<(), LdapError> {
632        let user_rdn_value = user.ldap_rdn_value();
633        debug!("Deleting user {user}");
634        let dn = self.config.user_dn(user_rdn_value);
635        debug!("Removing group memberships first...");
636        let user_groups = self.get_user_groups(&dn).await?;
637        debug!("Removing user from groups: {user_groups:?}");
638        for group in user_groups {
639            debug!("Removing user from group {group:?}");
640            match self.group_entry_to_name(group) {
641                Ok(groupname) => {
642                    self.remove_user_from_group(user, &groupname).await?;
643                    debug!("Removed user from group {groupname}");
644                }
645                Err(e) => {
646                    warn!("Failed to remove user from group: {e}");
647                }
648            }
649        }
650        self.delete(&dn).await?;
651        info!("Deleted user {user}");
652
653        Ok(())
654    }

675    /// Changes user password.
676    pub async fn set_password<I>(
677        &mut self,
678        user: &User<I>,
679        password: &str,
680    ) -> Result<(), LdapError> {
681        debug!("Setting password for user {user}");
682        let user_dn = self.config.user_dn(user.ldap_rdn_value());
683
684        if self.config.ldap_uses_ad {
685            let unicode_pwd = hash::unicode_pwd(password);
686            // We need to deal with bytes here as both the attribute and value are expected to be in
687            // binary
688            let mods = vec![Mod::Replace(
689                "unicodePwd".as_bytes(),
690                hashset![unicode_pwd.as_ref()],
691            )];
692            self.modify(&user_dn, &user_dn, mods).await?;
693            info!("Password set for user {user}");
694        } else {
695            let ssha_password = hash::salted_sha1_hash(password);
696            let nt_password = hash::nthash(password);
697            let mut mods = Vec::new();
698            if self
699                .config
700                .ldap_user_auxiliary_obj_classes
701                .contains(&UserObjectClass::SimpleSecurityObject.into())
702            {
703                mods.push(Mod::Replace(
704                    "userPassword",
705                    hashset![ssha_password.as_str()],
706                ));
707            }
708
709            if self
710                .config
711                .ldap_user_auxiliary_obj_classes
712                .contains(&UserObjectClass::SambaSamAccount.into())
713            {
714                mods.push(Mod::Replace(
715                    "sambaNTPassword",
716                    hashset![nt_password.as_str()],
717                ));
718            }
719
720            if mods.is_empty() {
721                return Err(LdapError::MissingSettings(
722                    format!("Can't set password as no password object class has been defined for the user {user}."),
723                ));
724            }
725
726            self.modify(&user_dn, &user_dn, mods).await?;
727            info!("Password set for user {user}");
728        };
729
730        Ok(())
731    }

733    /// This exists as some LDAP servers don't allow for creating empty groups
734    /// Notable example: OpenLDAP
735    pub async fn add_group_with_members<I>(
736        &mut self,
737        group_name: &str,
738        members: Vec<&User<I>>,
739    ) -> Result<(), LdapError> {
740        debug!("Adding LDAP group {}", group_name);
741        let dn = self.config.group_dn(group_name);
742        let group_obj_class = self.config.ldap_group_obj_class.clone();
743        let groupname_attr = self.config.ldap_groupname_attr.clone();
744        let mut group_attrs = vec![
745            ("objectClass", hashset![group_obj_class.as_str()]),
746            (groupname_attr.as_str(), hashset![group_name]),
747        ];
748        //   extent the group attr with multiple members
749        let member_dns = members
750            .into_iter()
751            .map(|member| self.config.user_dn(member.ldap_rdn_value()))
752            .collect::<Vec<_>>();
753        let member_group_attr = self.config.ldap_group_member_attr.clone();
754        let member_refs: HashSet<&str> = member_dns.iter().map(|s| s.as_str()).collect();
755
756        for member_ref in member_refs {
757            group_attrs.push((member_group_attr.as_str(), hashset![member_ref]));
758        }
759
760        self.add(&dn, group_attrs).await?;
761        info!(
762            "Added LDAP group {} with members {}",
763            group_name,
764            member_dns
765                .iter()
766                .map(|dn| dn.as_str())
767                .collect::<Vec<_>>()
768                .join(", ")
769        );
770
771        Ok(())
772    }

807    /// Add user to a group.
808    pub async fn add_user_to_group<I>(
809        &mut self,
810        user: &User<I>,
811        groupname: &str,
812    ) -> Result<(), LdapError> {
813        debug!(
814            "Adding user {user} to group {groupname} in LDAP, checking if that group exists first..."
815        );
816        let user_dn = self.config.user_dn(user.ldap_rdn_value());
817        if self.is_member_of(&user_dn, groupname).await? {
818            debug!("User {user} is already a member of group {groupname}, skipping");
819            return Ok(());
820        }
821        if !self.group_exists(groupname).await? {
822            debug!("Group {groupname} doesn't exist in LDAP, creating it");
823            self.add_group_with_members(groupname, vec![user]).await?;
824            debug!("Group {groupname} created and member added in LDAP");
825        } else {
826            debug!("Group {groupname} exists in LDAP, adding user {user} to it");
827            let group_dn = self.config.group_dn(groupname);
828            self.modify(
829                &group_dn,
830                &group_dn,
831                vec![Mod::Add(
832                    &self.config.ldap_group_member_attr.clone(),
833                    hashset![&user_dn],
834                )],
835            )
836            .await?;
837            debug!("Added user {user} to group {groupname} in LDAP");
838        }
839        info!("Added user {user} to group {groupname} in LDAP");
840        Ok(())
841    }

843    /// Remove user from a group.
844    pub async fn remove_user_from_group<I>(
845        &mut self,
846        user: &User<I>,
847        groupname: &str,
848    ) -> Result<(), LdapError> {
849        debug!("Removing user {user} from group {groupname} in LDAP");
850        let user_dn = self.config.user_dn(user.ldap_rdn_value());
851        if !self.is_member_of(&user_dn, groupname).await? {
852            debug!("User {user} is not a member of group {groupname}, skipping");
853            return Ok(());
854        }
855        let members = self.get_group_members(groupname).await?;
856        if members.len() > 1 {
857            let group_dn = self.config.group_dn(groupname);
858            self.modify(
859                &group_dn,
860                &group_dn,
861                vec![Mod::Delete(
862                    &self.config.ldap_group_member_attr.clone(),
863                    hashset![&user_dn],
864                )],
865            )
866            .await?;
867            debug!("Removed user {user} from group {groupname} in LDAP");
868        } else {
869            debug!("Group {groupname} has only one member, removing the whole group",);
870            self.delete_group(groupname).await?;
871            debug!("Removed group {groupname} from LDAP");
872        }
873
874        info!("Removed user {user} from group {groupname} in LDAP");
875
876        Ok(())
877    }
