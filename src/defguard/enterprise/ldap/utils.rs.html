15pub(crate) async fn login_through_ldap(
16    pool: &PgPool,
17    username: &str,
18    password: &str,
19) -> Result<User<Id>, LdapError> {
20    debug!("Logging in user {username} through LDAP");
21    let mut ldap_connection = LDAPConnection::create().await?;
22    let mut ldap_user = ldap_connection
23        .fetch_user_by_credentials(username, password)
24        .await?;
25    if !ldap_connection.user_in_ldap_sync_groups(&ldap_user).await? {
26        info!("User {username} is not in LDAP sync groups, not allowing to login through LDAP.",);
27        return Err(LdapError::UserNotInLDAPSyncGroups(
28            username.to_string(),
29            "LDAP",
30        ));
31    }
32    debug!("User {ldap_user} logged in through LDAP");
33    // The user is logging in through LDAP, so we can infer that there are no other login options (Defguard password),
34    // so we should mark them as from_ldap.
35    let user = if let Some(mut defguard_user) =
36        User::find_by_username(pool, &ldap_user.username).await?
37    {
38        debug!("User {defguard_user} already exists in Defguard, marking them as coming from LDAP and proceeding with login");
39        defguard_user.from_ldap = true;
40        defguard_user.save(pool).await?;
41        defguard_user
42    } else {
43        debug!(
44            "User {ldap_user} doesn't exist in Defguard, creating them first based on LDAP data"
45        );
46        ldap_user.from_ldap = true;
47        ldap_user.save(pool).await?
48    };
49
50    Ok(user)
51}

53pub(crate) async fn user_from_ldap(
54    pool: &PgPool,
55    username: &str,
56    password: &str,
57) -> Result<User<Id>, LdapError> {
58    debug!("Getting user {username} from LDAP");
59    let mut ldap_connection = LDAPConnection::create().await?;
60
61    let mut ldap_user = ldap_connection
62        .fetch_user_by_credentials(username, password)
63        .await?;
64    if !ldap_connection.user_in_ldap_sync_groups(&ldap_user).await? {
65        return Err(LdapError::UserNotInLDAPSyncGroups(
66            username.to_string(),
67            "LDAP",
68        ));
69    }
70    ldap_user.from_ldap = true;
71    let user = ldap_user.save(pool).await?;
72
73    debug!("User {user} found in LDAP");
74
75    Ok(user)
76}

95/// Adds user to LDAP, if no password was specified, a temporary random password will be used.
96/// This will set the `ldap_pass_randomized` field to `true` in the user.
97///
98/// If the user already exists, the creation will be skipped.
99pub(crate) async fn ldap_add_user(user: &mut User<Id>, password: Option<&str>, pool: &PgPool) {
100    let _: Result<(), LdapError> = with_ldap_status(pool, async {
101        debug!("Creating user {user} in LDAP");
102        if !user.ldap_sync_allowed(pool).await? {
103            debug!("User {user} is not allowed to be synced to LDAP as he is not in the specified sync groups, skipping");
104            return Ok(());
105        }
106        let mut ldap_connection = LDAPConnection::create().await?;
107        if ldap_connection.user_exists(user).await? {
108            debug!("User {user} already exists in LDAP, skipping creation");
109            return Ok(());
110        }
111        match ldap_connection.add_user(user, password, pool).await {
112            Ok(()) => Ok(()),
113            // this user might exist in LDAP, just try to set the password
114            Err(err) => {
115                warn!("There was an error while trying to create the user {user} in LDAP: {err}");
116                debug!(
117                    "Trying to set password for user {user} in LDAP, in case he already existed",
118                );
119                if let Some(password) = password {
120                    ldap_connection.set_password(user, password).await?;
121                    debug!("Password set for user {user} in LDAP");
122                } else {
123                    debug!(
124                        "No password provided, skipping password setting for user {user} in LDAP"
125                    );
126                }
127                Ok(())
128            }
129        }
130    })
131    .await;
132}

134/// Applies user modifications to LDAP. May update the user object if
135/// his RDN in Defguard needs updating. Fails and sets the sync status to desynced
136/// if the user does not exist in LDAP despite updating his state.
137///
138/// Warning: This function does not check if the user is allowed to be synced to LDAP. You must
139/// do that manually before calling this function. For example, by calling the [`User::ldap_sync_allowed`] method on the user.
140pub(crate) async fn ldap_handle_user_modify(
141    old_username: &str,
142    current_user: &mut User<Id>,
143    pool: &PgPool,
144) {
145    let _: Result<(), LdapError> = with_ldap_status(pool, async {
146        debug!("Handling user modify for {old_username} in LDAP");
147
148        let mut ldap_connection = LDAPConnection::create().await?;
149        if !ldap_connection.user_exists(current_user).await? {
150            debug!("User {current_user} doesn't exist in LDAP, updating his state first as it may be stale");
151            ldap_connection.update_users_state(vec![current_user], pool).await?;
152        } else {
153            debug!("User {current_user} exists in LDAP, modifying it");
154        }
155        ldap_connection
156            .modify_user(old_username, current_user)
157            .await
158    })
159    .await;
160}

172/// Deletes multiple users from LDAP.
173///
174/// WARNING: You must check whether the users should be deleted from LDAP or not before calling this function.
175/// For example, by calling the [`User::ldap_sync_allowed`] method on each user.
176//
177// The mentioned method can't be called here since the user is already dropped from the database
178pub(crate) async fn ldap_delete_users<I>(users: Vec<&User<I>>, pool: &PgPool) {
179    let _: Result<(), LdapError> = with_ldap_status(pool, async {
180        debug!("Deleting {:?} users from LDAP", users.len());
181        let mut ldap_connection = LDAPConnection::create().await?;
182        for user in users {
183            debug!("Deleting user {user} from LDAP");
184            ldap_connection.delete_user(user).await?;
185            debug!("User {user} deleted from LDAP");
186        }
187
188        Ok(())
189    })
190    .await;
191}

209/// Bulk add users to groups in ldap.
210pub(crate) async fn ldap_add_users_to_groups(
211    user_groups: HashMap<&User<Id>, HashSet<&str>>,
212    pool: &PgPool,
213) {
214    let _: Result<(), LdapError> = with_ldap_status(pool, async {
215        let mut ldap_connection = LDAPConnection::create().await?;
216        let sync_groups = ldap_connection.config.ldap_sync_groups.clone();
217        let sync_groups_lookup = sync_groups.iter().map(|s| s.as_str()).collect::<HashSet<_>>();
218
219        for (user, groups) in user_groups {
220            let adding_to_sync_groups = groups
221                .iter()
222                .any(|group| sync_groups_lookup.contains(*group));
223            if !user.ldap_sync_allowed(pool).await? && !adding_to_sync_groups {
224                debug!("User {user} is not allowed to be synced to LDAP as he is not in the specified sync groups, skipping");
225                continue;
226            }
227
228            for group in groups {
229                ldap_connection.add_user_to_group(user, group).await?;
230            }
231        }
232
233        Ok(())
234    })
235    .await;
236}

238/// Bulk remove users from groups in ldap.
239pub(crate) async fn ldap_remove_users_from_groups(
240    user_groups: HashMap<&User<Id>, HashSet<&str>>,
241    pool: &PgPool,
242) {
243    let _: Result<(), LdapError> = with_ldap_status(pool, async {
244        let mut ldap_connection = LDAPConnection::create().await?;
245        let sync_groups = ldap_connection.config.ldap_sync_groups.clone();
246        let sync_groups_lookup = sync_groups.iter().map(|s| s.as_str()).collect::<HashSet<_>>();
247
248        for (user, groups) in user_groups {
249            let removing_from_sync_groups = groups
250                .iter()
251                .any(|group| sync_groups_lookup.contains(*group));
252            if !user.ldap_sync_allowed(pool).await? && !removing_from_sync_groups {
253                debug!("User {user} is not allowed to be synced to LDAP as he is not in the specified sync groups, skipping");
254                continue;
255            }
256            for group in groups {
257                if ldap_connection.group_exists(group).await? {
258                    ldap_connection.remove_user_from_group(user, group).await?;
259                } else {
260                    debug!("Group {group} doesn't exist in LDAP, skipping removal of user {user}");
261                }
262            }
263        }
264
265        Ok(())
266    })
267    .await;
268}

270pub(crate) async fn ldap_change_password(user: &mut User<Id>, password: &str, pool: &PgPool) {
271    let _: Result<(), LdapError> = with_ldap_status(pool, async {
272        debug!("Changing password for user {user} in LDAP");
273        if !user.ldap_sync_allowed(pool).await? {
274            debug!("User {user} is not allowed to be synced to LDAP as he is not in the specified sync groups, skipping");
275            return Ok(());
276        }
277        let mut ldap_connection = LDAPConnection::create().await?;
278        if !ldap_connection.user_exists(user).await? {
279            debug!("User {user} doesn't exist in LDAP, creating it with the provided password");
280            let user_groups = user.member_of_names(pool).await?;
281            ldap_connection.add_user(user, Some(password), pool).await?;
282            for group in user_groups {
283                ldap_connection.add_user_to_group(user, &group).await?;
284            }
285           debug!("User {user} created in LDAP with the provided password");
286        } else {
287            debug!("User {user} exists in LDAP, changing password");
288            ldap_connection
289                .set_password(user, password)
290                .await?;
291            debug!(
292                "Password changed for user {user} in LDAP, marking the LDAP password as set in Defguard"
293            );
294            user.ldap_pass_randomized = false;
295            user.save(pool).await?;
296            debug!(
297                "LDAP password state updated in Defguard for user {user}"
298            );
299        }
300
301        Ok(())
302    })
303    .await;
304}
