29/// Converts ACLs into firewall rules which can be sent to a gateway over gRPC.
30///
31/// Each ACL is translated into two rules:
32/// - ALLOW which determines which devices can access a destination
33/// - DENY which stops all other traffic to a given destination
34///
35/// In the resulting list all ALLOW rules are placed first and then DENY rules are added to the
36/// end. This way we can avoid conflicts when some ACLs are overlapping.
37pub async fn generate_firewall_rules_from_acls(
38    location_id: Id,
39    ip_version: IpVersion,
40    acl_rules: Vec<AclRuleInfo<Id>>,
41    conn: &mut PgConnection,
42) -> Result<Vec<FirewallRule>, FirewallError> {
43    debug!("Generating firewall rules for location {location_id} with IP version {ip_version:?}");
44    // initialize empty rules Vec
45    let mut allow_rules = Vec::new();
46    let mut deny_rules = Vec::new();
47
48    // convert each ACL into a corresponding `FirewallRule`s
49    for acl in acl_rules {
50        debug!("Processing ACL rule: {acl:?}");
51        // fetch allowed users
52        let allowed_users = acl.get_all_allowed_users(&mut *conn).await?;
53
54        // fetch denied users
55        let denied_users = acl.get_all_denied_users(&mut *conn).await?;
56
57        // get relevant users for determining source IPs
58        let users = get_source_users(allowed_users, denied_users);
59
60        // get network IPs for devices belonging to those users
61        let user_device_ips = get_user_device_ips(&users, location_id, &mut *conn).await?;
62
63        // fetch allowed network devices
64        let allowed_network_devices = acl.get_all_allowed_devices(&mut *conn, location_id).await?;
65
66        // fetch denied network devices
67        let denied_network_devices = acl.get_all_denied_devices(&mut *conn, location_id).await?;
68
69        // get network device IPs for rule source
70        let network_devices =
71            get_source_network_devices(allowed_network_devices, denied_network_devices);
72        let network_device_ips =
73            get_network_device_ips(&network_devices, location_id, &mut *conn).await?;
74
75        // convert device IPs into source addresses for a firewall rule
76        let source_addrs = get_source_addrs(user_device_ips, network_device_ips, ip_version);
77
78        // extract destination parameters from ACL rule
79        let AclRuleInfo {
80            id,
81            mut destination,
82            destination_ranges,
83            mut ports,
84            mut protocols,
85            aliases,
86            ..
87        } = acl;
88
89        // split aliases into types
90        let (destination_aliases, component_aliases): (Vec<_>, Vec<_>) = aliases
91            .into_iter()
92            .partition(|alias| alias.kind == AliasKind::Destination);
93
94        // store alias ranges separately since they use a different struct
95        let mut alias_destination_ranges = Vec::new();
96
97        // process component aliases by appending destination parameters from each of them to existing lists
98        for alias in component_aliases {
99            // fetch destination ranges for a given alias
100            alias_destination_ranges.extend(alias.get_destination_ranges(&mut *conn).await?);
101
102            // extend existing parameter lists
103            destination.extend(alias.destination);
104            ports.extend(
105                alias
106                    .ports
107                    .into_iter()
108                    .map(|port_range| port_range.into())
109                    .collect::<Vec<PortRange>>(),
110            );
111            protocols.extend(alias.protocols);
112        }
113
114        // prepare destination addresses
115        let destination_addrs =
116            process_destination_addrs(destination, destination_ranges, ip_version);
117
118        // prepare destination ports
119        let destination_ports = merge_port_ranges(ports);
120
121        // remove duplicates protocol entries
122        protocols.sort();
123        protocols.dedup();
124
125        // skip creating default firewall rules if given ACL includes only destination aliases and no manual destination config
126        let has_no_manual_destination =
127            destination_addrs.is_empty() && destination_ports.is_empty() && protocols.is_empty();
128        let has_destination_aliases = !destination_aliases.is_empty();
129        let is_destination_alias_only_rule = has_destination_aliases && has_no_manual_destination;
130
131        if !is_destination_alias_only_rule {
132            // check if source addrs list is empty
133            if source_addrs.is_empty() {
134                debug!(
135                    "Source address list is empty. Skipping generating the ALLOW rule for this ACL"
136                );
137            } else {
138                // prepare ALLOW rule for this ACL
139                let allow_rule = FirewallRule {
140                    id: acl.id,
141                    source_addrs: source_addrs.clone(),
142                    destination_addrs: destination_addrs.clone(),
143                    destination_ports,
144                    protocols,
145                    verdict: i32::from(FirewallPolicy::Allow),
146                    comment: Some(format!("ACL {} - {} ALLOW", acl.id, acl.name)),
147                };
148                debug!("ALLOW rule generated from ACL: {allow_rule:?}");
149                allow_rules.push(allow_rule);
150            };
151
152            // prepare DENY rule for this ACL
153            //
154            // it should specify only the destination addrs to block all remaining traffic
155            let deny_rule = FirewallRule {
156                id: acl.id,
157                source_addrs: Vec::new(),
158                destination_addrs,
159                destination_ports: Vec::new(),
160                protocols: Vec::new(),
161                verdict: i32::from(FirewallPolicy::Deny),
162                comment: Some(format!("ACL {} - {} DENY", acl.id, acl.name)),
163            };
164            debug!("DENY rule generated from ACL: {deny_rule:?}");
165            deny_rules.push(deny_rule);
166        }
167
168        // process destination aliases by creating a dedicated set of rules for each of them
169        if !destination_aliases.is_empty() {
170            debug!(
171                "Generating firewall rules for {} aliases used in ACL rule {id:?}",
172                destination_aliases.len()
173            );
174        }
175        for alias in destination_aliases {
176            debug!("Processing ACL alias: {alias:?}");
177
178            // fetch destination ranges for a given alias
179            let alias_destination_ranges = alias.get_destination_ranges(&mut *conn).await?;
180
181            // combine destination addrs
182            let destination_addrs = process_alias_destination_addrs(
183                alias.destination,
184                alias_destination_ranges,
185                ip_version,
186            );
187
188            // process alias ports
189            let alias_ports = alias
190                .ports
191                .into_iter()
192                .map(|port_range| port_range.into())
193                .collect::<Vec<PortRange>>();
194            let destination_ports = merge_port_ranges(alias_ports);
195
196            // remove duplicates protocol entries
197            let mut protocols = alias.protocols;
198            protocols.sort();
199            protocols.dedup();
200
201            if source_addrs.is_empty() {
202                debug!(
203                    "Source address list is empty. Skipping generating the ALLOW rule for this alias"
204                );
205            } else {
206                // prepare ALLOW rule for this alias
207                let alias_allow_rule = FirewallRule {
208                    id: acl.id,
209                    source_addrs: source_addrs.clone(),
210                    destination_addrs: destination_addrs.clone(),
211                    destination_ports,
212                    protocols,
213                    verdict: i32::from(FirewallPolicy::Allow),
214                    comment: Some(format!(
215                        "ACL {} - {}, ALIAS {} - {} ALLOW",
216                        acl.id, acl.name, alias.id, alias.name
217                    )),
218                };
219                debug!("ALLOW rule generated from ACL: {alias_allow_rule:?}");
220                allow_rules.push(alias_allow_rule);
221            };
222
223            // prepare DENY rule for this ACL
224            //
225            // it should specify only the destination addrs to block all remaining traffic
226            let alias_deny_rule = FirewallRule {
227                id: acl.id,
228                source_addrs: Vec::new(),
229                destination_addrs,
230                destination_ports: Vec::new(),
231                protocols: Vec::new(),
232                verdict: i32::from(FirewallPolicy::Deny),
233                comment: Some(format!(
234                    "ACL {} - {}, ALIAS {} - {} DENY",
235                    acl.id, acl.name, alias.id, alias.name
236                )),
237            };
238            debug!("DENY rule generated from ACL: {alias_deny_rule:?}");
239            deny_rules.push(alias_deny_rule);
240        }
241    }
242
243    // combine both rule lists
244    Ok(allow_rules.into_iter().chain(deny_rules).collect())
245}

320/// Combines user device IPs and network device IPs into a list of source addresses which can be
321/// used by a firewall rule.
322fn get_source_addrs(
323    user_device_ips: Vec<IpAddr>,
324    network_device_ips: Vec<IpAddr>,
325    ip_version: IpVersion,
326) -> Vec<IpAddress> {
327    // combine both lists into a single iterator
328    let source_ips = user_device_ips.into_iter().chain(network_device_ips);
329
330    // prepare source addrs by removing incompatible IP version elements
331    // and converting them to expected gRPC format
332    let source_addrs = source_ips
333        .filter_map(|ip| match ip_version {
334            IpVersion::Ipv4 => {
335                if ip.is_ipv4() {
336                    Some(ip_to_range(ip, ip))
337                } else {
338                    None
339                }
340            }
341            IpVersion::Ipv6 => {
342                if ip.is_ipv6() {
343                    Some(ip_to_range(ip, ip))
344                } else {
345                    None
346                }
347            }
348        })
349        .collect();
350
351    // merge address ranges into non-overlapping elements
352    merge_addrs(source_addrs)
353}

355/// Convert destination networks and ranges configured in an ACL rule
356/// into the correct format for a firewall rule. This includes:
357/// - combining all addr lists
358/// - converting to gRPC IpAddress struct
359/// - filtering out incompatible IP version
360/// - merging into the smallest possible list of non-overlapping ranges,
361///   subnets and addresses
362fn process_destination_addrs(
363    destination_ips: Vec<IpNetwork>,
364    destination_ranges: Vec<AclRuleDestinationRange<Id>>,
365    ip_version: IpVersion,
366) -> Vec<IpAddress> {
367    // filter out destinations with incompatible IP version and convert to intermediate
368    // range representation for merging
369    let destination_iterator = destination_ips.iter().filter_map(|dst| match ip_version {
370        IpVersion::Ipv4 => {
371            if dst.is_ipv4() {
372                Some(ip_to_range(dst.network(), dst.broadcast()))
373            } else {
374                None
375            }
376        }
377        IpVersion::Ipv6 => {
378            if let IpNetwork::V6(subnet) = dst {
379                let range_start = subnet.network().into();
380                let range_end = get_last_ip_in_v6_subnet(subnet);
381                Some(ip_to_range(range_start, range_end))
382            } else {
383                None
384            }
385        }
386    });
387
388    // filter out destination ranges with incompatible IP version and convert to intermediate
389    // range representation for merging
390    let destination_range_iterator = destination_ranges
391        .iter()
392        .filter_map(|dst| match ip_version {
393            IpVersion::Ipv4 => {
394                if dst.start.is_ipv4() && dst.end.is_ipv4() {
395                    Some(ip_to_range(dst.start, dst.end))
396                } else {
397                    None
398                }
399            }
400            IpVersion::Ipv6 => {
401                if dst.start.is_ipv6() && dst.end.is_ipv6() {
402                    Some(ip_to_range(dst.start, dst.end))
403                } else {
404                    None
405                }
406            }
407        });
408
409    // combine both iterators to return a single list
410    let destination_addrs = destination_iterator
411        .chain(destination_range_iterator)
412        .collect();
413
414    // merge address ranges into non-overlapping elements
415    merge_addrs(destination_addrs)
416}

484fn ip_to_range(first_ip: IpAddr, last_ip: IpAddr) -> Range<IpAddr> {

488fn range_to_ip(ip_range: Range<IpAddr>) -> (IpAddr, IpAddr) {
489    let first_ip = ip_range.start;
490    let last_ip = previous_ip(ip_range.end);
491
492    (first_ip, last_ip)

495/// Returns the next IP address in sequence, handling overflow via wrapping
496fn next_ip(ip: IpAddr) -> IpAddr {
497    match ip {
498        IpAddr::V4(ipv4) => {
499            let octets = ipv4.octets();
500            let mut num: u32 = ((octets[0] as u32) << 24)
501                | ((octets[1] as u32) << 16)
502                | ((octets[2] as u32) << 8)
503                | octets[3] as u32;
504            num = num.wrapping_add(1);
505            IpAddr::V4(Ipv4Addr::from(num))
506        }
507        IpAddr::V6(ipv6) => {
508            let segments = ipv6.segments();
509            let mut num: u128 = ((segments[0] as u128) << 112)
510                | ((segments[1] as u128) << 96)
511                | ((segments[2] as u128) << 80)
512                | ((segments[3] as u128) << 64)
513                | ((segments[4] as u128) << 48)
514                | ((segments[5] as u128) << 32)
515                | ((segments[6] as u128) << 16)
516                | segments[7] as u128;
517            num = num.wrapping_add(1);
518            IpAddr::V6(Ipv6Addr::from(num))
519        }
520    }
521}

523/// Returns the previous IP address in sequence, handling underflow via wrapping
524fn previous_ip(ip: IpAddr) -> IpAddr {
525    match ip {
526        IpAddr::V4(ipv4) => {
527            let octets = ipv4.octets();
528            let mut num: u32 = ((octets[0] as u32) << 24)
529                | ((octets[1] as u32) << 16)
530                | ((octets[2] as u32) << 8)
531                | octets[3] as u32;
532            num = num.wrapping_sub(1);
533            IpAddr::V4(Ipv4Addr::from(num))
534        }
535        IpAddr::V6(ipv6) => {
536            let segments = ipv6.segments();
537            let mut num: u128 = ((segments[0] as u128) << 112)
538                | ((segments[1] as u128) << 96)
539                | ((segments[2] as u128) << 80)
540                | ((segments[3] as u128) << 64)
541                | ((segments[4] as u128) << 48)
542                | ((segments[5] as u128) << 32)

545            num = num.wrapping_sub(1);
546            IpAddr::V6(Ipv6Addr::from(num))
547        }
548    }

551fn get_last_ip_in_v6_subnet(subnet: &Ipv6Network) -> IpAddr {
552    // get subnet IP portion as u128
553    let first_ip = subnet.ip().to_bits();
554
555    let last_ip = first_ip | (!u128::from(subnet.mask()));
556
557    IpAddr::V6(last_ip.into())

560/// Converts an arbitrary list of ip address ranges into the smallest possible list
561/// of non-overlapping elements which can be used in a firewall rule.
562/// It assumes that all ranges with an invalid IP version have already been filtered out.
563fn merge_addrs(addr_ranges: Vec<Range<IpAddr>>) -> Vec<IpAddress> {
564    // merge into non-overlapping ranges
565    let addr_ranges = merge_ranges(addr_ranges);
566
567    // convert to gRPC format
568    let mut result = Vec::new();
569    for range in addr_ranges {
570        let (range_start, range_end) = range_to_ip(range);
571        if range_start == range_end {
572            // single IP address
573            result.push(IpAddress {
574                address: Some(Address::Ip(range_start.to_string())),
575            });
576        } else {
577            // TODO: find largest subnet in range
578            // address range
579            result.push(IpAddress {
580                address: Some(Address::IpRange(IpRange {
581                    start: range_start.to_string(),
582                    end: range_end.to_string(),
583                })),
584            });
585        }
586    }
587
588    result
589}

591/// Takes a list of port ranges and returns the smallest possible non-overlapping list of `Port`s.
592fn merge_port_ranges(port_ranges: Vec<PortRange>) -> Vec<Port> {
593    // convert ranges to a list of tuples for merging
594    let port_ranges = port_ranges.into_iter().map(|range| range.0).collect();
595
596    // merge into non-overlapping ranges
597    let port_ranges = merge_ranges(port_ranges);
598
599    // convert resulting ranges into gRPC format
600    port_ranges
601        .into_iter()
602        .map(|range| {
603            let range_start = range.start;
604            let range_end = range.end - 1;
605            if range_start == range_end {
606                Port {
607                    port: Some(PortInner::SinglePort(range_start as u32)),
608                }
609            } else {
610                Port {
611                    port: Some(PortInner::PortRange(PortRangeProto {
612                        start: range_start as u32,
613                        end: range_end as u32,
614                    })),
615                }
616            }
617        })
618        .collect()
619}

621/// Helper function which implements merging a set of ranges of arbitrary elements
622/// into the smallest possible set of non-overlapping ranges.
623/// It can then be reused for merging port and address ranges.
624fn merge_ranges<T: Ord + std::fmt::Debug>(mut ranges: Vec<Range<T>>) -> Vec<Range<T>> {
625    // return early if list is empty
626    if ranges.is_empty() {
627        return Vec::new();
628    }
629
630    // sort elements by range start
631    ranges.sort_by(|a, b| {
632        let a_start = &a.start;
633        let b_start = &b.start;
634        a_start.cmp(b_start)
635    });
636
637    // initialize result vector
638    let mut merged_ranges = Vec::new();
639
640    // start with first range
641    let current_range = ranges.remove(0);
642    let mut current_range_start = current_range.start;
643    let mut current_range_end = current_range.end;
644
645    // iterate over remaining ranges
646    for range in ranges {
647        let range_start = range.start;
648        let range_end = range.end;
649
650        // compare with current range
651        if range_start <= current_range_end {
652            // ranges are overlapping, merge them
653            // if range is not contained within current range
654            if range_end > current_range_end {
655                current_range_end = range_end;
656            }
657        } else {
658            // ranges are not overlapping, add current range to result
659            merged_ranges.push(current_range_start..current_range_end);
660            current_range_start = range_start;
661            current_range_end = range_end;
662        }
663    }
664
665    // add last remaining range
666    merged_ranges.push(Range {

676    /// Fetches all active ACL rules for a given location.
677    /// Filters out rules which are disabled, expired or have not been deployed yet.
678    pub(crate) async fn get_active_acl_rules(
679        &self,
680        conn: &mut PgConnection,
681    ) -> Result<Vec<AclRuleInfo<Id>>, SqlxError> {
682        debug!("Fetching active ACL rules for location {self}");
683        let rules: Vec<AclRule<Id>> = query_as(
684            "SELECT DISTINCT ON (a.id) a.id, name, allow_all_users, deny_all_users, all_networks, allow_all_network_devices, \
685                deny_all_network_devices, destination, ports, protocols, expires, enabled, parent_id, state \
686                FROM aclrule a \
687                LEFT JOIN aclrulenetwork an \
688                ON a.id = an.rule_id \
689                WHERE (an.network_id = $1 OR a.all_networks = true) AND enabled = true \
690                AND state = 'applied'::aclrule_state \
691                AND (expires IS NULL OR expires > NOW())",
692        )
693        .bind(self.id)
694        .fetch_all(&mut *conn)
695        .await?;
696        debug!("Found {} active ACL rules for location {self}", rules.len());
697
698        // convert to `AclRuleInfo`
699        let mut rules_info = Vec::new();
700        for rule in rules {
701            let rule_info = rule.to_info(&mut *conn).await?;
702            rules_info.push(rule_info);
703        }
704        Ok(rules_info)
705    }

707    /// Prepares firewall configuration for a gateway based on location config and ACLs
708    /// Returns `None` if firewall management is disabled for a given location.
709    pub async fn try_get_firewall_config(
710        &self,
711        conn: &mut PgConnection,
712    ) -> Result<Option<FirewallConfig>, FirewallError> {
713        // do a license check
714        if !is_enterprise_enabled() {
715            debug!(
716                "Enterprise features are disabled, skipping generating firewall config for location {self}"
717            );
718            return Ok(None);
719        }
720
721        // check if ACLs are enabled
722        if !self.acl_enabled {
723            debug!(
724                "ACL rules are disabled for location {self}, skipping generating firewall config"
725            );
726            return Ok(None);
727        }
728
729        info!("Generating firewall config for location {self}");
730        // fetch all active ACLs for location
731        let location_acls = self.get_active_acl_rules(&mut *conn).await?;
732
733        // determine IP version based on location subnet
734        let ip_version = self.get_ip_version();
735
736        let default_policy = match self.acl_default_allow {
737            true => FirewallPolicy::Allow,
738            false => FirewallPolicy::Deny,
739        };
740        let firewall_rules =
741            generate_firewall_rules_from_acls(self.id, ip_version, location_acls, &mut *conn)
742                .await?;
743
744        let firewall_config = FirewallConfig {
745            ip_version: ip_version.into(),
746            default_policy: default_policy.into(),
747            rules: firewall_rules,
748        };
749
750        debug!("Firewall config generated for location {self}: {firewall_config:?}");
751        Ok(Some(firewall_config))
752    }
753
754    fn get_ip_version(&self) -> IpVersion {
755        // get the subnet from which device IPs are being assigned
756        // by default only the first configured subnet is being used
757        let vpn_subnet = self
758            .address
759            .first()
760            .expect("WireguardNetwork must have an address");
761
762        let ip_version = match vpn_subnet {
763            IpNetwork::V4(_ipv4_network) => IpVersion::Ipv4,
764            IpNetwork::V6(_ipv6_network) => IpVersion::Ipv6,
765        };
766        debug!("VPN subnet {vpn_subnet:?} for location {self} has IP version {ip_version:?}");
767
768        ip_version
769    }

1476                id: NoId,
1477                allow: false,
1478                rule_id,
1479                user_id,
1480            };
1481            obj.save(&mut *conn).await.unwrap();
1482        }

1486            let obj = AclRuleGroup {
1487                id: NoId,
1488                allow: true,
1489                rule_id,
1490                group_id,
1491            };
1492            obj.save(&mut *conn).await.unwrap();
1493        }

1497            let obj = AclRuleGroup {
1498                id: NoId,
1499                allow: false,
1500                rule_id,
1501                group_id,
1502            };
1503            obj.save(&mut *conn).await.unwrap();
1504        }

1514            obj.save(&mut *conn).await.unwrap();

1524            };

1545            };
1546            obj.save(&mut *conn).await.unwrap();

1550        rule.to_info(&mut conn).await.unwrap()
1551    }
