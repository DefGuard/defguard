27pub fn extract_next_link(
28    link_header: Option<&String>,
29) -> Result<Option<String>, DirectorySyncError> {
30    if let Some(header) = link_header {
31        let mut res = parse_with_rel(header).map_err(|e| {
32            DirectorySyncError::InvalidUrl(format!("Failed to parse link header: {e:?}"))
33        })?;
34        Ok(res.remove("next").map(|x| x.raw_uri))
35    } else {
36        Ok(None)
37    }
38}

40#[derive(Debug, Serialize, Deserialize)]
41struct Claims {
42    iss: String,
43    aud: String,
44    sub: String,
45    exp: i64,
46    iat: i64,
47}
48
49impl Claims {
50    #[must_use]
51    fn new(client_id: &str, base_url: &str) -> Self {
52        let now = Utc::now();
53        let now_timestamp = now.timestamp();
54        let exp = now_timestamp + 3600;
55        Self {
56            iss: client_id.into(),
57            aud: ACCESS_TOKEN_URL.replace("{BASE_URL}", base_url),
58            sub: client_id.into(),
59            exp,
60            iat: now_timestamp,
61        }
62    }
63}

142    pub async fn refresh_access_token(&mut self) -> Result<(), DirectorySyncError> {
143        debug!("Refreshing Okta directory sync access token");
144        let token_response = self.query_access_token().await?;
145        let expires_in = TimeDelta::seconds(token_response.expires_in);
146        debug!(
147            "Access token refreshed, the new token expires in {} seconds",
148            token_response.expires_in
149        );
150        self.access_token = Some(token_response.token);
151        self.token_expiry = Some(Utc::now() + expires_in);
152        Ok(())
153    }

156        debug!("Checking if Okta directory sync token is expired");
157        // No token = expired token
158        let result = self.token_expiry.is_none_or(|expiry| expiry < Utc::now());
159        debug!("Token is expired: {}", result);
160        result

179    async fn query_user_groups(&self, user_id: &str) -> Result<Vec<Group>, DirectorySyncError> {
180        if self.is_token_expired() {
181            return Err(DirectorySyncError::AccessTokenExpired);
182        }
183        let access_token = self
184            .access_token
185            .as_ref()
186            .ok_or(DirectorySyncError::AccessTokenExpired)?;
187        let mut url = USER_GROUPS
188            .replace("{BASE_URL}", &self.base_url)
189            .replace("{USER_ID}", user_id);
190        let mut combined_response: Vec<Group> = Vec::new();
191        let mut query = Some([("limit", MAX_RESULTS)].as_slice());
192
193        for _ in 0..MAX_REQUESTS {
194            let response = make_get_request(&url, access_token, query).await?;
195            let link_header = {
196                let links = response
197                    .headers()
198                    .get_all("link")
199                    .iter()
200                    .filter_map(|link| link.to_str().ok())
201                    .collect::<Vec<&str>>();
202
203                (!links.is_empty()).then(|| links.join(", "))
204            };
205            let result: Vec<Group> =
206                parse_response(response, "Failed to query user groups in the Okta API.").await?;
207            combined_response.extend(result);
208
209            if let Some(next_link) = extract_next_link(link_header.as_ref())? {
210                url = next_link;
211                // Query is already appended to the URL we received from the link header
212                query = None;
213                debug!("Found next page of results, querying it: {url}");
214            } else {
215                debug!("No more pages of results found, finishing query.");
216                break;
217            }
218
219            sleep(REQUEST_PAGINATION_SLOWDOWN).await;
220        }
221
222        Ok(combined_response)
223    }

318    fn build_token(&self) -> Result<String, DirectorySyncError> {
319        debug!("Building Okta directory sync auth token");
320        let claims = Claims::new(&self.client_id, &self.base_url);
321        debug!("Using the following token claims: {:?}", claims);
322        // Users provide a JWK format private key. The jsonwebtoken library currently doesn't support
323        // converting JWK to PEM or encoding key so the jsonwebkey library is used to convert the key
324        // to a PEM format.
325        debug!("Building Okta directory sync encoding key");
326        let jwk = jsonwebkey::JsonWebKey::from_str(&self.jwk_private_key)
327            .map_err(|e| DirectorySyncError::InvalidProviderConfiguration(e.to_string()))?;
328        let kid = jwk
329            .key_id
330            .ok_or(DirectorySyncError::InvalidProviderConfiguration(
331                "Missing key id in the provided JSON key".to_string(),
332            ))?;
333        let encoding_key_pem = jwk
334            .key
335            .try_to_pem()
336            .map_err(|e| DirectorySyncError::InvalidProviderConfiguration(e.to_string()))?;
337        let key = EncodingKey::from_rsa_pem(encoding_key_pem.as_bytes())?;
338        debug!("Successfully built Okta directory sync encoding key for encoding the auth token");
339        let mut header = Header::new(Algorithm::RS256);
340        header.kid = Some(kid);
341        let token = encode(&header, &claims, &key)?;
342        debug!("Successfully built Okta directory sync auth token");
343        Ok(token)
344    }

346    async fn query_access_token(&self) -> Result<AccessTokenResponse, DirectorySyncError> {
347        let token = self.build_token()?;
348        let client = reqwest::Client::new();
349        let response = client
350            .post(ACCESS_TOKEN_URL.replace("{BASE_URL}", &self.base_url))
351            .form(&[
352                ("grant_type", GRANT_TYPE),
353                ("client_assertion_type", CLIENT_ASSERTION_TYPE),
354                ("client_assertion", &token),
355                ("scope", TOKEN_SCOPE),
356            ])
357            .send()
358            .await?;
359        parse_response(response, "Failed to get access token from Okta API.").await
360    }

406impl DirectorySync for OktaDirectorySync {

408        debug!("Getting all groups");

416        user_id: &str,

418        debug!("Getting groups of user {user_id}");
419        let response = self.query_user_groups(user_id).await?;
420        debug!("Got groups response for user {user_id}");
421        Ok(response.into_iter().map(Into::into).collect())
422    }
423
424    async fn get_group_members(
425        &self,
426        group: &DirectoryGroup,
427    ) -> Result<Vec<String>, DirectorySyncError> {
428        debug!("Getting group members of group {}", group.name);
429        let response = self.query_group_members(group).await?;
430        debug!("Got group members response for group {}", group.name);
431        Ok(response
432            .into_iter()
433            .map(|user| user.profile.email)
434            .collect())
435    }
436
437    async fn prepare(&mut self) -> Result<(), DirectorySyncError> {
438        debug!("Preparing Okta directory sync...");
439        if self.is_token_expired() {
440            debug!("Access token is expired, refreshing.");
441            self.refresh_access_token().await?;
442            debug!("Access token refreshed.");
443        } else {
444            debug!("Access token is still valid, skipping refresh.");
445        }
446        debug!("Okta directory sync prepared.");
447        Ok(())
448    }
449
450    async fn get_all_users(&self) -> Result<Vec<DirectoryUser>, DirectorySyncError> {
451        debug!("Getting all users");
452        let response: Vec<User> = self.query_all_users().await?;
453        debug!("Got all users response");
454        Ok(response.into_iter().map(Into::into).collect())
455    }
