223    fn decode(bytes: &[u8]) -> Result<Vec<u8>, LicenseError> {
224        let bytes = BASE64_STANDARD.decode(bytes).map_err(|_| {
225            LicenseError::DecodeError(
226                "Failed to decode the license key, check if the provided key is correct."
227                    .to_string(),
228            )
229        })?;
230        Ok(bytes)
231    }
232
233    fn verify_signature(data: &[u8], signature: &[u8]) -> Result<(), LicenseError> {
234        let sig = StandaloneSignature::from_bytes(signature)
235            .map_err(|_| LicenseError::InvalidSignature)?;
236        let (public_key, _headers_public) =
237            SignedPublicKey::from_string(PUBLIC_KEY).expect("Failed to parse the public key");
238
239        // If the public key has subkeys, extract the signing key from them
240        // Otherwise, use the primary key
241        if public_key.public_subkeys.is_empty() {
242            debug!(
243                "Using the public key's primary key {:?} to verify the signature...",
244                public_key.key_id()
245            );
246            sig.verify(&public_key, data)
247                .map_err(|_| LicenseError::SignatureMismatch)
248        } else {
249            let signing_key = public_key
250                .public_subkeys
251                .into_iter()
252                .find(PublicKeyTrait::is_signing_key)
253                .ok_or(LicenseError::LicenseServerError(
254                    "Failed to find a signing key in the provided public key".to_string(),
255                ))?;
256            debug!(
257                "Using the public key's subkey {:?} to verify the signature...",
258                signing_key.key_id()
259            );
260            sig.verify(&signing_key, data)
261                .map_err(|_| LicenseError::SignatureMismatch)
262        }
263    }

265    /// Deserialize the license object from a base64 encoded string.
266    /// Also verifies the signature of the license
267    pub fn from_base64(key: &str) -> Result<License, LicenseError> {
268        debug!("Decoding the license key from a provided base64 string...");
269        let bytes = key.as_bytes();
270        let decoded = Self::decode(bytes)?;
271        let slice: &[u8] = &decoded;
272        debug!("Decoded the license key, deserializing the license object...");
273
274        let license_key = LicenseKey::decode(slice).map_err(|_| {
275            LicenseError::DecodeError(
276                "The license key is malformed, check if the provided key is correct.".to_string(),
277            )
278        })?;
279        let metadata_bytes: &[u8] = &license_key.metadata;
280        let signature_bytes: &[u8] = &license_key.signature;
281        debug!("Deserialized the license object, verifying the license signature...");
282
283        match Self::verify_signature(metadata_bytes, signature_bytes) {
284            Ok(()) => {
285                info!("Successfully decoded the license and validated the license signature");
286                let metadata = LicenseMetadata::decode(metadata_bytes).map_err(|_| {
287                    LicenseError::DecodeError("Failed to decode the license metadata".to_string())
288                })?;
289
290                let valid_until = match metadata.valid_until {
291                    Some(until) => DateTime::from_timestamp(until, 0),
292                    None => None,
293                };
294
295                let license = License::new(
296                    metadata.customer_id,
297                    metadata.subscription,
298                    valid_until,
299                    metadata.limits,
300                );
301
302                if license.requires_renewal() {
303                    if license.is_max_overdue() {
304                        warn!("The provided license has expired and reached its maximum overdue time, please contact sales<at>defguard.net");
305                    } else {
306                        warn!("The provided license is about to expire and requires a renewal. An automatic renewal process will attempt to renew the license soon. Alternatively, automatic renewal attempt will be also performed at the next defguard start.");
307                    }
308                }
309
310                if !license.subscription && license.is_expired() {
311                    warn!("The provided license is not a subscription and has expired, please contact sales<at>defguard.net");
312                }
313
314                Ok(license)
315            }
316            Err(_) => Err(LicenseError::SignatureMismatch),
317        }
318    }

320    /// Get the key from the database
321    fn get_key() -> Option<String> {
322        let settings = Settings::get_current_settings();
323        settings.license.filter(|key| !key.is_empty())
324    }

326    /// Create the license object based on the license key stored in the database.
327    /// Automatically decodes and deserializes the keys and verifies the signature.
328    pub fn load() -> Result<Option<License>, LicenseError> {
329        if let Some(key) = Self::get_key() {
330            Ok(Some(Self::from_base64(&key)?))
331        } else {
332            debug!("No license key found in the database");
333            Ok(None)
334        }
335    }
336
337    /// Try to load the license from the database, if the license requires a renewal, try to renew it.
338    /// If the renewal fails, it will return the old license for the renewal service to renew it later.
339    pub async fn load_or_renew(pool: &PgPool) -> Result<Option<License>, LicenseError> {
340        match Self::load()? {
341            Some(license) => {
342                if license.requires_renewal() {
343                    if license.is_max_overdue() {
344                        Err(LicenseError::LicenseExpired)
345                    } else {
346                        info!("License requires renewal, trying to renew it...");
347                        match renew_license().await {
348                            Ok(new_key) => {
349                                let new_license = License::from_base64(&new_key)?;
350                                save_license_key(pool, &new_key).await?;
351                                info!("Successfully renewed and loaded the license, new license key saved to the database");
352                                Ok(Some(new_license))
353                            }
354                            Err(err) => {
355                                error!("Failed to renew the license: {err}");
356                                Ok(Some(license))
357                            }
358                        }
359                    }
360                } else {
361                    info!("Successfully loaded the license from the database.");
362                    Ok(Some(license))
363                }
364            }
365            None => Ok(None),
366        }
367    }
368
369    /// Checks whether the license is past its expiry date (`valid_until` field)
370    ///
371    /// NOTE: license should be considered valid for an additional period of `MAX_OVERDUE_TIME`.
372    /// If you want to check if the license reached this point, use `is_max_overdue` instead.
373    #[must_use]
374    pub fn is_expired(&self) -> bool {
375        match self.valid_until {
376            Some(time) => time < Utc::now(),
377            None => false,
378        }
379    }
380
381    /// Checks how much time has left until the `valid_until` time.
382    #[must_use]
383    pub fn time_left(&self) -> Option<TimeDelta> {
384        self.valid_until.map(|time| time - Utc::now())
385    }
386
387    /// Gets the time the license is past its expiry date.
388    /// If the license doesn't have a `valid_until` field, it will return 0.
389    #[must_use]
390    pub fn time_overdue(&self) -> TimeDelta {
391        match self.valid_until {
392            Some(time) => {
393                let delta = Utc::now() - time;
394                if delta <= TimeDelta::zero() {
395                    TimeDelta::zero()
396                } else {
397                    delta
398                }
399            }
400            None => TimeDelta::zero(),
401        }
402    }
403
404    /// Checks whether we should try to renew the license.
405    #[must_use]
406    pub fn requires_renewal(&self) -> bool {
407        if self.subscription {
408            if let Some(remaining) = self.time_left() {
409                remaining <= RENEWAL_TIME
410            } else {
411                false
412            }
413        } else {
414            false
415        }
416    }
417
418    /// Checks if the license has reached its maximum overdue time.
419    #[must_use]
420    pub fn is_max_overdue(&self) -> bool {
421        if self.subscription {
422            self.time_overdue() > MAX_OVERDUE_TIME
423        } else {
424            // Non-subscription licenses are considered expired immediately, no grace period is required
425            self.is_expired()
426        }
427    }
428}

430/// Exchange the currently stored key for a new one from the license server.
431///
432/// Doesn't update the cached license, nor does it save the new key in the database.
433async fn renew_license() -> Result<String, LicenseError> {
434    debug!("Exchanging license for a new one...");
435    let Some(old_license_key) = Settings::get_current_settings().license else {
436        return Err(LicenseError::LicenseNotFound);
437    };
438
439    let client = reqwest::Client::new();
440
441    let request_body = RefreshRequestResponse {
442        key: old_license_key,
443    };
444
445    let new_license_key =
446        match client
447            .post(LICENSE_SERVER_URL)
448            .json(&request_body)
449            .header(reqwest::header::USER_AGENT, format!("DefGuard/{VERSION}"))
450            .timeout(Duration::from_secs(10))
451            .send()
452            .await
453        {
454            Ok(response) => match response.status() {
455                reqwest::StatusCode::OK => {
456                    let response: RefreshRequestResponse = response.json().await.map_err(|err| {
457                    error!("Failed to parse the response from the license server while trying to \
458                        renew the license: {err:?}");
459                    LicenseError::LicenseServerError(err.to_string())
460                })?;
461                    response.key
462                }
463                status => {
464                    let status_message = response.text().await.unwrap_or_default();
465                    let message = format!(
466                        "Failed to renew the license, the license server returned a status code \
467                    {status} with error: {status_message}"
468                    );
469                    return Err(LicenseError::LicenseServerError(message));
470                }
471            },
472            Err(err) => {
473                return Err(LicenseError::LicenseServerError(err.to_string()));
474            }
475        };
476
477    info!("Successfully exchanged the license for a new one");
478
479    Ok(new_license_key)
480}

482/// Helper function used to check if the cached license should be considered valid.
483/// As the license is often passed around in the form of `Option<License>`, this function takes care
484/// of the whole logic related to checking whether the license is even present in the first place.
485///
486/// This function checks the following two things:
487/// 1. Does the cached license exist
488/// 2. Is the cached license past its maximum expiry date
489/// 3. Does current object count exceed license limits
490pub(crate) fn validate_license(
491    license: Option<&License>,
492    counts: &Counts,
493) -> Result<(), LicenseError> {
494    debug!("Validating if the license is present, not expired and not exceeding limits...");
495    match license {
496        Some(license) => {
497            if license.is_max_overdue() {
498                return Err(LicenseError::LicenseExpired);
499            }
500            if counts.is_over_license_limits(license) {
501                return Err(LicenseError::LicenseLimitsExceeded);
502            }
503            Ok(())
504        }
505        None => Err(LicenseError::LicenseNotFound),
506    }
507}

509/// Helper function to save the license key string in the database
510async fn save_license_key(pool: &PgPool, key: &str) -> Result<(), LicenseError> {
511    debug!("Saving the license key to the database...");
512    let mut settings = Settings::get_current_settings();
513    settings.license = Some(key.to_string());
514    update_current_settings(pool, settings).await?;
515
516    info!("Successfully saved license key to the database.");
517
518    Ok(())
519}

521/// Helper function to update the in-memory cached license mutex.
522pub fn update_cached_license(key: Option<&str>) -> Result<(), LicenseError> {
523    debug!("Updating the cached license information with the provided key...");
524    let license = if let Some(key) = key {
525        // Handle the Some("") case
526        if key.is_empty() {
527            debug!("The new license key is empty, clearing the cached license");
528            None
529        } else {
530            debug!("A new license key has been provided, decoding and validating it...");
531            Some(License::from_base64(key)?)
532        }
533    } else {
534        None
535    };
536    set_cached_license(license);
537
538    info!("Successfully updated the cached license information.");
539
540    Ok(())
541}
542/// Amount of time before the license expiry date we should start the renewal attempts.
543const RENEWAL_TIME: TimeDelta = TimeDelta::hours(24);
544const MAX_OVERDUE_TIME: TimeDelta = TimeDelta::days(14);
545
546#[instrument(skip_all)]
547pub async fn run_periodic_license_check(pool: &PgPool) -> Result<(), LicenseError> {
548    let config = server_config();
549    let mut check_period: Duration = *config.check_period;
550    info!(
551        "Starting periodic license renewal check every {}",
552        format_duration(check_period)
553    );
554    loop {
555        debug!("Checking the license status...");
556        // Check if the license is present in the mutex, if not skip the check
557        if get_cached_license().is_none() {
558            debug!("No license found, skipping license check");
559            sleep(*config.check_period_no_license).await;
560            continue;
561        }
562
563        // Check if the license requires renewal, uses the cached value to be more efficient
564        // The block here is to avoid holding the lock through awaits
565        //
566        // Multiple locks here may cause a race condition if the user decides to update the license key
567        // while the renewal is in progress. However this seems like a rare case and shouldn't be very problematic.
568        let requires_renewal = {
569            let license = get_cached_license();
570            debug!("Checking if the license {license:?} requires a renewal...");
571
572            if let Some(license) = license.as_ref() {
573                if license.requires_renewal() {
574                    // check if we are pass the maximum expiration date, after which we don't
575                    // want to try to renew the license anymore
576                    if license.is_max_overdue() {
577                        check_period = *config.check_period;
578                        warn!("Your license has expired and reached its maximum overdue date, please contact sales at sales<at>defguard.net");
579                        debug!("Changing check period to {}", format_duration(check_period));
580                        false
581                    } else {
582                        debug!("License requires renewal, as it is about to expire and is not past the maximum overdue time");
583                        true
584                    }
585                } else {
586                    // This if is only for logging purposes, to provide more detailed information
587                    if license.subscription {
588                        debug!("License doesn't need to be renewed yet, skipping renewal check");
589                    } else {
590                        debug!("License is not a subscription, skipping renewal check");
591                    }
592                    false
593                }
594            } else {
595                debug!("No license found, skipping license check");
596                false
597            }
598        };
599
600        if requires_renewal {
601            info!("License requires renewal, renewing license...");
602            check_period = *config.check_period_renewal_window;
603            debug!("Changing check period to {}", format_duration(check_period));
604            match renew_license().await {
605                Ok(new_license_key) => match save_license_key(pool, &new_license_key).await {
606                    Ok(()) => {
607                        update_cached_license(Some(&new_license_key))?;
608                        check_period = *config.check_period;
609                        debug!("Changing check period to {}", format_duration(check_period));
610                        info!("Successfully renewed the license");
611                    }
612                    Err(err) => {
613                        error!("Couldn't save the newly fetched license key to the database, error: {}", err);
614                    }
615                },
616                Err(err) => {
617                    warn!(
618                        "Failed to renew the license: {err}. Retrying in {}",
619                        format_duration(check_period)
620                    );
621                }
622            }
623        }
624
625        sleep(check_period).await;
626    }
627}
