1use sqlx::PgPool;
2use tonic::Status;

4use crate::{

13pub struct PollingServer {
14    pool: PgPool,
15}
16
17impl PollingServer {
18    #[must_use]
19    pub fn new(pool: PgPool) -> Self {
20        Self { pool }
21    }
22
23    /// Checks validity of polling session
24    async fn validate_session(&self, token: &str) -> Result<PollingToken<Id>, Status> {
25        debug!("Validating polling token. Token: {token}");
26
27        // Polling service is enterprise-only, check the lincense
28        if !is_enterprise_enabled() {
29            debug!("Instance has enterprise features disabled, denying instance polling info");
30            return Err(Status::failed_precondition("no valid license"));
31        }
32
33        // Validate the token
34        let Some(token) = PollingToken::find(&self.pool, token).await.map_err(|err| {
35            error!("Failed to retrieve token: {err}");
36            Status::internal("failed to retrieve token")
37        })?
38        else {
39            error!("Invalid token {token:?}");
40            return Err(Status::permission_denied("invalid token"));
41        };
42
43        // Polling tokens are valid indefinitely
44        debug!("Token validation successful {token:?}.");
45
46        Ok(token)
47    }
48
49    /// Prepares instance info for polling requests. Enterprise only.
50    #[instrument(skip_all)]
51    pub async fn info(&self, request: InstanceInfoRequest) -> Result<InstanceInfoResponse, Status> {
52        trace!("Polling info start");
53        let token = self.validate_session(&request.token).await?;
54        let Some(device) = Device::find_by_id(&self.pool, token.device_id)
55            .await
56            .map_err(|err| {
57                error!("Failed to retrieve device id {}: {err}", token.device_id);
58                Status::internal("failed to retrieve device")
59            })?
60        else {
61            error!("Device id {} not found", token.device_id);
62            return Err(Status::internal("device not found"));
63        };
64        debug!("Polling info for device: {}", device.wireguard_pubkey);
65
66        // Ensure user is active
67        let Some(user) = User::find_by_device_id(&self.pool, device.id)
68            .await
69            .map_err(|err| {
70                error!("Failed to retrieve user for device id {}: {err}", device.id);
71                Status::internal("failed to retrieve user")
72            })?
73        else {
74            error!("User for device id {} not found", device.id);
75            return Err(Status::internal("user not found"));
76        };
77        if !user.is_active {
78            warn!(
79                "Denying polling info for inactive user {}({})",
80                user.username, user.id
81            );
82            return Err(Status::permission_denied("user inactive"));
83        }
84
85        // Build and return polling info.
86        let device_config = build_device_config_response(&self.pool, device, None).await?;
87
88        Ok(InstanceInfoResponse {
89            device_config: Some(device_config),
90        })
91    }
92}
