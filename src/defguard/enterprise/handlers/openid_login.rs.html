42/// Prune the given username from illegal characters in accordance with the following rules:
43///
44/// To enable LDAP sync usernames need to avoid reserved characters.
45/// Username requirements:
46/// - 64 characters long
47/// - only lowercase or uppercase latin alphabet letters (A-Z, a-z) and digits (0-9)
48/// - starts with non-special character
49/// - only special characters allowed: . - _
50/// - no whitespaces
51pub fn prune_username(username: &str, handling: OpenidUsernameHandling) -> String {
52    let mut result = username.to_string();
53
54    // Go through the string and remove any non-alphanumeric characters at the beginning
55    result = result
56        .trim_start_matches(|c: char| !c.is_ascii_alphanumeric())
57        .to_string();
58
59    let is_char_valid = |c: char| c.is_ascii_alphanumeric() || c == '.' || c == '-' || c == '_';
60
61    match handling {
62        OpenidUsernameHandling::RemoveForbidden => {
63            result.retain(&is_char_valid);
64        }
65        OpenidUsernameHandling::ReplaceForbidden => {
66            result = result
67                .chars()
68                .map(|c| if is_char_valid(c) { c } else { '_' })
69                .collect();
70        }
71        OpenidUsernameHandling::PruneEmailDomain => {
72            if let Some(at_index) = result.find('@') {
73                result.truncate(at_index);
74            }
75            result = result
76                .chars()
77                .map(|c| if is_char_valid(c) { c } else { '_' })
78                .collect();
79        }
80    }
81
82    result.truncate(64);
83
84    result
85}

98async fn get_provider_metadata(url: &str) -> Result<CoreProviderMetadata, WebError> {
99    let issuer_url = IssuerUrl::new(url.to_string()).map_err(|err| {
100        WebError::BadRequest(format!(
101            "Failed to create issuer URL from the provided URL: {url}. Error details: {err:?}",
102        ))
103    })?;
104    let async_http_client = get_async_http_client().await?;
105    // Discover the provider metadata based on a known base issuer URL
106    // The url should be in the form of e.g. https://accounts.google.com
107    // The url shouldn't contain a .well-known part, it will be added automatically
108    match CoreProviderMetadata::discover_async(issuer_url, &async_http_client).await {
109        Ok(provider_metadata) => Ok(provider_metadata),
110        Err(err) => {
111            Err(WebError::Authorization(format!(
112                "Failed to discover provider metadata, make sure the provider's URL is correct: {url}. Error details: {err:?}",
113            )))
114        }
115    }
116}

118/// Build OpenID Connect client.
119/// `url`: redirect/callback URL
120pub(crate) async fn make_oidc_client(
121    url: Url,
122    provider: &OpenIdProvider<Id>,
123) -> Result<
124    (
125        ClientId,
126        CoreClient<
127            EndpointSet,
128            EndpointNotSet,
129            EndpointNotSet,
130            EndpointNotSet,
131            EndpointMaybeSet,
132            EndpointMaybeSet,
133        >,
134    ),
135    WebError,
136> {
137    let provider_metadata = get_provider_metadata(&provider.base_url).await?;
138    let client_id = ClientId::new(provider.client_id.to_string());
139    let client_secret = ClientSecret::new(provider.client_secret.to_string());
140    let core_client = CoreClient::from_provider_metadata(
141        provider_metadata,
142        client_id.clone(),
143        Some(client_secret),
144    )
145    .set_redirect_uri(RedirectUrl::from_url(url));
146
147    Ok((client_id, core_client))
148}

150/// Get or create `User` from OpenID claims.
151pub(crate) async fn user_from_claims(
152    pool: &PgPool,
153    nonce: Nonce,
154    code: AuthorizationCode,
155    callback_url: Url,
156) -> Result<User<Id>, WebError> {
157    let Some(provider) = OpenIdProvider::get_current(pool).await? else {
158        return Err(WebError::ObjectNotFound(
159            "OpenID provider not set".to_string(),
160        ));
161    };
162    let (client_id, core_client) = make_oidc_client(callback_url, &provider).await?;
163    let async_http_client = get_async_http_client().await?;
164    // Exchange code for ID token.
165    let token_response = match core_client
166        .exchange_code(code)
167        .map_err(|err| {
168            error!("Failed to exchange code for ID token: {err:?}");
169            WebError::Http(StatusCode::INTERNAL_SERVER_ERROR)
170        })?
171        .request_async(&async_http_client)
172        .await
173    {
174        Ok(token) => token,
175        Err(err) => {
176            return Err(WebError::Authorization(format!(
177                "Failed to exchange code for ID token; OpenID provider error: {err:?}"
178            )));
179        }
180    };
181    let Some(id_token) = token_response.extra_fields().id_token() else {
182        return Err(WebError::Authorization(
183            "Server did not return an ID token".to_string(),
184        ));
185    };
186
187    let access_token = token_response.access_token();
188
189    // Verify ID token against the nonce value received in the callback.
190    let token_verifier = core_client
191        .id_token_verifier()
192        .require_audience_match(false);
193    // claims = user attributes
194    let token_claims = match id_token.claims(&token_verifier, &nonce) {
195        Ok(claims) => claims,
196        Err(error) => {
197            return Err(WebError::Authorization(format!(
198                "Failed to verify ID token, error: {error:?}",
199            )));
200        }
201    };
202    // Custom `aud` (audience) verfication. According to OpenID specification:
203    // "The Client MUST validate that the aud (audience) Claim contains its client_id value
204    // registered at the Issuer identified by the iss (issuer) Claim as an audience. The ID
205    // Token MUST be rejected if the ID Token does not list the Client as a valid audience,
206    // or if it contains additional audiences not trusted by the Client."
207    // But some providers, like Zitadel, send additional values in `aud`, so allow that.
208    let audiences = token_claims.audiences();
209    if !audiences.iter().any(|aud| **aud == *client_id) {
210        return Err(WebError::Authorization(format!(
211            "Invalid OpenID claims: 'aud' must contain '{}' (found audiences: {})",
212            client_id.as_str(),
213            audiences
214                .iter()
215                .map(|aud| aud.to_string())
216                .collect::<Vec<_>>()
217                .join(", ")
218        )));
219    };
220    if audiences.len() > 1 {
221        warn!(
222            "OpenID claims: 'aud' should not contain these additional fields {}",
223            audiences
224                .iter()
225                .filter(|&aud| **aud != *client_id)
226                .map(|aud| aud.to_string())
227                .collect::<Vec<_>>()
228                .join(", ")
229        );
230    }
231
232    // Only email and username is required for user lookup and login
233    let email = token_claims.email().ok_or(WebError::BadRequest(
234        "Email not found in the information returned from provider. Make sure your provider is \
235        configured correctly and that you have granted the necessary permissions to retrieve \
236        such information."
237            .to_string(),
238    ))?;
239
240    // Try to get the username from the preferred_username claim.
241    // If it's not there, extract it from email.
242    let username = if let Some(username) = token_claims.preferred_username() {
243        debug!("Preferred username {username:?} found in the claims, extracting username from it.");
244        username
245    } else {
246        debug!("Preferred username not found in the claims, extracting from email address.");
247        // Extract the username from the email address
248        let username = email.split('@').next().ok_or(WebError::BadRequest(
249            "Failed to extract username from email address".to_string(),
250        ))?;
251        debug!("Username extracted from email ({email:?}): {username})");
252        username
253    };
254    let settings = Settings::get_current_settings();
255
256    let username = prune_username(username, settings.openid_username_handling);
257    // Check if the username is valid just in case, not everything can be handled by the pruning.
258    check_username(&username)?;
259
260    // Get the *sub* claim from the token.
261    let sub = token_claims.subject().to_string();
262
263    // Handle logging in or creating user.
264    let user = match User::find_by_sub(pool, &sub)
265        .await
266        .map_err(|err| WebError::Authorization(err.to_string()))?
267    {
268        Some(user) => {
269            debug!(
270                "User {} is trying to log in using an OpenID provider.",
271                user.username
272            );
273            // Make sure the user is not disabled
274            if !user.is_active {
275                debug!("User {} tried to log in, but is disabled", user.username);
276                return Err(WebError::Authorization("User is disabled".into()));
277            }
278            user
279        }
280        None => {
281            if let Some(mut user) = User::find_by_email(pool, email).await? {
282                if !user.is_active {
283                    debug!("User {} tried to log in, but is disabled", user.username);
284                    return Err(WebError::Authorization("User is disabled".into()));
285                }
286                // User with the same email already exists, merge the accounts
287                info!(
288                    "User with email address {} is logging in through OpenID Connect for the \
289                    first time and we've found an existing account with the same email \
290                    address. Merging accounts.",
291                    user.email
292                );
293                user.openid_sub = Some(sub);
294                user.save(pool).await?;
295                user
296            } else {
297                // Check if the user should be created if they don't exist (default: true)
298                if !settings.openid_create_account {
299                    warn!(
300                        "User with email address {} is trying to log in through OpenID Connect \
301                        for the first time, but the account creation is disabled. An enrollment \
302                        should performed.",
303                        email.as_str()
304                    );
305                    return Err(WebError::Authorization(
306                        "User not found and the automatic account creation is disabled. \
307                        Enable it or create the user."
308                            .into(),
309                    ));
310                }
311
312                info!(
313                    "User {username} is logging in through OpenID Connect for the first time and \
314                    there is no account with the same email address ({}). Creating a new account.",
315                    email.as_str()
316                );
317                // Check if user with the same username already exists (usernames are unique).
318                if User::find_by_username(pool, &username).await?.is_some() {
319                    return Err(WebError::Authorization(format!(
320                        "User with username {username} already exists"
321                    )));
322                }
323
324                // Extract all necessary information from the token or call the userinfo endpoint
325                let given_name = token_claims
326                    .given_name()
327                    // 'None' gets you the default value from a localized claim.
328                    //  Otherwise you would need to pass a locale.
329                    .and_then(|claim| claim.get(None));
330                let family_name = token_claims.family_name().and_then(|claim| claim.get(None));
331                let phone = token_claims.phone_number();
332
333                let userinfo_response: CoreUserInfoClaims;
334                let (given_name, family_name, phone) = if let (
335                    Some(given_name),
336                    Some(family_name),
337                    phone,
338                ) = (given_name, family_name, phone)
339                {
340                    debug!("Given name and family name found in the claims for user {username}.");
341                    (given_name, family_name, phone)
342                } else {
343                    debug!(
344                        "Given name or family name not found in the claims for user {username}, trying to get them \
345                        from the user info endpoint. Current values: given_name: {given_name:?}, family_name: {family_name:?}, phone: {phone:?}"
346                    );
347
348                    let async_http_client = get_async_http_client().await?;
349
350                    let retrieval_error = "Failed to retrieve given name and family name from provider's userinfo endpoint. \
351                        Make sure you have configured your provider correctly and that you have granted the \
352                        necessary permissions to retrieve such information from the token or the userinfo endpoint.";
353                    userinfo_response = core_client
354                        .user_info(access_token.clone(), Some(token_claims.subject().clone()))
355                        .map_err(
356                            |err| {
357                                error!(
358                                    "Failed to get family name and given name from provider's userinfo endpoint, they may not support this. Error details: {err:?}",
359                                );
360
361                                WebError::BadRequest(
362                                    retrieval_error.into(),
363                                )
364                            }
365                        )?
366                        .request_async(&async_http_client)
367                        .await
368                        .map_err(
369                            |err| {
370                                error!(
371                                    "Failed to get family name and given name from provider's userinfo endpoint. Error details: {err:?}",
372                                );
373
374                                WebError::BadRequest(
375                                    retrieval_error.into(),
376                                )
377                            }
378                        )?;
379
380                    let claim_error = |claim_name: &str| {
381                        format!(
382                            "Failed to retrieve {claim_name} from provider's userinfo endpoint and the ID token. \
383                            Make sure you have configured your provider correctly and that you have \
384                            granted the necessary permissions to retrieve such information from the token or the userinfo endpoint.",
385                        )
386                    };
387                    let given_name = userinfo_response
388                        .given_name()
389                        .and_then(|claim| claim.get(None))
390                        .ok_or(WebError::BadRequest(claim_error("given name")))?;
391                    let family_name = userinfo_response
392                        .family_name()
393                        .and_then(|claim| claim.get(None))
394                        .ok_or(WebError::BadRequest(claim_error("family name")))?;
395                    let phone = userinfo_response.phone_number();
396
397                    debug!(
398                        "Given name and family name successfully retrieved from the user info endpoint for user {username}."
399                    );
400
401                    (given_name, family_name, phone)
402                };
403
404                let mut user = User::new(
405                    username.to_string(),
406                    None,
407                    family_name.to_string(),
408                    given_name.to_string(),
409                    email.to_string(),
410                    phone.map(|v| v.to_string()),
411                );
412                user.openid_sub = Some(sub);
413                user.save(pool).await?
414            }
415        }
416    };
417
418    update_counts(pool).await?;
419    Ok(user)
420}

422pub(crate) async fn get_auth_info(
423    _license: LicenseInfo,
424    private_cookies: PrivateCookieJar,
425    State(appstate): State<AppState>,
426) -> Result<(PrivateCookieJar, ApiResponse), WebError> {
427    let provider = OpenIdProvider::get_current(&appstate.pool).await?;
428    let Some(provider) = provider else {
429        return Err(WebError::ObjectNotFound(
430            "OpenID provider not set".to_string(),
431        ));
432    };
433
434    let config = server_config();
435    let (_client_id, client) = make_oidc_client(config.callback_url(), &provider).await?;
436
437    // Generate the redirect URL and the values needed later for callback authenticity verification
438    let (authorize_url, csrf_state, nonce) = client
439        .authorize_url(
440            CoreAuthenticationFlow::AuthorizationCode,
441            CsrfToken::new_random,
442            Nonce::new_random,
443        )
444        .add_scope(Scope::new("email".into()))
445        .add_scope(Scope::new("profile".into()))
446        .url();
447
448    let cookie_domain = config
449        .cookie_domain
450        .as_ref()
451        .expect("Cookie domain not found");
452    let nonce_cookie = Cookie::build((NONCE_COOKIE_NAME, nonce.secret().clone()))
453        .domain(cookie_domain)
454        .path("/api/v1/openid/callback")
455        .http_only(true)
456        .same_site(SameSite::Strict)
457        .secure(!config.cookie_insecure)
458        .max_age(COOKIE_MAX_AGE)
459        .build();
460    let csrf_cookie = Cookie::build((CSRF_COOKIE_NAME, csrf_state.secret().clone()))
461        .domain(cookie_domain)
462        .path("/api/v1/openid/callback")
463        .http_only(true)
464        .same_site(SameSite::Strict)
465        .secure(!config.cookie_insecure)
466        .max_age(COOKIE_MAX_AGE)
467        .build();
468    let private_cookies = private_cookies.add(nonce_cookie).add(csrf_cookie);
469
470    Ok((
471        private_cookies,
472        ApiResponse {
473            json: json!(
474                {
475                    "url": authorize_url,
476                    "button_display_name": provider.display_name
477                }
478            ),
479            status: StatusCode::OK,
480        },
481    ))
482}

490pub(crate) async fn auth_callback(
491    _license: LicenseInfo,
492    cookies: CookieJar,
493    mut private_cookies: PrivateCookieJar,
494    user_agent: TypedHeader<UserAgent>,
495    InsecureClientIp(insecure_ip): InsecureClientIp,
496    State(appstate): State<AppState>,
497    Json(payload): Json<AuthenticationResponse>,
498) -> Result<(CookieJar, PrivateCookieJar, ApiResponse), WebError> {
499    debug!("Auth callback received, logging in user...");
500
501    // Get the nonce and CSRF cookies, we need them to verify the callback
502    let cookie_nonce = private_cookies
503        .get(NONCE_COOKIE_NAME)
504        .ok_or(WebError::Authorization("Nonce cookie not found".into()))?
505        .value_trimmed()
506        .to_string();
507    let cookie_csrf = private_cookies
508        .get(CSRF_COOKIE_NAME)
509        .ok_or(WebError::BadRequest("CSRF cookie not found".into()))?
510        .value_trimmed()
511        .to_string();
512
513    // Verify the CSRF token
514    if payload.state.secret() != &cookie_csrf {
515        return Err(WebError::Authorization("CSRF token mismatch".into()));
516    };
517
518    private_cookies = private_cookies
519        .remove(Cookie::from(NONCE_COOKIE_NAME))
520        .remove(Cookie::from(CSRF_COOKIE_NAME));
521
522    let config = server_config();
523    let mut user = user_from_claims(
524        &appstate.pool,
525        Nonce::new(cookie_nonce),
526        payload.code,
527        config.callback_url(),
528    )
529    .await?;
530
531    let (session, user_info, mfa_info) = create_session(
532        &appstate.pool,
533        &appstate.mail_tx,
534        insecure_ip,
535        user_agent.as_str(),
536        &mut user,
537    )
538    .await?;
539
540    let max_age = Duration::seconds(config.auth_cookie_timeout.as_secs() as i64);
541    let cookie_domain = config
542        .cookie_domain
543        .as_ref()
544        .expect("Cookie domain not found");
545    let auth_cookie = Cookie::build((SESSION_COOKIE_NAME, session.id))
546        .domain(cookie_domain)
547        .path("/")
548        .http_only(true)
549        .secure(!config.cookie_insecure)
550        .same_site(SameSite::Lax)
551        .max_age(max_age);
552    let cookies = cookies.add(auth_cookie);
553
554    // The user may not be yet authorized (pre-MFA) but syncing their groups should be fine here, since he already managed to login through the provider.
555    // There is currently no other way to sync the groups for the MFA enabled user logging in through the provider without firing it
556    // on every login attempt, even for standard, non-provider users.
557    if let Err(err) =
558        sync_user_groups_if_configured(&user, &appstate.pool, &appstate.wireguard_tx).await
559    {
560        error!(
561            "Failed to sync user groups for user {} with the directory while the user was trying to login in through an external provider: {err:?}",
562            user.username
563        );
564    } else {
565        ldap_update_user_state(&mut user, &appstate.pool).await;
566    }
567
568    if let Some(mfa_info) = mfa_info {
569        return Ok((
570            cookies,
571            private_cookies,
572            ApiResponse {
573                json: json!(mfa_info),
574                status: StatusCode::CREATED,
575            },
576        ));
577    }
578
579    if let Some(user_info) = user_info {
580        let url = if let Some(openid_cookie) = private_cookies.get(SIGN_IN_COOKIE_NAME) {
581            debug!("Found OpenID session cookie, returning the redirect URL stored in it.");
582            let url = openid_cookie.value().to_string();
583            private_cookies = private_cookies.remove(openid_cookie);
584            Some(url)
585        } else {
586            debug!("No OpenID session found, proceeding with login to Defguard.");
587            None
588        };
589
590        Ok((
591            cookies,
592            private_cookies,
593            ApiResponse {
594                json: json!(AuthResponse {
595                    user: user_info,
596                    url
597                }),
598                status: StatusCode::OK,
599            },
600        ))
601    } else {
602        unimplemented!("Impossible to get here");
603    }
604}

611    fn test_prune_username() {
612        // Test RemoveForbidden handling
613        let handling_remove = OpenidUsernameHandling::RemoveForbidden;
614        assert_eq!(prune_username("zenek", handling_remove), "zenek");
615        assert_eq!(prune_username("zenek34", handling_remove), "zenek34");
616        assert_eq!(prune_username("zenek@34", handling_remove), "zenek34");
617        assert_eq!(prune_username("first.last", handling_remove), "first.last");
618        assert_eq!(prune_username("__zenek__", handling_remove), "zenek__");
619        assert_eq!(prune_username("zenek?", handling_remove), "zenek");
620        assert_eq!(prune_username("zenek!", handling_remove), "zenek");
621        assert_eq!(
622            prune_username(
623                "averylongnameeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee",
624                handling_remove
625            ),
626            "averylongnameeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee"
627        );
628        assert_eq!(prune_username("", handling_remove), "");
629        assert_eq!(prune_username("!@#$%^&*()", handling_remove), "");
630        assert_eq!(prune_username("!zenek", handling_remove), "zenek");
631        assert_eq!(prune_username("...zenek", handling_remove), "zenek");
632
633        // Test ReplaceForbidden handling

635        assert_eq!(prune_username("zenek", handling_replace), "zenek");

639        assert_eq!(prune_username("__zenek__", handling_replace), "zenek__");
640        assert_eq!(prune_username("zenek?", handling_replace), "zenek_");
641        assert_eq!(prune_username("zenek!", handling_replace), "zenek_");
642        assert_eq!(
643            prune_username(
644                "averylongnameeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee",
645                handling_replace
646            ),
647            "averylongnameeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee"
648        );
